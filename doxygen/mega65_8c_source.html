142 static inline void nmi_set ( int level, int mask )
143 {
144  // NMI is a low active _EDGE_ triggered 65xx input ... In my emulator though, the signal
145  // is "high active", and also we must form the "edge" ourselves from "level". NMI level is
146  // set as a 2bit number, on bit 0, CIA2, on bit 1, keyboard RESTORE key. Thus having zero
147  // value for level means (in the emu!) that not RESTORE key is pressed neither CIA2 has active
148  // IRQ output, non-zero value means some activation. Well, if I am not confused enough here,
149  // this should mean that nmi_level zero->non-zero transit should produce the edge (which should
150  // be the falling edge in the real hardware anyway ... but the rising here. heh, I should follow
151  // the signal level of the hardware in my emulator, honestly ...)
152  int nmi_new_level;
153  if (level)
154  nmi_new_level = nmi_level | mask;
155  else
156  nmi_new_level = nmi_level & (~mask);
157  if ((!nmi_level) && nmi_new_level) {
158  DEBUG("NMI edge is emulated towards the CPU (%d->%d)" NL, nmi_level, nmi_new_level);
159  cpu65.nmiEdge = 1; // the "NMI edge" trigger is deleted by the CPU emulator itself (it's not a level trigger)
160  }
161  nmi_level = nmi_new_level;
162 }

218 static int load_memory_preinit_cache ( int to_prezero, const char *config_name, const char *description, Uint8 *memory_pointer, int size )
219 {
220  const char *p = xemucfg_get_str(config_name);
221  if (to_prezero) {
222  DEBUGPRINT("BIN: config \"%s\" as \"%s\": pre-zero policy, clearing memory content." NL, config_name, description);
223  memset(memory_pointer, 0, size);
224  } else
225  DEBUGPRINT("BIN: config \"%s\" as \"%s\": no pre-zero policy, using some (possible) built-in default content." NL, config_name, description);
226  if (p) {
227  if (!strcmp(p, "-")) {
228  DEBUGPRINT("BIN: config \"%s\" as \"%s\": has option override policy to force and use pre-zeroed content." NL, config_name, description);
229  memset(memory_pointer, 0, size);
230  return 0;
231  } else {
232  DEBUGPRINT("BIN: config \"%s\" as \"%s\": has option override, trying to load content: \"%s\"." NL, config_name, description, p);
233  return xemu_load_file(p, memory_pointer, size, size, description);
234  }
235  } else {
236  DEBUGPRINT("BIN: config \"%s\" as \"%s\": has no option override, using the previously stated policy." NL, config_name, description);
237  return 0;
238  }

272 static void mega65_init ( int sid_cycles_per_sec, int sound_mix_freq )
273 {
274  const char *p;
275 #ifdef AUDIO_EMULATION
276  SDL_AudioSpec audio_want, audio_got;
277 #endif
278  hypervisor_debug_init(xemucfg_get_str("kickuplist"), xemucfg_get_bool("hyperdebug"), xemucfg_get_bool("hyperserialascii"));
279  hid_init(
280  c64_key_map,
281  VIRTUAL_SHIFT_POS,
282  SDL_ENABLE // joy HID events enabled
283  );
284 #ifdef HID_KBD_MAP_CFG_SUPPORT
285  hid_keymap_from_config_file(xemucfg_get_str("keymap"));
286 #endif
287  joystick_emu = 1;
288  nmi_level = 0;
289  // *** FPGA switches ...
290  do {
291  int switches[16], r = xemucfg_integer_list_from_string(xemucfg_get_str("fpga"), switches, 16, ",");
292  if (r < 0)
293  FATAL("Too many FPGA switches specified for option 'fpga'");
294  while (r--) {
295  DEBUGPRINT("FPGA switch is turned on: #%d" NL, switches[r]);
296  if (switches[r] < 0 || switches[r] > 15)
297  FATAL("Invalid switch sepcifictation for option 'fpga': %d", switches[r]);
298  fpga_switches |= 1 << (switches[r]);
299  }
300  } while (0);
301  // *** Init memory space
302  // *** Pre-init some memory areas from built-in resources, or by loaded binaries, see the
303  // previous part of the source here and refill_memory_from_preinit_cache() function definitation
304  // Notes about the used "built-in" contents as "binary blobs":
305  // C65 ROM image is not included in Xemu built-in, since it's a copyrighted, non-CPL licensed material
306  // Other "blobs" are built-in, they are extracted from mega65-core project (unmodified) and it's also a GPL
307  // project, with all sources available on-line, thus no licensing/copyright problem here.
308  // For mega65-core source, visit https://github.com/MEGA65/mega65-core
309  // For C000 utilties: mega65-core currently under reorganization, no C000 utilties are provided.
310  force_external_rom = ((load_memory_preinit_cache(1, "loadrom", "C65 ROM image", rom_init_image, sizeof rom_init_image) == (int)sizeof(rom_init_image)) && xemucfg_get_bool("forcerom"));
311  if (force_external_rom)
312  DEBUGPRINT("MEM: forcing external ROM usage (hypervisor leave memory re-fill policy)" NL);
313  else if (xemucfg_get_bool("forcerom"))
314  ERROR_WINDOW("-forcerom is ignored, because no -loadrom <filename> option was used, or it was not a succesfull load operation at least");
315  load_memory_preinit_cache(0, "loadcram", "CRAM utilities", meminitdata_cramutils, MEMINITDATA_CRAMUTILS_SIZE);
316  load_memory_preinit_cache(0, "loadbanner", "M65 logo", meminitdata_banner, MEMINITDATA_BANNER_SIZE);
317  load_memory_preinit_cache(1, "loadc000", "C000 utilities", c000_init_image, sizeof c000_init_image);
318  if (load_memory_preinit_cache(0, "kickup", "M65 kickstart", meminitdata_kickstart, MEMINITDATA_KICKSTART_SIZE) != MEMINITDATA_KICKSTART_SIZE)
319  hypervisor_debug_invalidate("no kickup is loaded, built-in one does not have debug info");
320  // *** Initializes memory subsystem of Mega65 emulation itself
321  memory_init();
322  // fill the actual M65 memory areas with values managed by load_memory_preinit_cache() calls
323  // This is a separated step, to be able to call refill_memory_from_preinit_cache() later as well, in case of a "deep reset" functionality is needed for Xemu (not just CPU/hw reset),
324  // without restarting Xemu for that purpose.
325  refill_memory_from_preinit_cache();
326  // *** Image file for SDCARD support
327  if (sdcard_init(xemucfg_get_str("sdimg"), xemucfg_get_str("8"), xemucfg_get_bool("virtsd")) < 0)
328  FATAL("Cannot find SD-card image (which is a must for Mega65 emulation): %s", xemucfg_get_str("sdimg"));
329  // *** Initialize VIC4
330  vic_init();
331  // *** CIAs
332  cia_init(&cia1, "CIA-1",
333  NULL, // callback: OUTA
334  NULL, // callback: OUTB
335  NULL, // callback: OUTSR
336  cia1_in_a, // callback: INA ~ joy#2
337  cia1_in_b, // callback: INB ~ keyboard
338  NULL, // callback: INSR
339  cia1_setint_cb // callback: SETINT
340  );
341  cia_init(&cia2, "CIA-2",
342  cia2_out_a, // callback: OUTA
343  NULL, // callback: OUTB
344  NULL, // callback: OUTSR
345  cia_port_in_dummy, // callback: INA
346  NULL, // callback: INB
347  NULL, // callback: INSR
348  cia2_setint_cb // callback: SETINT ~ that would be NMI in our case
349  );
350  cia2.DDRA = 3; // Ugly workaround ... I think, SD-card setup "CRAM UTIL" (or better: the kickstart) should set this by its own. Maybe Xemu bug, maybe not?
351  // *** Initialize DMA (we rely on memory and I/O decoder provided functions here for the purpose)
352  dma_init(newhack ? DMA_FEATURE_HACK | DMA_FEATURE_DYNMODESET : xemucfg_get_num("dmarev"));
353  // Initialize FDC
354  fdc_init(disk_buffers + FD_BUFFER_POS);
355  // SIDs, plus SDL audio
356  sid_init(&sid1, sid_cycles_per_sec, sound_mix_freq);
357  sid_init(&sid2, sid_cycles_per_sec, sound_mix_freq);
358 #ifdef AUDIO_EMULATION
359  SDL_memset(&audio_want, 0, sizeof(audio_want));
360  audio_want.freq = sound_mix_freq;
361  audio_want.format = AUDIO_S16SYS; // used format by SID emulation (ie: signed short)
362  audio_want.channels = 2; // that is: stereo, for the two SIDs
363  audio_want.samples = 1024; // Sample size suggested (?) for the callback to render once
364  audio_want.callback = audio_callback; // Audio render callback function, called periodically by SDL on demand
365  audio_want.userdata = NULL; // Not used, "userdata" parameter passed to the callback by SDL
366  audio = SDL_OpenAudioDevice(NULL, 0, &audio_want, &audio_got, 0);
367  if (audio) {
368  int i;
369  for (i = 0; i < SDL_GetNumAudioDevices(0); i++)
370  DEBUG("AUDIO: audio device is #%d: %s" NL, i, SDL_GetAudioDeviceName(i, 0));
371  // Sanity check that we really got the same audio specification we wanted
372  if (audio_want.freq != audio_got.freq || audio_want.format != audio_got.format || audio_want.channels != audio_got.channels) {
373  SDL_CloseAudioDevice(audio); // forget audio, if it's not our expected format :(
374  audio = 0;
375  ERROR_WINDOW("Audio parameter mismatches.");
376  }
377  DEBUGPRINT("AUDIO: initialized (#%d), %d Hz, %d channels, %d buffer sample size." NL, audio, audio_got.freq, audio_got.channels, audio_got.samples);
378  } else
379  ERROR_WINDOW("Cannot open audio device!");
380 #endif
381  //
382 #ifdef HAS_UARTMON_SUPPORT
383  uartmon_init(xemucfg_get_str("uartmon"));
384 #endif
385  fast_mhz = xemucfg_get_num("fastclock");
386  if (fast_mhz < 3 || fast_mhz > 200) {
387  ERROR_WINDOW("Fast clock given by -fastclock switch must be between 3...200MHz. Bad value, defaulting to %dMHz", MEGA65_DEFAULT_FAST_CLOCK);
388  fast_mhz = MEGA65_DEFAULT_FAST_CLOCK;
389  }
390  sprintf(fast_mhz_in_string, "%dMHz", fast_mhz);
391  cpu_cycles_per_scanline_for_fast_mode = 64 * fast_mhz;
392  DEBUGPRINT("SPEED: fast clock is set to %dMHz, %d CPU cycles per scanline." NL, fast_mhz, cpu_cycles_per_scanline_for_fast_mode);
393  cpu65_reset(); // reset CPU (though it fetches its reset vector, we don't use that on M65, but the KS hypervisor trap)
394  rom_protect = 0;
395  hypervisor_start_machine();
396  speed_current = 0;
397  machine_set_speed(1);
398  DEBUG("INIT: end of initialization!" NL);
399 #ifdef XEMU_SNAPSHOT_SUPPORT
400  xemusnap_init(m65_snapshot_definition);
401  p = xemucfg_get_str("snapload");
402  if (p) {
403  if (xemusnap_load(p))
404  FATAL("Couldn't load snapshot \"%s\": %s", p, xemusnap_error_buffer);
405  }
406  m65_snapshot_saver_filename = xemucfg_get_str("snapsave");
407  atexit(m65_snapshot_saver_on_exit_callback);
408 #endif
409 }

472 {

582 static void emulation_loop ( void )
583 {
584  for (;;) {
585  while (XEMU_UNLIKELY(paused)) { // paused special mode, ie tracing support, or something ...
586  if (XEMU_UNLIKELY(dma_status))
587  break; // if DMA is pending, do not allow monitor/etc features
588 #ifdef HAS_UARTMON_SUPPORT
589  if (m65mon_callback) { // delayed uart monitor command should be finished ...
590  m65mon_callback();
591  m65mon_callback = NULL;
592  uartmon_finish_command();
593  }
594 #endif
595  // we still need to feed our emulator with update events ... It also slows this pause-busy-loop down to every full frames (~25Hz)
596  // note, that it messes timing up a bit here, as there is update_emulator() calls later in the "normal" code as well
597  // this can be a bug, but real-time emulation is not so much an issue if you eg doing trace of your code ...
598  update_emulator();
599  if (trace_step_trigger) {
600  // if monitor trigges a step, break the pause loop, however we will get back the control on the next
601  // iteration of the infinite "for" loop, as "paused" is not altered
602  trace_step_trigger = 0;
603  break; // break the pause loop now
604  }
605  // Decorate window title about the mode.
606  // If "paused" mode is switched off ie by a monitor command (called from update_emulator() above!)
607  // then it will resets back the the original state, etc
608  window_title_custom_addon = paused ? (char*)emulator_paused_title : NULL;
609  if (paused != paused_old) {
610  paused_old = paused;
611  if (paused) {
612  DEBUGPRINT("TRACE: entering into trace mode @ $%04X" NL, cpu65.pc);
613  cpu_cycles_per_step = 0;
614  } else {
615  DEBUGPRINT("TRACE: leaving trace mode @ $%04X" NL, cpu65.pc);
616  if (breakpoint_pc < 0)
617  cpu_cycles_per_step = cpu_cycles_per_scanline;
618  else
619  cpu_cycles_per_step = 0;
620  }
621  }
622  }
623  if (XEMU_UNLIKELY(in_hypervisor)) {
624  hypervisor_debug();
625  }
626  if (XEMU_UNLIKELY(breakpoint_pc == cpu65.pc)) {
627  DEBUGPRINT("TRACE: Breakpoint @ $%04X hit, Xemu moves to trace mode after the execution of this opcode." NL, cpu65.pc);
628  paused = 1;
629  }
630  cycles += XEMU_UNLIKELY(dma_status) ? dma_update_multi_steps(cpu_cycles_per_scanline) : cpu65_step(
631 #ifdef CPU_STEP_MULTI_OPS
632  cpu_cycles_per_step
633 #endif
634  ); // FIXME: this is maybe not correct, that DMA's speed depends on the fast/slow clock as well?
635  if (cycles >= cpu_cycles_per_scanline) {
636  scanline++;
637  //DEBUG("VIC3: new scanline (%d)!" NL, scanline);
638  cycles -= cpu_cycles_per_scanline;
639  cia_tick(&cia1, 64);
640  cia_tick(&cia2, 64);
641  if (scanline == 312) {
642  //DEBUG("VIC3: new frame!" NL);
643  frameskip = !frameskip;
644  scanline = 0;
645  if (!frameskip) // well, let's only render every full frames (~ie 25Hz)
646  update_emulator();
647  sid1.sFrameCount++;
648  sid2.sFrameCount++;
649  frame_counter++;
650  if (frame_counter == 25) {
651  frame_counter = 0;
652  vic3_blink_phase = !vic3_blink_phase;
653  }
654  frames_total_counter++;
655  if (!frameskip) // FIXME: do this better!!!!!!
656  return;
657  }
658  //DEBUG("RASTER=%d COMPARE=%d" NL,scanline,compare_raster);
659  //vic_interrupt();
660  vic3_check_raster_interrupt();
661  }
662  }
