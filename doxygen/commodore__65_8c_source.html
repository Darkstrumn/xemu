77 /* You *MUST* call this every time, when *any* of these events applies:
78  * MAP 4510 opcode is issued, map_offset_low, map_offset_high, map_mask are modified
79  * "CPU port" data or DDR register has been written, witn cpu_port[0 or 1] modified
80  * VIC3 register $30 is written, with vic3_registers[0x30] modified
81  The reason of this madness: do the ugly work here, as memory configuration change is
82  less frequent than memory usage (read/write). Thus do more work here, but simplier
83  work when doing actual memory read/writes, with a simple addition and shift, or such.
84  The tables are 4K in steps, 4510 would require only 8K steps, but there are other
85  reasons (ie, I/O area is only 4K long, mapping is not done by the CPU).
86  More advanced technique can be used not to handle *everything* here, but it's better
87  for the initial steps, to have all address translating logic at once.
88 */
89 void apply_memory_config ( void )
90 {
91  // FIXME: what happens if VIC-3 reg $30 mapped ROM is tried to be written? Ignored, or RAM is used to write to, as with the CPU port mapping?
92  // About the produced signals on the "CPU port"
93  int cp = (memory[1] | (~memory[0]));
94  // Simple ones, only CPU MAP may apply not other factors
95  // Also, these are the "lower" blocks, needs the offset for the "lower" area in case of CPU MAP'ed state
96  addr_trans_wr[0] = addr_trans_rd[0] = addr_trans_wr[1] = addr_trans_rd[1] = (map_mask & 1) ? map_offset_low : 0; // $0XXX + $1XXX, MAP block 0 [mask 1]
97  addr_trans_wr[2] = addr_trans_rd[2] = addr_trans_wr[3] = addr_trans_rd[3] = (map_mask & 2) ? map_offset_low : 0; // $2XXX + $3XXX, MAP block 1 [mask 2]
98  addr_trans_wr[4] = addr_trans_rd[4] = addr_trans_wr[5] = addr_trans_rd[5] = (map_mask & 4) ? map_offset_low : 0; // $4XXX + $5XXX, MAP block 2 [mask 4]
99  addr_trans_wr[6] = addr_trans_rd[6] = addr_trans_wr[7] = addr_trans_rd[7] = (map_mask & 8) ? map_offset_low : 0; // $6XXX + $7XXX, MAP block 3 [mask 8]
100  // From this point, we must use the "high" area offset if it's CPU MAP'ed
101  // $8XXX and $9XXX, MAP block 4 [mask 16]
102  if (vic3_registers[0x30] & 8)
103  addr_trans_wr[8] = addr_trans_rd[8] = addr_trans_wr[9] = addr_trans_rd[9] = ROM_8000_REMAP;
104  else if (map_mask & 16)
105  addr_trans_wr[8] = addr_trans_rd[8] = addr_trans_wr[9] = addr_trans_rd[9] = map_offset_high;
106  else
107  addr_trans_wr[8] = addr_trans_rd[8] = addr_trans_wr[9] = addr_trans_rd[9] = 0;
108  // $AXXX and $BXXX, MAP block 5 [mask 32]
109  if (vic3_registers[0x30] & 16)
110  addr_trans_wr[0xA] = addr_trans_rd[0xA] = addr_trans_wr[0xB] = addr_trans_rd[0xB] = ROM_A000_REMAP;
111  else if ((map_mask & 32))
112  addr_trans_wr[0xA] = addr_trans_rd[0xA] = addr_trans_wr[0xB] = addr_trans_rd[0xB] = map_offset_high;
113  else {
114  addr_trans_wr[0xA] = addr_trans_wr[0xB] = 0;
115  addr_trans_rd[0xA] = addr_trans_rd[0xB] = ((cp & 3) == 3) ? ROM_C64_BASIC_REMAP : 0;
116  }
117  // $CXXX, MAP block 6 [mask 64]
118  // Warning: all VIC3 reg $30 related ROM maps are for 8K size, *except* of '@C000' (interface ROM) which is only 4K! Also this is in another ROM bank than the others
119  if (vic3_registers[0x30] & 32)
120  addr_trans_wr[0xC] = addr_trans_rd[0xC] = ROM_C000_REMAP;
121  else
122  addr_trans_wr[0xC] = addr_trans_rd[0xC] = (map_mask & 64) ? map_offset_high : 0;
123  // $DXXX, *still* MAP block 6 [mask 64]
124  if (map_mask & 64)
125  addr_trans_wr[0xD] = addr_trans_rd[0xD] = map_offset_high;
126  else {
127  if ((cp & 7) > 4) {
128  addr_trans_wr[0xD] = addr_trans_rd[0xD] = IO_REMAP_VIRTUAL;
129  } else {
130  addr_trans_wr[0xD] = 0;
131  addr_trans_rd[0xD] = (cp & 3) ? ROM_C64_CHR_REMAP : 0;
132  }
133  }
134  // $EXXX and $FXXX, MAP block 7 [mask 128]
135  if (vic3_registers[0x30] & 128)
136  addr_trans_wr[0xE] = addr_trans_rd[0xE] = addr_trans_wr[0xF] = addr_trans_rd[0xF] = ROM_E000_REMAP;
137  else if (map_mask & 128)
138  addr_trans_wr[0xE] = addr_trans_rd[0xE] = addr_trans_wr[0xF] = addr_trans_rd[0xF] = map_offset_high;
139  else {
140  addr_trans_wr[0xE] = addr_trans_wr[0xF] = 0;
141  addr_trans_rd[0xE] = addr_trans_rd[0xF] = ((cp & 3) > 1) ? ROM_C64_KERNAL_REMAP : 0;
142  }
143 }

157 static inline void nmi_set ( int level, int mask )
158 {
159  // NMI is a low active _EDGE_ triggered 65xx input ... In my emulator though, the signal
160  // is "high active", and also we must form the "edge" ourselves from "level". NMI level is
161  // set as a 2bit number, on bit 0, CIA2, on bit 1, keyboard RESTORE key. Thus having zero
162  // value for level means (in the emu!) that not RESTORE key is pressed neither CIA2 has active
163  // IRQ output, non-zero value means some activation. Well, if I am not confused enough here,
164  // this should mean that nmi_level zero->non-zero transit should produce the edge (which should
165  // be the falling edge in the real hardware anyway ... but the rising here. heh, I should follow
166  // the signal level of the hardware in my emulator, honestly ...)
167  int nmi_new_level;
168  if (level)
169  nmi_new_level = nmi_level | mask;
170  else
171  nmi_new_level = nmi_level & (~mask);
172  if ((!nmi_level) && nmi_new_level) {
173  DEBUG("NMI edge is emulated towards the CPU (%d->%d)" NL, nmi_level, nmi_new_level);
174  cpu65.nmiEdge = 1; // the "NMI edge" trigger is deleted by the CPU emulator itself (it's not a level trigger)
175  }
176  nmi_level = nmi_new_level;
177 }

293 static void c65_init ( int sid_cycles_per_sec, int sound_mix_freq )
294 {
295  const char *p;
296  SDL_AudioSpec audio_want, audio_got;
297  hid_init(
298  c64_key_map,
299  VIRTUAL_SHIFT_POS,
300  SDL_ENABLE // joy HID events enabled
301  );
302 #ifdef HID_KBD_MAP_CFG_SUPPORT
303  hid_keymap_from_config_file(xemucfg_get_str("keymap"));
304 #endif
305  joystick_emu = 1;
306  nmi_level = 0;
307  // *** host-FS
308  p = xemucfg_get_str("hostfsdir");
309  if (p)
310  hostfs_init(p, NULL);
311  else
312  hostfs_init(sdl_pref_dir, "hostfs");
313  // *** Init memory space
314  memset(memory, 0xFF, sizeof memory);
315  // *** Initialize VIC3
316  vic3_init();
317  // *** Memory configuration
318  memory[0] = memory[1] = 0xFF; // the "CPU I/O port" on 6510/C64, implemented by VIC3 for real in C65!
319  map_mask = 0; // as all 8K blocks are unmapped, we don't need to worry about the low/high offset to set here
320  apply_memory_config(); // VIC3 $30 reg is already filled, so it's OK to call this now
321  // *** CIAs
322  cia_init(&cia1, "CIA-1",
323  NULL, // callback: OUTA
324  NULL, // callback: OUTB
325  NULL, // callback: OUTSR
326  cia1_in_a, // callback: INA ~ joy#2
327  cia1_in_b, // callback: INB ~ keyboard
328  NULL, // callback: INSR
329  cia1_setint_cb // callback: SETINT
330  );
331  cia_init(&cia2, "CIA-2",
332  cia2_out_a, // callback: OUTA ~ eg VIC-II bank
333  NULL, // callback: OUTB
334  NULL, // callback: OUTSR
335  cia_port_in_dummy, // callback: INA
336  NULL, // callback: INB
337  NULL, // callback: INSR
338  cia2_setint_cb // callback: SETINT ~ that would be NMI in our case
339  );
340  // *** Initialize DMA
341  dma_init(xemucfg_get_num("dmarev") & 0xFF00); // initial DMA revision will be zero ...
342  // *** Load ROM image
343  if (c65_load_rom(xemucfg_get_str("rom"), xemucfg_get_num("dmarev"))) // ... but this overrides the DMA revision!
344  XEMUEXIT(1);
345  // Initialize FDC
346  fdc_init(disk_cache);
347  // Initialize D81 access abstraction for FDC
348  d81access_init();
349  atexit(d81access_close);
350  d81access_attach_fsobj(xemucfg_get_str("8"), D81ACCESS_IMG | D81ACCESS_PRG | D81ACCESS_DIR | D81ACCESS_AUTOCLOSE | (xemucfg_get_bool("d81ro") ? D81ACCESS_RO : 0));
351  // SIDs, plus SDL audio
352  sid_init(&sids[0], sid_cycles_per_sec, sound_mix_freq);
353  sid_init(&sids[1], sid_cycles_per_sec, sound_mix_freq);
354  SDL_memset(&audio_want, 0, sizeof(audio_want));
355  audio_want.freq = sound_mix_freq;
356  audio_want.format = AUDIO_S16SYS; // used format by SID emulation (ie: signed short)
357  audio_want.channels = 2; // that is: stereo, for the two SIDs
358  audio_want.samples = 1024; // Sample size suggested (?) for the callback to render once
359  audio_want.callback = audio_callback; // Audio render callback function, called periodically by SDL on demand
360  audio_want.userdata = NULL; // Not used, "userdata" parameter passed to the callback by SDL
361  audio = SDL_OpenAudioDevice(NULL, 0, &audio_want, &audio_got, 0);
362  if (audio) {
363  int i;
364  for (i = 0; i < SDL_GetNumAudioDevices(0); i++)
365  DEBUG("AUDIO: audio device is #%d: %s" NL, i, SDL_GetAudioDeviceName(i, 0));
366  // Sanity check that we really got the same audio specification we wanted
367  if (audio_want.freq != audio_got.freq || audio_want.format != audio_got.format || audio_want.channels != audio_got.channels) {
368  SDL_CloseAudioDevice(audio); // forget audio, if it's not our expected format :(
369  audio = 0;
370  ERROR_WINDOW("Audio parameter mismatches.");
371  }
372  DEBUG("AUDIO: initialized (#%d), %d Hz, %d channels, %d buffer sample size." NL, audio, audio_got.freq, audio_got.channels, audio_got.samples);
373  } else
374  ERROR_WINDOW("Cannot open audio device!");
375  // *** RESET CPU, also fetches the RESET vector into PC
376  cpu65_reset();
377  DEBUG("INIT: end of initialization!" NL);
378  // *** Snapshot init and loading etc should be the LAST!!!! (at least the load must be last to have initiated machine state, xemusnap_init() can be called earlier too)
379 #ifdef XEMU_SNAPSHOT_SUPPORT
380  xemusnap_init(c65_snapshot_definition);
381  p = xemucfg_get_str("snapload");
382  if (p) {
383  if (xemusnap_load(p))
384  FATAL("Couldn't load snapshot \"%s\": %s", p, xemusnap_error_buffer);
385  }
386  c65_snapshot_saver_filename = xemucfg_get_str("snapsave");
387  atexit(c65_snapshot_saver_on_exit_callback);
388 #endif
389 }

397 // *** Implements the MAP opcode of 4510, called by the 65CE02 emulator
398 void cpu65_do_aug_callback ( void )
399 {
400  cpu65.cpu_inhibit_interrupts = 1; // disable interrupts to the next "EOM" (ie: NOP) opcode
401  DEBUG("CPU: MAP opcode, input A=$%02X X=$%02X Y=$%02X Z=$%02X" NL, cpu65.a, cpu65.x, cpu65.y, cpu65.z);
402  map_offset_low = (cpu65.a << 8) | ((cpu65.x & 15) << 16); // offset of lower half (blocks 0-3)
403  map_offset_high = (cpu65.y << 8) | ((cpu65.z & 15) << 16); // offset of higher half (blocks 4-7)
404  map_mask = (cpu65.z & 0xF0) | (cpu65.x >> 4); // "is mapped" mask for blocks (1 bit for each)
405  DEBUG("MEM: applying new memory configuration because of MAP CPU opcode" NL);
406  DEBUG("LOW -OFFSET = $%X" NL, map_offset_low);
407  DEBUG("HIGH-OFFSET = $%X" NL, map_offset_high);
408  DEBUG("MASK = $%02X" NL, map_mask);
409  apply_memory_config();
410 }

452 Uint8 io_read ( int addr )
453 {
454  addr &= 0xFFF; // Internally, we use I/O addresses $0000-$0FFF only!
455  switch ((addr >> 8) | vic_new_mode) { // "addr >> 8" is from 0...F, vic_new_mode can be $10 or 0, that is $00-$1F for all cases
456  /* --- I/O read in old VIC I/O mode --- */
457  case 0x00: // $D000-$D0FF
458  case 0x01: // $D100-$D1FF, C65 VIC-III palette, red components (in old VIC mode, cannot be accessed)
459  case 0x02: // $D200-$D2FF, C65 VIC-III palette, green components (in old VIC mode, cannot be accessed)
460  case 0x03: // $D300-$D3FF, C65 VIC-III palette, blue components (in old VIC mode, cannot be accessed)
461  // In old VIC-II mode, according to c65manual.txt and C64 "standard", VIC-II registers can be seen at every 64 bytes in I/O range $D000-$D3FF
462  return vic3_read_reg(addr & 0x3F);
463  case 0x04: // $D400-$D4FF, SID stuffs

527 {
528  addr &= 0xFFF; // Internally, we use I/O addresses $0000-$0FFF only!
529  switch ((addr >> 8) | vic_new_mode) { // "addr >> 8" is from 0...F, vic_new_mode can be $10 or 0, that is $00-$1F for all cases
530  /* --- I/O write in old VIC I/O mode --- */

532  case 0x01: // $D100-$D1FF, C65 VIC-III palette, red components (in old VIC mode, cannot be accessed)

534  case 0x03: // $D300-$D3FF, C65 VIC-III palette, blue components (in old VIC mode, cannot be accessed)
535  // In old VIC-II mode, according to c65manual.txt and C64 "standard", VIC-II registers can be seen at every 64 bytes in I/O range $D000-$D3FF
536  vic3_write_reg(addr & 0x3F, data);
537  return;
538  case 0x04: // $D400-$D4FF, SID stuffs
539  case 0x05: // $D500-$D5FF
540  case 0x06: // $D600-$D6FF, would be C65 UART, in old I/O mode: SID images still

714 // Called in case of an RMW (read-modify-write) opcode write access.
715 // Original NMOS 6502 would write the old_data first, then new_data.
716 // It has no inpact in case of normal RAM, but it *does* with an I/O register in some cases!
717 // CMOS line of 65xx (probably 65CE02 as well?) seems not write twice, but read twice.
718 // However this leads to incompatibilities, as some software used the RMW behavour by intent.
719 // Thus Mega65 fixed the problem to "restore" the old way of RMW behaviour.
720 // I also follow this path here, even if it's *NOT* what 65CE02 would do actually!
721 void cpu65_write_rmw_callback ( Uint16 addr, Uint8 old_data, Uint8 new_data )
722 {
723  int phys_addr = addr_trans_wr[addr >> 12] + addr; // translating address with the WRITE table created by apply_memory_config()
724  if (XEMU_LIKELY(phys_addr < 0x10FF00))
725  write_phys_mem(phys_addr, new_data); // "normal" memory, just write once, no need to emulate the behaviour
726  else {
727  DEBUG("CPU: RMW opcode is used on I/O area for $%04X" NL, addr);
728  io_write(phys_addr, old_data); // first write back the old data ...
729  io_write(phys_addr, new_data); // ... then the new
730  }
