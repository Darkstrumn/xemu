34 static void emulation_loop ( void )
35 {
36  if (XEMU_UNLIKELY(stop_emulation)) {
37  if (console_input() == 32)
38  exit(0);
39  } else {
40  if (XEMU_UNLIKELY(trace)) {
41  char disasm_buffer[128];
42  while (cpu_cycles < cpu_cycles_per_frame) {
43  z80_custom_disasm(Z80_PC, disasm_buffer, sizeof disasm_buffer);
44  if (*disasm_buffer)
45  puts(disasm_buffer);
46  cpu_cycles += z80ex_step();
47  }
48  } else {
49  while (cpu_cycles < cpu_cycles_per_frame)
50  cpu_cycles += z80ex_step();
51  }
52  cpu_cycles -= cpu_cycles_per_frame;
53 #if 0
54  if (emu_cost_cycles) {
55  cpu_cycles += emu_cost_cycles;
56  emu_cost_cycles = 0;
57  }
58  if (emu_cost_usecs) {
59  cpu_cycles += emu_cost_usecs / cpu_mhz;
60  emu_cost_usecs = 0;
61  }
62 #endif
63  }
64  console_cursor_blink(5);
65  console_iteration();
66  xemu_timekeeping_delay(1000000 / FRAME_RATE);
67 }

71 static int get_guarded_cfg_num ( const char *optname, int min, int max )
72 {
73  int ret = xemucfg_get_num(optname);
74  if (ret < min) {
75  DEBUGPRINT("BADNUM: too low value (%d) for option \"%s\", the minimum is %d, applying that, instead." NL, ret, optname, min);
76  return min;
77  } else if (ret > max) {
78  DEBUGPRINT("BADNUM: too high value (%d) for option \"%s\", the maximum is %d, applying that, instead." NL, ret, optname, max);
79  return max;
80  } else
81  return ret;
82 }

112 static int load ( const char *fn )
113 {
114  if (!fn)
115  return 0;
116  if (!*fn)
117  FATAL("Empty string for loading program.");
118  // the last parameter for mounting drive will instruct CPM-FS to get the drive path as the dirname part of "fn"
119  if (cpmfs_mount_drive(0, fn, 1))
120  return 1;
121  cpmfs_mount_drive(1, "/", 0);
122  int tpa_size = memory[6] + (memory[7] << 8) - 0x100;
123  DEBUGPRINT("LOAD: trying to load program \"%s\", max allowed size = %d bytes." NL, fn, tpa_size);
124  int size = xemu_load_file(fn, memory + 0x100, 1, tpa_size, "Cannot load program");
125  if (size < 0)
126  return 1;
127  DEBUGPRINT("LOAD: Program loaded, %d bytes, %d bytes TPA remained free" NL, size, tpa_size - size);
128  Z80_PC = 0x100;
129  cpm_dma = 0x80; // default DMA
130  Z80_C = 0; // FIXME: should be the same as zero page loc 4, drive + user number ...
131  memset(memory + 8, 0, 0x100 - 8);
132  memset(memory + 0x5C + 1, 32, 11);
133  memset(memory + 0x6C + 1, 32, 11);
134  return 0;
135 }

139 int main ( int argc, char **argv )
140 {
141  int memtop = 0x10000;
142  xemu_pre_init(APP_ORG, TARGET_NAME, "Re-CP/M");
143  xemucfg_define_switch_option("fullscreen", "Start in fullscreen mode");
144  xemucfg_define_switch_option("syscon", "Keep system console open (Windows-specific effect only)");
145  xemucfg_define_num_option("width", 80, "Terminal width in character columns");
146  xemucfg_define_num_option("height", 25, "Terminal height in character rows");
147  xemucfg_define_num_option("zoom", 100, "Zoom the window by the given percentage (50%-200%)");
148  xemucfg_define_num_option("cpmsize", 64, "Size of the CP/M system");
149  xemucfg_define_num_option("clock", 4, "Rough Z80 emulation speed in MHz with 'emulation cost'");
150  xemucfg_define_num_option("baud", 0, "Emulate serial terminal with about the given baud rate [0=disable]");
151  xemucfg_define_str_option("load", NULL, "Load and run a CP/M program");
152  xemucfg_define_switch_option("trace", "Trace the program, VERY spammy!");
153  xemucfg_define_switch_option("mapvideo", "Map video+colour RAM into the end of addr space");
154  if (xemucfg_parse_all(argc, argv))
155  return 1;
156  trace = xemucfg_get_bool("trace");
157  memset(memory, 0, sizeof memory);
158  memset(modded, 0, sizeof modded);
159  if (console_init(
160  get_guarded_cfg_num("width", 38, 160),
161  get_guarded_cfg_num("height", 20, 60),
162  get_guarded_cfg_num("zoom", 50, 200),
163  xemucfg_get_bool("mapvideo") ? &memtop : NULL,
164  xemucfg_get_num("baud") > 0 ? get_guarded_cfg_num("baud", 300, 1000000) : 0
165  ))
166  return 1;
167  memtop &= ~0xFF;
168  DEBUGPRINT("System RAM size: %d Kbytes." NL, memtop >> 10);
169  bios_install(memtop - 0x100);
170  bdos_install(memtop - 0x200);
171  cpmfs_init();
172  osd_init_with_defaults();
173  /* Intialize memory and load ROMs */
174  clear_emu_events(); // also resets the keyboard
175  cpu_mhz = get_guarded_cfg_num("clock", 1, 33);
176  cpu_cycles_per_frame = (1000000 * cpu_mhz) / FRAME_RATE;
177  DEBUGPRINT("Z80: setting CPU speed to %dMHz, %d CPU cycles per refresh-rate (=%dHz)" NL, cpu_mhz, cpu_cycles_per_frame, FRAME_RATE);
178  z80ex_init();
179  if (!xemucfg_get_bool("syscon"))
180  sysconsole_close(NULL);
181  Z80_PC = 0;
182  Z80_SP = 0x100;
183  if (load(xemucfg_get_str("load")))
184  return 1;
185  conputs("re-CP/M\r\n");
186  xemu_set_full_screen(xemucfg_get_bool("fullscreen"));
187  xemu_timekeeping_start(); // we must call this once, right before the start of the emulation
188  XEMU_MAIN_LOOP(emulation_loop, FRAME_RATE, 1);
189  return 0;
190 }
