161 static inline void block_uint32 ( unsigned int ofs, Uint32 data )
162 {
163  block[ofs ] = (data ) & 0xff;
164  block[ofs + 1] = (data >> 8) & 0xff;
165  block[ofs + 2] = (data >> 16) & 0xff;
166  block[ofs + 3] = (data >> 24) & 0xff;

169 {

175 // Make a FAT32FS. Called by fdisk() function
176 // I copied the algorithm from mega65-fdisk closely (and hopefully well enough) not to have any compatibility issue.
177 static int mkfs_fat32 ( Uint32 start_block, Uint32 number_of_blocks )
178 {
179  DEBUGPRINT("SDCARD: FAT32FS: creating file system between sectors $%X-$%X, $%X sectors (%uK) in size" NL, start_block, start_block + number_of_blocks - 1, number_of_blocks, number_of_blocks >> 11);
180  // *** Let's do some calculations. According to mega65-fdisk, as usual in these mkfs parts :)
181  Uint8 sectors_per_cluster = 8; // 4Kbyte clusters
182  Uint32 reserved_sectors = 1024 * 1024 / 512;
183  Uint32 fat_available_sectors = number_of_blocks - reserved_sectors;
184  Uint32 fs_clusters = fat_available_sectors / sectors_per_cluster;
185  Uint32 fat_sectors = fs_clusters/(512 / 4);
186  if (fs_clusters % (512 / 4))
187  fat_sectors++;
188  Uint32 sectors_required = 2 * fat_sectors+((fs_clusters - 2) * sectors_per_cluster);
189  while (sectors_required > fat_available_sectors) {
190  Uint32 excess_sectors = sectors_required - fat_available_sectors;
191  Uint32 delta = (excess_sectors / (1 + sectors_per_cluster));
192  if (delta < 1)
193  delta = 1;
194  fs_clusters -= delta;
195  fat_sectors = fs_clusters / (512 / 4);
196  if (fs_clusters % (512 / 4))
197  fat_sectors++;
198  sectors_required = 2 * fat_sectors + ((fs_clusters - 2) * sectors_per_cluster);
199  }
200  Uint32 fat1_sector = reserved_sectors;
201  Uint32 fat2_sector = fat1_sector + fat_sectors;
202  Uint32 rootdir_sector = fat2_sector + fat_sectors;
203  // *** Create boot sector
204  ZERO_BUFFER();
205  memcpy(block, boot_bytes, sizeof boot_bytes);
206  // 0x20-0x23 = 32-bit number of data sectors in file system
207  //for (int i = 0; i < 4; i++)
208  // block[0x20 + i] = (number_of_blocks >> (i * 8)) & 0xff;
209  block_uint32(0x20, number_of_blocks);
210  // 0x24-0x27 = 32-bit number of sectors per fat
211  //for (int i = 0; i < 4; i++)
212  // block[0x24 + i] = (fat_sectors >> (i * 8)) & 0xff;
213  block_uint32(0x24, fat_sectors);
214  block[510] = 0x55; // boot signature
215  block[511] = 0xaa; // boot signature
216  WRITE_BLOCK(start_block); // write boot sector
217  WRITE_BLOCK(start_block + 6); // write backup boot sector (the very same, with @ +6 sector)
218  // *** Create FAT FS information sector
219  ZERO_BUFFER();
220  block[0] = 0x52;
221  block[1] = 0x52;
222  block[2] = 0x61;
223  block[3] = 0x41;
224  block[0x1e4] = 0x72;
225  block[0x1e5] = 0x72;
226  block[0x1e6] = 0x41;
227  block[0x1e7] = 0x61;
228  //for(int i = 0; i < 4; i++) // Number of free clusters
229  // block[0x1e8 + i] = ((fs_clusters - 3) >> (i * 8)) & 0xff;
230  block_uint32(0x1e8, fs_clusters - 3);
231  // First free cluster = 2
232  block[0x1ec] = 0x02 + 1; // OSX newfs/fsck puts 3 here instead?
233  // Boot sector signature
234  block[510] = 0x55;
235  block[511] = 0xaa;
236  WRITE_BLOCK(start_block + 1);
237  WRITE_BLOCK(start_block + 7);
238  // *** Build empty FATs
239  ZERO_BUFFER();
240  memcpy(block, fat_bytes, sizeof fat_bytes);
241  WRITE_BLOCK(start_block + fat1_sector);
242  WRITE_BLOCK(start_block + fat2_sector);
243  // *** Build empty root directory
244  ZERO_BUFFER();
245  memcpy(block, default_volume_name, 11);
246  memcpy(block + 11, dir_bytes, sizeof dir_bytes);
247  WRITE_BLOCK(start_block + rootdir_sector);
248  // *** Erase some sectors
249  ZERO_BUFFER();
250  WRITE_BLOCK_RANGE(start_block + 1 + 1, start_block + 6 - 1);
251  WRITE_BLOCK_RANGE(start_block + 6 + 1, start_block + fat1_sector - 1);
252  WRITE_BLOCK_RANGE(start_block + fat1_sector + 1, start_block + fat2_sector - 1);
253  WRITE_BLOCK_RANGE(start_block + fat2_sector + 1, start_block + rootdir_sector - 1);
254  WRITE_BLOCK_RANGE(start_block + rootdir_sector + 1, start_block + rootdir_sector + 1 + sectors_per_cluster - 1);
255  return 0;
256 }

260 static int mksyspart ( Uint32 start_block, Uint32 number_of_blocks )
261 {
262  Uint32 slot_size = 512 * 1024 / 512; // slot_size units is sectors
263  // Take 1MB from partition size, for reserved space when
264  // calculating what can fit.
265  Uint32 reserved_sectors = 1024 * 1024 / 512;
266  Uint32 slot_count = (number_of_blocks - reserved_sectors) / (slot_size * 2 + 1);
267  if (slot_count >= 0xffff)
268  slot_count = 0xffff;
269  Uint16 dir_size = 1 + (slot_count / 4);
270  Uint16 freeze_dir_sectors = dir_size;
271  Uint16 service_dir_sectors = dir_size;
272  // Freeze directory begins at 1MB
273  Uint32 sys_partition_freeze_dir = reserved_sectors;
274  // System service directory begins after that
275  Uint32 sys_partition_service_dir = sys_partition_freeze_dir + slot_size * slot_count;
276  // *** System sector
277  ZERO_BUFFER();
278  memcpy(block, sys_part_magic, sizeof sys_part_magic); // magic bytes
279  block_uint32(0x10, 0); // $010-$013 = Start of freeze program area
280  block_uint32(0x14, slot_size * slot_count + dir_size); // $014-$017 = Size of freeze program area
281  block_uint32(0x18, slot_size); // $018-$01b = Size of each freeze program slot
282  block_uint16(0x1c, slot_count); // $01c-$01d = Number of freeze slots
283  block_uint16(0x1e, dir_size); // $01e-$01f = Number of sectors in freeze slot directory
284  block_uint32(0x20, slot_size * slot_count + dir_size); // $020-$023 = Start of freeze program area
285  block_uint32(0x24, slot_size * slot_count + dir_size); // $024-$027 = Size of service program area
286  block_uint32(0x28, slot_size); // $028-$02b = Size of each service slot
287  block_uint16(0x2c, slot_count); // $02c-$02d = Number of service slots
288  block_uint16(0x2e, dir_size); // $02e-$02f = Number of sectors in service slot directory
289  // Now make sector numbers relative to start of disk for later use
290  sys_partition_freeze_dir += start_block;
291  sys_partition_service_dir += start_block;
292  WRITE_BLOCK(start_block);
293  // *** Config sector
294  ZERO_BUFFER();
295  block[0x000] = 0x01; // Structure version bytes
296  block[0x001] = 0x01;
297  block[0x002] = 0x80; // PAL=$00, NTSC=$80
298  block[0x003] = 0x41; // Enable audio amp, mono output
299  block[0x004] = 0x00; // Use SD card for floppies
300  block[0x005] = 0x01; // Enable use of Amiga mouses automatically
301  block[0x006] = 0x41; // Ethenet MAC, start, from mega65-fdisk: "Should do a better job of this!"
302  block[0x007] = 0x41;
303  block[0x008] = 0x41;
304  block[0x009] = 0x41;
305  block[0x00A] = 0x41;
306  block[0x00B] = 0x41; // Ethernet MAC, end
307  // Set name of default disk image
308  memcpy(block + 0x10, default_disk_image, strlen(default_disk_image));
309  // DMAgic version: default, new revision (F011B)
310  block[0x020]=0x01;
311  WRITE_BLOCK(start_block + 1U);
312  // ***** erase some sectors
313  ZERO_BUFFER();
314  // Erase the rest of the 1MB reserved area
315  WRITE_BLOCK_RANGE(start_block + 2, start_block + 1023);
316  // erase frozen program directory
317  WRITE_BLOCK_RANGE(sys_partition_freeze_dir, sys_partition_freeze_dir + freeze_dir_sectors - 1);
318  // erase system service image directory
319  WRITE_BLOCK_RANGE(sys_partition_service_dir, sys_partition_service_dir + service_dir_sectors - 1);
320  return 0;
321 }

326 // Actually this only makes the MBR and partition boundaries, and calls mkfs_fat32() for the filesystem creation on a partition.
327 // I copied the algorithm from mega65-fdisk closely (and hopefully well enough) not to have any compatibility issue.
328 static int fdisk ( Uint32 device_size )
329 {
330  our_data_partition = -1;
331  DEBUGPRINT("SDCARD: FDISK: creating partition table ..." NL);
332  // *** Current policy to create sizes and boundaries of the partitions are based on mega65-fdisk source
333  Uint32 sys_part_sects = (device_size - 0x0800) >> 1;
334  if (sys_part_sects > 0x400000)
335  sys_part_sects = 0x400000;
336  sys_part_sects &= 0xfffff800; // round down to nearest 1MB boundary
337  Uint32 usr_part_sects = device_size - 0x0800 - sys_part_sects;
338  Uint32 usr_part_start = 0x0800;
339  Uint32 sys_part_start = usr_part_start + usr_part_sects;
340  // *** Create partition table (according to mega65-fdisk source itself)
341  // note: zero values are not set to zero individually, as those are the default after ZERO_BUFFER()
342  ZERO_BUFFER();
343  // disk signature, fixed value
344  block[0x1b8] = 0x83;
345  block[0x1b9] = 0x7d;
346  block[0x1ba] = 0xcb;
347  block[0x1bb] = 0xa6;
348  // MBR signature
349  block[0x1fe] = 0x55;
350  block[0x1ff] = 0xaa;
351  // FAT32 data ("user" partition)
352  block[0x1c2] = 0x0c; // partition type, VFAT32
353  block[0x1c6] = (usr_part_start ) & 0xff;
354  block[0x1c7] = (usr_part_start >> 8) & 0xff; // LBA start of our FS
355  block[0x1c8] = (usr_part_start >> 16) & 0xff;
356  block[0x1c9] = (usr_part_start >> 24) & 0xff;
357  block[0x1ca] = (usr_part_sects ) & 0xff; // LBA size of our FS
358  block[0x1cb] = (usr_part_sects >> 8) & 0xff;
359  block[0x1cc] = (usr_part_sects >> 16) & 0xff;
360  block[0x1cd] = (usr_part_sects >> 24) & 0xff;
361  // The 'system' area/partition
362  block[0x1d2] = 0x41; // partition type, Mega-65 system partition
363  block[0x1d6] = (sys_part_start ) & 0xff; // LBA start of our sys-area
364  block[0x1d7] = (sys_part_start >> 8) & 0xff;
365  block[0x1d8] = (sys_part_start >> 16) & 0xff;
366  block[0x1d9] = (sys_part_start >> 24) & 0xff;
367  block[0x1da] = (sys_part_sects ) & 0xff; // LBA size of our sys-area
368  block[0x1db] = (sys_part_sects >> 8) & 0xff;
369  block[0x1dc] = (sys_part_sects >> 16) & 0xff;
370  block[0x1dd] = (sys_part_sects >> 24) & 0xff;
371  WRITE_BLOCK(0);
372  // *** Create FAT32 (data/user area)
373  if (mkfs_fat32(usr_part_start, usr_part_sects))
374  return -1;
375  // *** Create SYSTEM (system area/partition)
376  if (mksyspart(sys_part_start, sys_part_sects))
377  return -1;
378  // *** Make fat32.c to re-read partition table ...
379  DEBUGPRINT("SDCARD: FDISK: Re-reading MBR ..." NL);
380  // this should give result '0' (the 0'th aka first partition is valid)
381  // Note: in this implementation we ALWAYS make data partition as the first one (well, the zero'th ...)
382  if (mfat_init_mbr() != 0)
383  FATAL("First partition entry is not valid even after FDISK!");
384  if (mfat_use_part(0) != 0)
385  FATAL("First partition cannot be used as FAT32FS even after FDISK/MKFS!");
386  our_data_partition = 0;
387  return 0;
388 }

390 /* Parameters:
391  on_card_name: file of the name on the card, will be "DOS-ized", ie 8+3 chars, all capitals. Cannot have directory component!
392  options: options with SDCONTENT_* stuff, to check overwrite is OK, should ask, etc
393  fn_or_data: # if size_to_install > 0, it's a memory pointer, with size_to_install bytes to install from there
394  # if size_to_install = 0, it's a file name to load data from and install that, with the size of the given file
395  # if size_to_install < 0, it's a file name, with -size_to_install long, and size mismatch will be checked and even asserted
396  size_to_install: see above
397  Notes:
398  Currently policy: if file already existed on the image, we will reuse the FAT chain, _IF_ it's not fragmented and would fit.
399  If the FAT chain is not long enough (ie, the size of the new file wouldn't fit in terms of number of clusters) we rather
400  create a NEW fat chain, and free old one. This is because it's easier to do so, and some Mega65 features (disk images) needs
401  strictly UNFRAGMENTED state anyway.
402 */
403 static int update_sdcard_file ( const char *on_card_name, int options, const char *fn_or_data, int size_to_install )
404 {
405  int fd = -1;
406  if (size_to_install <= 0) {
407  fd = open(fn_or_data, O_RDONLY | O_BINARY);
408  if (fd < 0) {
409  ERROR_WINDOW("SD-card image updater cannot open file:\n%s\n(%s)", fn_or_data, strerror(errno));
410  goto error_on_maybe_sys_file;
411  }
412  off_t oft = xemu_safe_file_size_by_fd(fd);
413  if (oft == OFF_T_ERROR || (size_to_install < 0 && oft != (off_t)(-size_to_install))) {
414  DEBUGPRINT("Got size: " PRINTF_LLD " instruct size: %d" NL, (long long int)oft, size_to_install);
415  ERROR_WINDOW("Bad file, size is incorrect or other I/O error\n%s", fn_or_data);
416  goto error_on_maybe_sys_file;
417  }
418  size_to_install = (int)oft;
419  if (!size_to_install)
420  return 0; // FIXME: we don't allow empty files to be written. Though we fake OK result no to disturb the user
421  }
422  //mfat_dirent_t rootdir;
423  //mfat_open_rootdir(&rootdir.stream);
424  Uint32 block = mfat_overwrite_file_with_direct_linear_device_block_write(&sd_rootdirent, on_card_name, size_to_install);
425  if (block == 0)
426  goto error_on_maybe_sys_file;
427  // Copy file block by block
428  while (size_to_install) {
429  Uint8 buffer[512];
430  int need = (size_to_install < sizeof buffer) ? size_to_install : sizeof buffer;
431  if (need < sizeof(buffer))
432  memset(buffer + need, 0, sizeof(buffer) - need);
433  if (fd >= 0) {
434  // Read from external file
435  int got = xemu_safe_read(fd, buffer, need);
436  if (got < 0)
437  goto error;
438  if (got == 0)
439  goto error;
440  if (got != need)
441  goto error;
442  } else {
443  // Read from memory
444  memcpy(buffer, fn_or_data, need);
445  fn_or_data += need;
446  }
447  // And now WRITE!!!!!
448  sdcard_write_block(block++, buffer); // FIXME: error handling!!!
449  size_to_install -= need;
450  }
451  if (fd >= 0)
452  close(fd);
453  return 0;
454 error_on_maybe_sys_file:
455  if ((options & SDCONTENT_SYS_FILE))
456  ERROR_WINDOW(
457  "This file is a must for Xemu/Mega65, however it's under\n"
458  "copyright by their respective owners.\n"
459  "It's totally the user's responsibility to get/use/own/handle this file!\n%s",
460  fn_or_data
461  );
462 error:
463  if (fd >= 0)
464  close(fd);
465  return -1;
466 }

469 static int update_from_directory ( const char *dirname, int options )
470 {
471  DIR *dir = opendir(dirname);
472  if (!dir) {
473  ERROR_WINDOW("Cannot update SD image / virtual-disk:\nCannot open specified directory:\n%s\n%s",
474  dirname,
475  strerror(errno)
476  );
477  return errno;
478  }
479  for (;;) {
480  errno = 0;
481  struct dirent *entry = readdir(dir);
482  if (!entry)
483  break;
484  if (entry->d_name[0] == '.')
485  continue;
486  char fn[PATH_MAX];
487  snprintf(fn, sizeof(fn), "%s" DIRSEP_STR "%s", dirname, entry->d_name);
488  struct stat st;
489  if (stat(fn, &st)) {
490  DEBUGPRINT("SDCARD: CONTENT: skipping updating file \"%s\": start() did not worked: %s" NL, fn, strerror(errno));
491  continue;
492  }
493  if ((st.st_mode & S_IFMT) != S_IFREG) {
494  DEBUGPRINT("SDCARD: CONTENT: skipping updating file \"%s\": not a regular file" NL, fn);
495  continue;
496  }
497  if (st.st_size == 0 || st.st_size > MAX_IMPORT_FILE_SIZE) { // FIXME: I limit here the max file size, but is there a sane limit?
498  DEBUGPRINT("SDCARD: CONTENT: skipping updating file \"%s\": too large (limit is %d bytes) or null-sized file" NL, fn, MAX_IMPORT_FILE_SIZE);
499  continue;
500  }
501  int ret = update_sdcard_file(entry->d_name, options, fn, -(int)st.st_size);
502  DEBUGPRINT("SDCARD: CONTENT: updated file \"%s\" status is %d" NL, fn, ret);
503  }
504  int ret = errno;
505  closedir(dir);
506  return ret;
507 }
