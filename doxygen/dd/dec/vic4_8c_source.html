105 void vic_init ( void )
106 {
107  int r, g, b, i;
108  // *** Init 4096 element palette with RGB components for faster access later on palette register changes (avoid SDL calls to convert)
109  // TODO: for M65 mode, where a colour component is 8 bit, this is kind of problematic though :(
110  for (r = 0, i = 0; r < 16; r++)
111  for (g = 0; g < 16; g++)
112  for (b = 0; b < 16; b++)
113  rgb_palette[i++] = SDL_MapRGBA(sdl_pix_fmt, r * 17, g * 17, b * 17, 0xFF); // 15*17=255, last arg 0xFF: alpha channel for SDL
114  force_fast = 0;
115  // *** Init VIC3 registers and palette
116  vic2_16k_bank = 0;
117  vic_iomode = VIC2_IOMODE;
118  interrupt_status = 0;
119  palette = vic3_rom_palette;
120  scanline = 0;
121  compare_raster = 0;
122  for (i = 0; i < 0x100; i++) { // Initiailize all palette registers to zero, initially, to have something ...
123  if (i < sizeof vic_registers)
124  vic_registers[i] = 0; // Also the VIC registers ...
125  vic3_rom_palette[i] = vic3_palette[i] = rgb_palette[0];
126  vic3_palette_nibbles[i] = 0;
127  vic3_palette_nibbles[i + 0x100] = 0;
128  vic3_palette_nibbles[i + 0x200] = 0;
129  }
130  // *** the ROM palette "fixed" colours
131  vic3_rom_palette[ 0] = RGB( 0, 0, 0); // black
132  vic3_rom_palette[ 1] = RGB(15, 15, 15); // white
133  vic3_rom_palette[ 2] = RGB(15, 0, 0); // red
134  vic3_rom_palette[ 3] = RGB( 0, 15, 15); // cyan
135  vic3_rom_palette[ 4] = RGB(15, 0, 15); // magenta
136  vic3_rom_palette[ 5] = RGB( 0, 15, 0); // green
137  vic3_rom_palette[ 6] = RGB( 0, 0, 15); // blue
138  vic3_rom_palette[ 7] = RGB(15, 15, 0); // yellow
139  vic3_rom_palette[ 8] = RGB(15, 6, 0); // orange
140  vic3_rom_palette[ 9] = RGB(10, 4, 0); // brown
141  vic3_rom_palette[10] = RGB(15, 7, 7); // pink
142  vic3_rom_palette[11] = RGB( 5, 5, 5); // dark grey

148  // It may also sets/initializes some internal variables sets by register writes, which would cause a crash on screen rendering without prior setup!
149  for (i = 0; i < 0x140; i++) {
150  vic_write_reg(i, 0);
151  (void)vic_read_reg(i);
152  }
153  c128_d030_reg = 0xFE; // this may be set to 2MHz in the previous step, so be sure to set to FF here, BUT FIX: bit 0 should be inverted!!
154  machine_set_speed(0);
155  DEBUG("VIC4: has been initialized." NL);
156 }

160 static void vic3_interrupt_checker ( void )
161 {
162  int vic_irq_old = cpu65.irqLevel & 2;
163  int vic_irq_new;
164  if ((interrupt_status & vic_registers[0x1A])) {
165  interrupt_status |= 128;
166  vic_irq_new = 2;
167  } else {
168  interrupt_status &= 127;
169  vic_irq_new = 0;
170  }
171  if (vic_irq_old != vic_irq_new) {
172  DEBUG("VIC3: interrupt change %s -> %s" NL, vic_irq_old ? "active" : "inactive", vic_irq_new ? "active" : "inactive");
173  if (vic_irq_new)
174  cpu65.irqLevel |= 2;
175  else
176  cpu65.irqLevel &= ~2;
177  }
178 }

193 /* DESIGN of vic_read_reg() and vic_write_reg() functions:
194  addr = 00-7F, VIC-IV registers 00-7F (ALWAYS, regardless of current I/O mode!)
195  addr = 80-FF, VIC-III registers 00-7F (ALWAYS, regardless of current I/O mode!) [though for VIC-III, many registers are ignored after the last one]
196  addr = 100-13F, VIC-II registers 00-3F (ALWAYS, regardless of current I/O mode!)
197  NOTES:
198  * on a real VIC-II last used register is $2E. However we need the KEY register ($2F) and the C128-style 2MHz mode ($30) on M65 too.
199  * ALL cases must be handled!! from 000-13F for both of reading/writing funcs, otherwise Xemu will panic! this is a safety stuff
200  * on write, later an M65-alike solution is needed: ie "hot registers" for VIC-II,VIC-III also writes VIC-IV specific registers then
201  * currently MANY things are not handled, it will be the task of "move to VIC-IV internals" project ...
202  * the purpose of ugly "tons of case" implementation that it should compile into a simple jump-table, which cannot be done faster too much ...
203  * do not confuse these "vic reg mode" ranges with the vic_iomode variable, not so much direct connection between them! vic_iomode referred
204  for the I/O mode used on the "classic $D000 area" and DMA I/O access only
205 */
