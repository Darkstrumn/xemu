24 /* NOTES ABOUT C65/M65 DMAgic "F018", AND THIS EMULATION:
25 
26  * this emulation part uses externally supplied functions and it depends on the target emulator how to implement read/write ops by DMA!
27  * modulo currently not handled, and it seems there is not so much decent specification how it does work
28  * INT / interrupt is not handled
29  * MIX command is implemented, though without decent specification is more like a guessing only
30  * SWAP command is also a guessing, but a better one, as it's kinda logical what it is used for
31  * COPY/FILL should be OK
32  * DMA length of ZERO probably means $10000 [but not so much info on this ...] FIXME?
33  * Current emulation timing is incorrect
34  * Speed of DMA in a real C65 would be affected by many factors, kind of op, or even VIC-3 fetchings etc ... "Of course" it's not emulated ...
35  * DMA sees the upper memory regions (ie in BANK 4 and above), which may have "something", though REC (Ram Expansion Controller) is not implemented yet
36  * It's currently unknown that DMA registers are modified as DMA list is read, ie what happens if only reg 0 is re-written
37  * C65 specification tells about "not implemented sub-commands": I am curious what "sub commands" are or planned as, etc ...
38  * Reading status would resume interrupted DMA operation (?) it's not emulated
39  * MEGA65 macro is defined in case of M65 target, then we handle the "megabyte slice stuff" as well.
40  * FIXME: I am not sure what's the truth: DMA session "warps" within the 1Mbyte address range or within the 64K address set?!
41  Since M65 VHDL says within 64K, I go the same here currently ...
42 */

91 // In case of Mega65 we should support fractional steps (1 byte for fraction). We do this to have
92 // common code base with C65 emulator's DMA to have a single variable for source (source_addr) and
93 // target (target_addr) "within the megabyte" variable, but in case of C65 it's the "clean" address,
94 // while in case of Mega-65, the lower 8 bits are the fraction. This macro is used to reference the
95 // integer part only, in case of C65 it's simply as-is, for Mega65, it's skipping the low 8 bits, ie
96 // shifting data to the right by 8. Note, that since within a megabyte DMA can operate only, 32 bits
97 // are more than enough even this way (20 bits + 8 bits = 28 bits, even signed 32 bit int type is
98 // suitable).
99 // DMA_*_SKIP_RATE macros just gives the step rate (always positive). In case of C65, it's of course
100 // fixed to be 1 ('hold' is not handled this way). For Mega65, we use (see the source fix-point
101 // arithmetic description above!) the actual M65 DMA registers for those settings.
102 // DMA_ADDR_FRACT_PART() is used only in debug functions to log
103 #ifdef MEGA65
104 #define DMA_ADDR_INTEGER_PART(p) ((p)>>8)
105 #define DMA_SOURCE_SKIP_RATE ((int)(dma_registers[0x08] | (dma_registers[0x09] << 8)))
106 #define DMA_TARGET_SKIP_RATE ((int)(dma_registers[0x0A] | (dma_registers[0x0B] << 8)))
107 #define DMA_ADDR_FRACT_PART(p) ((p) & 0xFF)
108 #else
109 #define DMA_ADDR_INTEGER_PART(p) (p)
110 #define DMA_SOURCE_SKIP_RATE 1
111 #define DMA_TARGET_SKIP_RATE 1
112 #define DMA_ADDR_FRACT_PART(p) 0
113 #endif

134 #define DMA_READ_SOURCE() (XEMU_UNLIKELY(source.is_io) ? DMA_SOURCE_IOREADER_FUNC(DMA_ADDRESSING(source)) : DMA_SOURCE_MEMREADER_FUNC(DMA_ADDRESSING(source)))
135 #define DMA_READ_TARGET() (XEMU_UNLIKELY(target.is_io) ? DMA_TARGET_IOREADER_FUNC(DMA_ADDRESSING(target)) : DMA_TARGET_MEMREADER_FUNC(DMA_ADDRESSING(target)))
136 #define DMA_WRITE_SOURCE(data) do { \
137  if (XEMU_UNLIKELY(source.is_io)) \
138  DMA_SOURCE_IOWRITER_FUNC(DMA_ADDRESSING(source), data); \

141  } while (0)

179 static XEMU_INLINE void mix_next ( void )
180 {
181  Uint8 sa = DMA_READ_SOURCE();
182  Uint8 da = DMA_READ_TARGET();
183  // NOTE: it's not clear from the specification, what MIX
184  // does. I assume, that it does some kind of minterm
185  // with source and target and writes the result to
186  // target. I'm not even sure how the minterms are
187  // interpreted on the bits of two bytes too much. FIXME!!!
188  da =
189  (( sa) & ( da) & minterms[3]) |
190  (( sa) & (~da) & minterms[2]) |
191  ((~sa) & ( da) & minterms[1]) |
192  ((~sa) & (~da) & minterms[0]) ;
193  DMA_WRITE_TARGET(da);
194  source.addr += source.step;
195  target.addr += target.step;
196 }

203 void dma_write_reg ( int addr, Uint8 data )
204 {
205  // DUNNO about DMAgic too much. It's merely guessing from my own ROM assembly tries, C65gs/Mega65 VHDL, and my ideas :)
206  // The following condition is commented out for now. FIXME: how it is handled for real?!
207  //if (vic_iomode != VIC4_IOMODE)
208  // addr &= 3;
209  if (XEMU_UNLIKELY(in_dma_update)) {
210  // this is just an emergency stuff to disallow DMA to update its own registers ... FIXME: what would be the correct policy?
211  // NOTE: without this, issuing a DMA transfer updating DMA registers would affect an on-going DMA transfer!
212  if (dma_self_write_warning) {
213  dma_self_write_warning = 0;
214  ERROR_WINDOW("DMA writes its own registers, ignoring!\nThere will be no more warning on this!");
215  }
216  DEBUG("DMA: WARNING: tries to write own register by DMA reg#%d with value of $%02X" NL, addr, data);
217  return;
218  }
219 #ifdef MEGA65
220  dma_registers[addr] = data;
221  hack.enhanced_dma = 0;
222  switch (addr) {
223  case 0x2: // for compatibility with C65, Mega65 here resets the MB part of the DMA list address
224  dma_registers[4] = 0; // this is the "MB" part of the DMA list address (hopefully ...)
225  break;
226  case 0xE: // Set low order bits of DMA list address, without starting (Mega65 feature, but without VIC4 new mode, this reg will never addressed here anyway)
227  dma_registers[0] = data;
228  break;
229  case 5:
230  if (hack.enabled) {
231  addr = 0; // fool the code that it's a DMA trigger ...
232  hack.enhanced_dma = 1; // ... and also set the enhanced mode flag though
233  dma_registers[0] = data;
234  dma_registers[5] = data;
235  }
236  break;
237  }
238 #else
239  if (addr > 3) { // in case of C65, the extended registers cannot be written
240  DEBUG("DMA: trying to write M65-specific DMA register (%d) with a C65 ..." NL, addr);
241  return;
242  }
243  dma_registers[addr] = data;
244 #endif
245  if (addr)
246  return; // Only writing register 0 starts the DMA operation, otherwise just return from this function (reg write already happened)
247  if (XEMU_UNLIKELY(dma_status))
248  FATAL("dma_write_reg(): new DMA op with dma_status != 0");
249  list_addr = dma_registers[0] | (dma_registers[1] << 8); // | ((dma_registers[2] & 0xF) << 16);
250 #ifdef MEGA65
251  if (hack.enhanced_dma) {
252  DEBUGDMA("DMA: initiation of ENCHANCED MODE DMA!!!!\n");
253  // FAKE enhanced mode DMA! By using old register-set pre-inited, and allow
254  // an enhanced option list to modify them :-@
255  dma_registers[0x08] = 0; // source skip rate, fraction part
256  dma_registers[0x09] = 1; // source skip rate, integer part
257  dma_registers[0x0A] = 0; // target skip rate, fraction part
258  dma_registers[0x0B] = 1; // target skip rate, integer part
259  dma_registers[0x05] = 0; // source MB
260  dma_registers[0x06] = 0; // target MB
261  hack.saved_revision = dma_chip_revision; // save this (will be restored at the end of DMA chain) so the "hack" can modify without loosing the original setting
262  } else
263  DEBUGDMA("DMA: initiation of normal mode DMA\n");
264  list_base = (dma_registers[4] << 20) | ((dma_registers[2] & 0xF) << 16);
265 #else
266  list_base = (dma_registers[2] & 0xF) << 16;
267 #endif
268  DEBUGDMA("DMA: list address is $%06X now, just written to register %d value $%02X @ PC=$%04X" NL, list_base | list_addr, addr, data, cpu65.pc);
269  dma_status = 0x80; // DMA is busy now, also to signal the emulator core to call dma_update() in its main loop
270  command = -1; // signal dma_update() that it's needed to fetch the DMA command, no command is fetched yet
271  cpu65.multi_step_stop_trigger = 1; // trigger stopping multi-op CPU emulation mode, otherwise "delayed DMAs" would overlap each other resulting "panic"
272 }

276 /* Main emulation loop should call this function regularly, if dma_status is not zero.
277  This way we have 'real' DMA, ie works while the rest of the machine is emulated too.
278  Please note, that the "exact" timing of DMA and eg the CPU is still incorrect, but it's far
279  better than the previous version where DMA was "blocky", ie the whole machine was "halted" while DMA worked ...
280  NOTE: there was an ugly description here about the differences between F018A,F018B. it's a too big topic
281  for a comment here, so it has been deleted. See here: http://c65.lgb.hu/dma.html */
282 int dma_update ( void )
283 {
284 #ifdef MEGA65
285 #endif
286  Uint8 subcommand;
287  int cycles = 0;
288  if (XEMU_UNLIKELY(!dma_status))
289  FATAL("dma_update() called with no dma_status set!");
290  if (XEMU_UNLIKELY(command == -1)) {
291 #ifdef MEGA65
292  /* don't do this here, it will reset these in chained list for enhanced DMA!
293  if (hack.enabled) {
294  dma_registers[0x08] = 0; // source skip rate, fraction part
295  dma_registers[0x09] = 1; // source skip rate, integer part
296  dma_registers[0x0A] = 0; // target skip rate, fraction part
297  dma_registers[0x0B] = 1; // target skip rate, integer part
298  dma_registers[0x05] = 0; // source MB
299  dma_registers[0x06] = 0; // target MB
300  } */
301  if (hack.enhanced_dma) {
302  Uint8 opt;
303  //memcpy(hack.saved_regs, dma_registers, sizeof(dma_registers));
304  hack.saved_revision = dma_chip_revision;
305  do {
306  opt = DMA_READ_LIST_NEXT_BYTE();
307  DEBUGDMA("DMA: enhanced option byte $%02X read" NL, opt);
308  cycles++;
309  switch (opt) {
310  case 0x86: // not supported yet
311  list_addr++; // skip the parameter too
312  cycles++;
313  // flow onto the next 'case'!!
314  case 0x06:
315  case 0x07:
316  DEBUGPRINT("DMA: enhanced DMA transparency is not supported yet (option=$%02X) @ PC=$%04X" NL, opt, cpu65.pc);
317  break;
318  case 0x0A:
319  dma_chip_revision_override = 0;
320  break;
321  case 0x0B:
322  dma_chip_revision_override = 1;
323  break;
324  case 0x80: // set MB of source
325  dma_registers[0x05] = DMA_READ_LIST_NEXT_BYTE();
326  cycles++;
327  break;
328  case 0x81: // set MB of target
329  dma_registers[0x06] = DMA_READ_LIST_NEXT_BYTE();
330  cycles++;
331  break;
332  case 0x82: // set source skip rate, fraction part
333  dma_registers[0x08] = DMA_READ_LIST_NEXT_BYTE();
334  cycles++;
335  break;
336  case 0x83: // set source skip rate, integer part
337  dma_registers[0x09] = DMA_READ_LIST_NEXT_BYTE();
338  cycles++;
339  break;
340  case 0x84: // set target skip rate, fraction part
341  dma_registers[0x0A] = DMA_READ_LIST_NEXT_BYTE();
342  cycles++;
343  break;
344  case 0x85: // set target skip rate, integer part
345  dma_registers[0x0B] = DMA_READ_LIST_NEXT_BYTE();
346  cycles++;
347  break;
348  case 0x00:
349  DEBUGDMA("DMA: end of enhanced options" NL);
350  break;
351  default:
352  // maybe later we should keep this quiet ...
353  DEBUGPRINT("DMA: *unknown* enhanced option: $%02X @ PC=$%04X" NL, opt, cpu65.pc);
354  break;
355  }
356  } while (opt);
357  }
358 #endif
359  // command == -1 signals the situation, that the (next) DMA command should be read!
360  // This part is highly incorrect, ie fetching so many bytes in one step only of dma_update()
361 #ifdef MEGA65
362  // set DMA revision based on register #3 bit 0, this is a Mega65 feature
363  if (dma_chip_revision_override >= 0) {
364  DEBUG("DMA: changing DMA revision by enhanced list %d -> %d" NL, dma_chip_revision, dma_chip_revision_override);
365  dma_chip_revision = dma_chip_revision_override;
366  } else if (dma_chip_revision != (dma_registers[3] & 1)) {
367  if (dma_chip_revision_is_dynamic) {
368  DEBUG("DMA: changing DMA revision %d -> %d" NL, dma_chip_revision, dma_registers[3] & 1);
369  dma_chip_revision = dma_registers[3] & 1;
370  } else {
371  DEBUG("DMA: WARNING: M65 software wants to change DMA revision, but you did not allow it!" NL);
372  }
373  }
374 #endif
375  // NOTE: in case of Mega65: DMA_READ_LIST_NEXT_BYTE() uses the "megabyte" part already (taken from reg#4, in case if that reg is written)
376  command = DMA_READ_LIST_NEXT_BYTE();
377  dma_op = (enum dma_op_types)(command & 3);
378  modulo.col_limit = DMA_READ_LIST_NEXT_BYTE();
379  modulo.row_limit = DMA_READ_LIST_NEXT_BYTE();
380  length = modulo.col_limit | (modulo.row_limit << 8);
381  filler_byte = DMA_READ_LIST_NEXT_BYTE() ; // source low byte is also the filler byte in case of FILL command
382  source.addr =(DMA_READ_LIST_NEXT_BYTE() << 8) | filler_byte; // -- "" --
383  source.addr |= DMA_READ_LIST_NEXT_BYTE() << 16;
384  target.addr = DMA_READ_LIST_NEXT_BYTE() ;
385  target.addr |= DMA_READ_LIST_NEXT_BYTE() << 8;
386  target.addr |= DMA_READ_LIST_NEXT_BYTE() << 16;
387  if (dma_chip_revision) // for F018B we have an extra byte fetch here! used later in this function [making DMA list one byte longer, indeed]
388  subcommand = DMA_READ_LIST_NEXT_BYTE();
389  else
390  subcommand = 0; // just make gcc happy not to generate warning later
391 #ifdef MEGA65
392  // On Mega65, modulo is used as a fixed point arithmetic value
393  modulo.value = DMA_READ_LIST_NEXT_BYTE() << 8;
394  modulo.value |= DMA_READ_LIST_NEXT_BYTE() << 16;
395 #else
396  modulo.value = DMA_READ_LIST_NEXT_BYTE() ;
397  modulo.value |= DMA_READ_LIST_NEXT_BYTE() << 8;
398 #endif
399  if (dma_chip_revision) {
400  cycles += 12; // FIXME: correct timing?
401  // F018B ("new") behaviour
402  source.step = (subcommand & 2) ? 0 : ((command & 16) ? -DMA_SOURCE_SKIP_RATE : DMA_SOURCE_SKIP_RATE);
403  target.step = (subcommand & 8) ? 0 : ((command & 32) ? -DMA_TARGET_SKIP_RATE : DMA_TARGET_SKIP_RATE);
404  source.is_modulo = (subcommand & 1);
405  target.is_modulo = (subcommand & 4);
406  if (dma_op == MIX_OP) { // if it's a MIX command
407  // FIXME: what about minterms in F018B?! Maybe upper bits of subcommand, but we can't [?] know ... Try that theory for now ...
408  minterms[0] = (subcommand & 16) ? 0xFF : 0x00;
409  minterms[1] = (subcommand & 32) ? 0xFF : 0x00;
410  minterms[2] = (subcommand & 64) ? 0xFF : 0x00;
411  minterms[3] = (subcommand & 128) ? 0xFF : 0x00;
412  }
413  } else {
414  cycles += 11; // FIXME: correct timing?
415  // F018A ("old") behaviour
416  source.step = (source.addr & 0x100000) ? 0 : ((source.addr & 0x400000) ? -DMA_SOURCE_SKIP_RATE : DMA_SOURCE_SKIP_RATE);
417  target.step = (target.addr & 0x100000) ? 0 : ((target.addr & 0x400000) ? -DMA_TARGET_SKIP_RATE : DMA_TARGET_SKIP_RATE);
418  source.is_modulo = (source.addr & 0x200000);
419  target.is_modulo = (target.addr & 0x200000);
420  if (dma_op == MIX_OP) { // if it's a MIX command
421  minterms[0] = (command & 16) ? 0xFF : 0x00;
422  minterms[1] = (command & 32) ? 0xFF : 0x00;
423  minterms[2] = (command & 64) ? 0xFF : 0x00;
424  minterms[3] = (command & 128) ? 0xFF : 0x00;
425  }
426  }
427  // use modulo mode if:
428  // * dma_init() is used with this feature to be enabled
429  // * any of source or target has the MODulo bit set
430  // FIXME: logically, in case of FILL command, source uses modulo setting does not make sense, and we don't want to use modulo length counting at all
431  if (XEMU_UNLIKELY(source.is_modulo || target.is_modulo)) {
432  if (modulo.enabled) {
433  modulo.used = 1;
434  modulo.col_counter = 0;
435  modulo.row_counter = 0;
436  // GUESSING/FIXME/TODO: with modulo, col/row counters = 0 means $100 for real, as with non-modulo mode counter = 0 means $10000 (16 bit counter there, 8 bit wides here!)
437  if (!modulo.col_limit)
438  modulo.col_limit = 0x100;
439  if (!modulo.row_limit)
440  modulo.row_limit = 0x100;
441  } else {
442  modulo.used = 0;
443  DEBUGPRINT("DMA: warning, MODulo mode wanted to be used, but not enabled by you!" NL);
444  }
445  } else
446  modulo.used = 0;
447  // if modulo.used is zero, source.is_modulo and target.is_modulo is never used (but maybe for logging)
448  // so commenting the part below:
449 #if 0
450  if (!modulo.used) {
451  if (source.is_modulo || target.is_modulo)
452  DEBUG("DMA: denying using MODulo ..." NL);
453  source.is_modulo = 0;
454  target.is_modulo = 0;
455  }
456 #endif
457  // It *seems* I/O stuff is still in the place even with F018B. FIXME: is it true?
458  source.is_io = (source.addr & 0x800000);
459  target.is_io = (target.addr & 0x800000);
460  /* source selection */
461  if (source.is_io) {
462  source.mask = 0xFFF; // 4K I/O size (warps within 4K only)
463  source.base = 0; // in case of I/O, base is not interpreted in Xemu (uses pure numbers 0-$FFF, no $DXXX, not even M65-spec mapping), and must be zero ...
464 #ifdef MEGA65
465  source.addr = (source.addr & 0xFFF) << 8; // for M65, it is fixed-point arithmetic
466 #else
467  source.addr = (source.addr & 0xFFF) ; // for C65, it is pure number, no fixed-point arith. here
468 #endif
469  } else {
470  source.mask = 0xFFFF; // warp around within 64K. I am still not sure, it happens with DMA on 64K or 1M. M65 VHDL does 64K, so I switched that too.
471 #ifdef MEGA65
472  // base selection for M65
473  // M65 has an "mbyte part" register for source (and target too)
474  // however, with F018B there are 3 bits over 1Mbyte as well, and it seems M65 (see VHDL code) add these together then. Interesting.
475  if (dma_chip_revision)
476  source.base = (source.addr & 0x0F0000) | (((dma_registers[5] << 20) + (source.addr & 0x700000)) & 0xFF00000);
477  else
478  source.base = (source.addr & 0x0F0000) | ( dma_registers[5] << 20);
479  source.addr = (source.addr & 0x00FFFF) << 8;// offset from base, for M65 this *IS* fixed point arithmetic!
480 #else
481  if (dma_chip_revision)
482  source.base = (source.addr & 0x7F0000) ; // base selection for C65, in case of F018B (3 more bits of addresses)
483  else
484  source.base = (source.addr & 0x0F0000) ; // base selection for C65, in case of F018A
485  source.addr = (source.addr & 0x00FFFF) ; // offset from base, for C65 this is pure number, *NO* fixed point arithmetic here!
486 #endif
487  }
488  /* target selection - see similar lines with comments above, for source ... */
489  if (target.is_io) {
490  target.mask = 0xFFF;
491  target.base = 0;
492 #ifdef MEGA65
493  target.addr = (target.addr & 0xFFF) << 8;
494 #else
495  target.addr = (target.addr & 0xFFF) ;
496 #endif
497  } else {
498  target.mask = 0xFFFF;
499 #ifdef MEGA65
500  if (dma_chip_revision)
501  target.base = (target.addr & 0x0F0000) | (((dma_registers[6] << 20) + (target.addr & 0x700000)) & 0xFF00000);
502  else
503  target.base = (target.addr & 0x0F0000) | ( dma_registers[6] << 20);
504  target.addr = (target.addr & 0x00FFFF) << 8;
505 #else
506  if (dma_chip_revision)
507  target.base = (target.addr & 0x7F0000) ;
508  else
509  target.base = (target.addr & 0x0F0000) ;
510  target.addr = (target.addr & 0x00FFFF) ;
511 #endif
512  }
513  /* other stuff */
514  chained = (command & 4);
515  // FIXME: this is a debug mesg, yeah, but with fractional step on M65, the step values needs to be interpreted with keep in mind the fixed point math ...
516  DEBUG("DMA: READ COMMAND: $%07X[%s%s %d:%d] -> $%07X[%s%s %d:%d] (L=$%04X) CMD=%d (%s)" NL,
517  DMA_ADDRESSING(source), source.is_io ? "I/O" : "MEM", source.is_modulo ? " MOD" : "", DMA_ADDR_INTEGER_PART(source.step), DMA_ADDR_FRACT_PART(source.step),
518  DMA_ADDRESSING(target), target.is_io ? "I/O" : "MEM", target.is_modulo ? " MOD" : "", DMA_ADDR_INTEGER_PART(target.step), DMA_ADDR_FRACT_PART(target.step),
519  length, dma_op, chained ? "CHAINED" : "LAST"
520  );
521  if (!length)
522  length = 0x10000; // I *think* length of zero means 64K. Probably it's not true!!
523  return cycles;
524  }
525  // We have valid command to be executed, or continue to execute
526  //DEBUG("DMA: EXECUTING: command=%d length=$%04X" NL, command & 3, length);
527  in_dma_update = 1;
528  switch (dma_op) {
529  case COPY_OP: // COPY command (0)
530  copy_next();
531  cycles = 2; // FIXME: correct timing?
532  break;
533  case MIX_OP: // MIX command (1)
534  mix_next();
535  cycles = 3; // FIXME: correct timing?
536  break;
537  case SWAP_OP: // SWAP command (2)
538  swap_next();
539  cycles = 4; // FIXME: correct timing?
540  break;
541  case FILL_OP: // FILL command (3)
542  fill_next();
543  cycles = 1; // FIXME: correct timing?
544  break;
545  }
546  // Maintain length counter or modulo-related counters (in the second case, also add modulo value if needed)
547  if (XEMU_UNLIKELY(modulo.used)) {
548  // modulo mode. we don't use the usual "length" but own counters.
549  // but we DO reset "length" to zero if modulo transfer is done, so
550  // it will be detected by the 'if' at the bottom of this huge function as the end of the DMA op
551  if (modulo.col_counter == modulo.col_limit) {
552  if (modulo.row_counter == modulo.row_limit) {
553  length = 0; // just to fullfish end-of-operation condition
554  } else {
555  // end of modulo "columns", but there are "rows" left, reset columns counter, increment row counter,
556  // also add the modulo value to source and/or target channels, if the given channel is in modulo mode
557  modulo.col_counter = 0;
558  modulo.row_counter++;
559  if (source.is_modulo)
560  source.addr += modulo.value;
561  if (target.is_modulo)
562  target.addr += modulo.value;
563  }
564  } else {
565  modulo.col_counter++;
566  }
567  } else {
568  length--; // non-MODulo mode, we simply decrement length counter ...
569  }
570  if (length <= 0) { // end of DMA operation for the current DMA list entry?
571  if (chained) { // chained?
572  DEBUGDMA("DMA: end of operation, but chained!" NL);
573  dma_status = 0x81; // still busy then, with also bit0 set (chained)
574  command = -1; // signal for next DMA command fetch
575  } else {
576  DEBUGDMA("DMA: end of operation, no chained next one." NL);
577  dma_status = 0; // end of DMA command
578  command = -1;
579 #ifdef MEGA65
580  if (hack.enhanced_dma) {
581  DEBUGDMA("DMA: enhanced-end-of-op, restoring context" NL);
582  //memcpy(dma_registers, hack.saved_regs, sizeof(dma_registers));
583  dma_chip_revision = hack.saved_revision;
584  hack.enhanced_dma = 0;
585  }
586  dma_chip_revision_override = -1;
587  // Mega65: reset fractional step registers to the default at the end! (it seems M65 does this, by reading its VHDL source)
588  // Note, this is the old DMA behaviour not connected to the "hack" ...
589  dma_registers[0x08] = 0; // source skip rate, fraction part
590  dma_registers[0x09] = 1; // source skip rate, integer part
591  dma_registers[0x0A] = 0; // target skip rate, fraction part
592  dma_registers[0x0B] = 1; // target skip rate, integer part
593 #endif
594  }
595  }
596  in_dma_update = 0;
597  return cycles;
598 }
