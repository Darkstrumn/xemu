39 static inline void ifr_check(struct Via65c22 *via)
40 {
41  if (via->IFR & via->IER & 127) {
42  via->IFR |= 128;
43  if (!via->irqLevel) INT(via, 1);
44  } else {
45  via->IFR &= 127;
46  if ( via->irqLevel) INT(via, 0);
47  }
48 }

50 static inline void ifr_clear (struct Via65c22 *via, Uint8 mask) { via->IFR &= 255 - mask; ifr_check(via); }
51 static inline void ifr_set (struct Via65c22 *via, Uint8 mask) { via->IFR |= mask; ifr_check(via); }
52 static inline void ifr_on_pa (struct Via65c22 *via) { ifr_clear(via, ((via->PCR & 0x0E) == 0x02 || (via->PCR & 0x0E) == 0x06) ? 2 : 3); }
53 static inline void ifr_on_pb (struct Via65c22 *via) { ifr_clear(via, ((via->PCR & 0xE0) == 0x20 || (via->PCR & 0xE0) == 0x60) ? 0x10 : 0x18); }

101 void via_write(struct Via65c22 *via, int addr, Uint8 data)
102 {
103  //DEBUG("%s: write reg %02X with data %02X" NL, via->name, addr, data);
104  switch (addr) {
105  case 0x0: // port B data
106  via->ORB = data; // FIXED BUG
107  OUTB(via, data);
108  ifr_on_pb(via);
109  break;
110  case 0x1: // port A data
111  via->ORA = data; // FIXED BUG
112  OUTA(via, data);
113  ifr_on_pa(via);
114  break;
115  case 0x2: // port B DDR
116  if (data != via->DDRB) {
117  via->DDRB = data;
118  OUTB(via, via->ORB);
119  }
120  break;
121  case 0x3: // port A DDR
122  if (data != via->DDRA) {
123  via->DDRA = data;
124  OUTA(via, via->ORA);
125  }
126  break;
127  case 0x4: //
128  via->T1LL = data;
129  break;
130  case 0x5:
131  via->T1LH = data;
132  ifr_clear(via, 64);
133  via->T1run = 1;
134  via->T1C = via->T1LL | (via->T1LH << 8);
135  break;
136  case 0x6:
137  via->T1LL = data;
138  break;
139  case 0x7:
140  via->T1LH = data;
141  ifr_clear(via, 64);
142  break;
143  case 0x8:
144  via->T2LL = data;
145  break;
146  case 0x9:
147  via->T2LH = data;
148  ifr_clear(via, 32);
149  via->T2run = 1;
150  via->T2C = via->T2LL | (via->T2LH << 8);
151  break;
152  case 0xA: // SR
153  ifr_clear(via, 4);
154  via->SR = data;
155  if (via->SRmode) via->SRcount = 8;
156  break;
157  case 0xB: // ACR
158  via->SRmode = (data >> 2) & 7;
159  via->ACR = data;
160  if (data & 32) alert(via, "pulse counting T2 mode is not supported!");
161  break;
162  case 0xC: // PCR
163  via->PCR = data;
164  break;
165  case 0xD: // IFR
166  ifr_clear(via, data);
167  break;
168  case 0xE: // IER
169  if (data & 128) via->IER |= data; else via->IER &= (255 - data);
170  ifr_check(via);
171  break;
172  case 0xF: // port A data (no handshake)
173  OUTA(via, data);
174  ifr_on_pa(via);
175  break;
176  }
177 }

179 Uint8 via_read(struct Via65c22 *via, int addr)
180 {
181  //DEBUG("%s: read reg %02X" NL, via->name, addr);
182  switch (addr) {
183  case 0x0: // port B data
184  ifr_on_pb(via);
185  return INB(via);
186  case 0x1: // port A data
187  ifr_on_pa(via);
188  return INA(via);
189  case 0x2: // port B DDR
190  return via->DDRB;
191  case 0x3: // port A DDR
192  return via->DDRA;
193  case 0x4: //
194  ifr_clear(via, 64);
195  return via->T1C & 0xFF;
196  case 0x5:
197  return via->T1C >> 8;
198  case 0x6:
199  return via->T1LL;
200  case 0x7:
201  return via->T1C >> 8;
202  case 0x8:
203  ifr_clear(via, 32);
204  return via->T2C & 0xFF;
205  case 0x9:
206  return via->T2C >> 8;
207  case 0xA: // SR
208  ifr_clear(via, 4);
209  if (via->SRmode) via->SRcount = 8;
210  return via->SR;
211  case 0xB: // ACR
212  return via->ACR;
213  case 0xC: // PCR
214  return via->PCR;
215  case 0xD: // IFR
216  return via->IFR;
217  case 0xE: // IER
218  return via->IER | 128;
219  case 0xF: // port A data (no handshake)
220  ifr_on_pa(via);
221  return INA(via);
222  }
223  return 0; // make gcc happy :)
224 }

226 void via_tick(struct Via65c22 *via, int ticks)
227 {
228  /* T1 */
229  if (via->T1run) {
230  via->T1C -= ticks;
231  if (via->T1C <= 0) {
232  //console.log("Expired T1");
233  ifr_set(via, 64);
234  if (via->ACR & 64) via->T1C = via->T1LL | (via->T1LH << 8); else via->T1run = 0;
235  }
236  }
237  /* T2 */
238  via->T2C -= ticks;
239  if (via->T2run) {
240  if (via->T2C <= 0) {
241  ifr_set(via, 32);
242  via->T2run = 0;
243  }
244  }
245  via->T2C &= 0xFFFF;
246  /* shift register */
247  if (via->SRcount) {
248  via->SRcount -= ticks;
249  if (via->SRcount <= 0) {
250  switch(via->SRmode) {
251  case 0: via->SRcount = 0; ifr_clear(via, 4); break; // disabled
252  case 2: via->SR = INSR(via); via->SRcount = 0; ifr_set(via, 4); break; // PHI2-in
253  case 4: OUTSR(via, via->SR); via->SRcount = via->T2LL + 2; ifr_clear(via, 4); break; // free-T2-out, VIA2 seems to use this mode!
254  default: via->SRcount = 0 ; ifr_clear(via, 4); alert(via, "SRmode is not supported!"); break;
255  }
256  }
257  }
258 }
