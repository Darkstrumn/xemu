2  level needed for a special version of GEOS to be able to run on it.

182 static void vic2_interrupt_checker ( void )
183 {
184  int vic_irq_old = cpu65.irqLevel & 2;
185  int vic_irq_new;
186  if (vic2_interrupt_status & vic2_registers[0x1A]) {
187  vic2_interrupt_status |= 128;
188  vic_irq_new = 2;
189  } else {
190  vic2_interrupt_status &= 127;
191  vic_irq_new = 0;
192  }
193  if (vic_irq_old != vic_irq_new) {
194  DEBUG("VIC2: interrupt change %s -> %s" NL, vic_irq_old ? "active" : "inactive", vic_irq_new ? "active" : "inactive");
195  if (vic_irq_new)
196  cpu65.irqLevel |= 2;
197  else
198  cpu65.irqLevel &= ~2;
199  }
200 }

204 void vic2_check_raster_interrupt ( void )
205 {
206  // I'm lame even with VIC2 knowledge it seems
207  // C65 seems to use raster interrupt to generate the usual periodic IRQ
208  // (which was done with CIA on C64) in raster line 511. However as
209  // raster line 511 can never be true, I really don't know what to do.
210  // To be able C65 ROM to work, I assume that raster 511 is raster 0.
211  // It's possible that this is an NTSC/PAL issue, as raster can be "negative"
212  // according to the specification in case of NTSC. I really don't know ...
213  if (scanline == compare_raster)
214  vic2_interrupt_status |= 1;
215  else
216  vic2_interrupt_status &= 0xFE;
217  vic2_interrupt_checker();
218 }

223 void vic2_write_reg ( int addr, Uint8 data )
224 {
225  addr &= 0x3F;
226  DEBUG("VIC3: write reg $%02X with data $%02X" NL, addr, data);
227  if (addr > 0x2E)
228  return;
229  vic2_registers[addr] = data;
230  switch (addr) {
231  case 0x11:
232  compare_raster = (compare_raster & 0xFF) | ((data & 128) ? 0x100 : 0);
233  DEBUG("VIC2: compare raster is now %d" NL, compare_raster);
234  break;
235  case 0x12:
236  compare_raster = (compare_raster & 0xFF00) | data;
237  DEBUG("VIC2: compare raster is now %d" NL, compare_raster);
238  break;
239  case 0x19:
240  vic2_interrupt_status = vic2_interrupt_status & (~data) & 15;
241  vic2_interrupt_checker();
242  break;
243  case 0x1A:
244  vic2_registers[0x1A] &= 15;
245  break;
246  }
247 }

252 Uint8 vic2_read_reg ( int addr )
253 {
254  Uint8 result;
255  addr &= 0x3F;
256  switch (addr) {
257  case 0x11:
258  result = (vic2_registers[0x11] & 0x7F) | ((scanline & 256) ? 0x80 : 0);
259  break;
260  case 0x12:
261  result = scanline & 0xFF;
262  break;
263  case 0x16:
264  result = vic2_registers[addr] | (128 + 64); // unused bits
265  break;
266  case 0x19:
267  result = vic2_interrupt_status | (64 + 32 + 16); // unused bits
268  break;
269  case 0x1A:
270  result = vic2_registers[addr] | 0xF0; // unused bits
271  break;
272  case 0x18:
273  result = vic2_registers[addr] | 1; // unused bit
274  break;
275  default:
276  result = vic2_registers[addr];
277  if (addr >= 0x20 && addr < 0x2F)
278  result |= 0xF0; // unused bits
279  break;
280  }
281  DEBUG("VIC2: read reg $%02X with result $%02X" NL, addr, result);
282  return result;
283 }

291 static inline void vic2_render_screen_text ( Uint32 *p, int tail )
292 {
293  Uint32 bg;

297  // and fixed chargen address.
298  // Fixed character info, heh ... FIXME
299  chrg = memory + CHAR_ROM_OFFSET;
300  // Note: VIC2 sees ROM at some addresses thing is not emulated yet!
301  vidp = memory + ((vic2_registers[0x18] & 0xF0) << 6) + vic2_16k_bank;
302  vic2_sprite_pointers = vidp + 1016;
303  // Target SDL pixel related format for the background colour
304  bg = palette[vic2_registers[0x21] & 15];
305  PIXEL_POINTER_CHECK_INIT(p, tail, "vic2_render_screen_text");
306  for (;;) {
307  Uint8 chrdata = chrg[((*(vidp++)) << 3) + charline];
308  Uint8 coldata = *(colp++);
309  Uint32 fg = palette[coldata & 15];
310  // FIXME: no ECM, MCM stuff ...
311  PIXEL_POINTER_CHECK_ASSERT(p + 7);
312  *(p++) = chrdata & 128 ? fg : bg;
313  *(p++) = chrdata & 64 ? fg : bg;
314  *(p++) = chrdata & 32 ? fg : bg;
315  *(p++) = chrdata & 16 ? fg : bg;
316  *(p++) = chrdata & 8 ? fg : bg;
317  *(p++) = chrdata & 4 ? fg : bg;
318  *(p++) = chrdata & 2 ? fg : bg;
319  *(p++) = chrdata & 1 ? fg : bg;
320  if (x == 39) {
321  p += tail;
322  x = 0;
323  if (charline == 7) {
324  if (y == 24)
325  break;
326  y++;
327  charline = 0;
328  } else {
329  charline++;
330  vidp -= 40;
331  colp -= 40;
332  }
333  } else
334  x++;
335  }

337 }

391 /* Extremely incorrect sprite emulation! BUGS:
392  * Sprites cannot be behind the background (sprite priority)
393  * Multicolour sprites are not supported
394  * No sprite-background collision detection
395  * No sprite-sprite collision detection
396  * This is a simple, after-the-rendered-frame render-sprites one-by-one algorithm
397  * This also requires to give up direct rendering if a sprite is enabled
398  * Very ugly, quick&dirty hack, not so optimal either, even without the other mentioned bugs ...
399 */
400 static void vic2_render_sprite ( int sprite_no, int sprite_mask, Uint8 *data, Uint32 *p, int tail )
401 {
402  int sprite_y = vic2_registers[sprite_no * 2 + 1] - SPRITE_Y_START_SCREEN;
403  int sprite_x = ((vic2_registers[sprite_no * 2] | ((vic2_registers[16] & sprite_mask) ? 0x100 : 0)) - SPRITE_X_START_SCREEN) * 1;
404  Uint32 colour = palette[vic2_registers[39 + sprite_no] & 15];
405  int expand_x = vic2_registers[29] & sprite_mask;
406  int expand_y = vic2_registers[23] & sprite_mask;
407  int lim_y = sprite_y + ((expand_y) ? 42 : 21);
408  int y;
409  p += (320 + tail) * sprite_y;
410  for (y = sprite_y; y < lim_y; y += (expand_y ? 2 : 1), p += (320 + tail) * (expand_y ? 2 : 1))
411  if (y < 0 || y >= 200)
412  data += 3; // skip one line (three bytes) of sprite data if outside of screen
413  else {
414  int mask, a, x = sprite_x;
415  for (a = 0; a < 3; a++) {
416  for (mask = 128; mask; mask >>= 1) {
417  if (*data & mask) {
418  if (x >= 0 && x < 320) {
419  p[x] = colour;
420  if (expand_y && y < 200)
421  p[x + 320 + tail] = colour;
422  }
423  x++;
424  if (expand_x && x >= 0 && x < 320) {
425  p[x] = colour;
426  if (expand_y && y < 200)
427  p[x + 320 + tail] = colour;
428  x++;
429  }
430  } else
431  x += expand_x ? 2 : 1;
432  }
433  data++;
434  }
435  }
436 }

441 /* This is the one-frame-at-once (highly incorrect implementation, that is)
442  renderer. It will call legacy VIC2 text mode render (optionally with
443  80 columns mode, though, ECM, MCM, hardware attributes are not supported),
444  VIC2 legacy HIRES mode (MCM is not supported), or bitplane modes (V400,
445  H1280, odd scanning/interlace is not supported). Sprites, screen positioning,
446  etc is not supported */
447 void vic2_render_screen ( void )
448 {
449  int tail_sdl;
450  Uint32 *p_sdl = xemu_start_pixel_buffer_access(&tail_sdl);
451  int sprites = vic2_registers[0x15];
452  if (vic2_registers[0x11] & 32)
453  vic2_render_screen_bmm(p_sdl, tail_sdl);
454  else
455  vic2_render_screen_text(p_sdl, tail_sdl);
456  if (sprites) { // Render sprites. VERY BAD. We ignore sprite priority as well (cannot be behind the background)
457  int a;
458  for (a = 7; a >= 0; a--) {
459  int mask = 1 << a;
460  if (sprites & mask)
461  vic2_render_sprite(a, mask, memory + vic2_16k_bank + (vic2_sprite_pointers[a] << 6), p_sdl, tail_sdl); // sprite_pointers are set by the renderer functions above!
462  }
463  }
464  xemu_update_screen();
465 }

479 static inline void nmi_set ( int level, int mask )
480 {
481  // NMI is a low active _EDGE_ triggered 65xx input ... In my emulator though, the signal
482  // is "high active", and also we must form the "edge" ourselves from "level". NMI level is
483  // set as a 2bit number, on bit 0, CIA2, on bit 1, keyboard RESTORE key. Thus having zero
484  // value for level means (in the emu!) that not RESTORE key is pressed neither CIA2 has active
485  // IRQ output, non-zero value means some activation. Well, if I am not confused enough here,
486  // this should mean that nmi_level zero->non-zero transit should produce the edge (which should
487  // be the falling edge in the real hardware anyway ... but the rising here. heh, I should follow
488  // the signal level of the hardware in my emulator, honestly ...)
489  int nmi_new_level;
490  if (level)
491  nmi_new_level = nmi_level | mask;
492  else
493  nmi_new_level = nmi_level & (~mask);
494  if ((!nmi_level) && nmi_new_level) {
495  DEBUG("NMI edge is emulated towards the CPU (%d->%d)" NL, nmi_level, nmi_new_level);
496  cpu65.nmiEdge = 1; // the "NMI edge" trigger is deleted by the CPU emulator itself (it's not a level trigger)
497  }
498  nmi_level = nmi_new_level;
499 }

627 static Uint8 io_read ( int addr )
628 {
629  DEBUG("IO: reading $%04X @ PC=$%04X" NL, addr, cpu65.pc);
630  if (addr < 0xD400) // D000-D3FF VIC-II
631  return vic2_read_reg(addr);
632  if (addr < 0xD800) // D400-D7FF SID (not emulated here)
633  return 0xFF;
634  if (addr < 0xDC00) // D800-DBFF Colour SRAM (1K, 4 bit)
635  return colour_sram[addr & 0x3FF];
636  if (addr < 0xDD00) // DC00-DCFF CIA-1
637  return cia_read(&cia1, addr & 15);
638  if (addr < 0xDE00) // DD00-DDFF CIA-2
639  return cia_read(&cia2, addr & 15);
640  return 0xFF; // DE00-DFFF the rest, I/O-1 and I/O-2 exp area
641 }

645 static void io_write ( int addr, Uint8 data )
646 {
647  DEBUG("IO: writing $%04X with $%02X @ PC=$%04X" NL, addr, data, cpu65.pc);
648  if (addr < 0xD400) { // D000-D3FF VIC-II
649  vic2_write_reg(addr, data);

651  }
652  if (addr < 0xD800) // D400-D7FF SID (not emulated here)
653  return;
654  if (addr < 0xDC00) { // D800-DBFF Colour SRAM (1K, 4 bit)
655  colour_sram[addr & 0x3FF] = data | 0xF0; // 4 upper bits are always set (the SRAM has only 4 bits)
656  return;
657  }

659  cia_write(&cia1, addr & 15, data);
660  return;
661  }
662  if (addr < 0xDE00) { // DD00-DDFF CIA-2
663  cia_write(&cia2, addr & 15, data);
664  return;
665  }
666  // DE00-DFFF the rest, I/O-1 and I/O-2 exp area
667 }

671 int cpu65_trap_callback ( Uint8 opcode )
672 {
673  Uint8 *pc_p = GET_READ_P(cpu65.pc);
674  if (pc_p != 1 + &PATCH_P) {
675  if (warp)
676  FATAL("FATAL: CPU trap at unknown address in warp mode (pre-GEOS loading) PC=$%04X OP=$%02X" NL, cpu65.pc, opcode);
677  if (pc_p >= memory + 0x10000)
678  FATAL("FATAL: unknown CPU trap not in the RAM PC=$%04X OP=$%02X" NL, cpu65.pc, opcode);
679  if (geos_loaded != 2)
680  FATAL("FATAL: unknown CPU without GEOS loaded PC=$%04X OP=$%02X" NL, cpu65.pc, opcode);
681  geos_cpu_trap(opcode);
682  return 1;
683  }
684  warp = 0; // turn warp speed off
685  // Try to load a custom GEOS kernal directly into the RAM
686  if (geos_loaded) {
687  cpu65.pc = memory[0x300] | (memory[0x301] << 8);
688  return 1;
689  }
690  if (!geos_load_kernal(xemucfg_get_str("geoskernal"))) {
691  geos_loaded = 2; // GEOS was OK!!!!
692  return 1; // if no error, return with '1' (as not zero) to signal CPU emulator that trap should not be executed
693  }
694  geos_loaded = 1;
695  // In case if we cannot load some GEOS kernal stuff, continue in "C64 mode" ... :-/
696  // Some ugly method to produce custom "startup screen" :)
697  inject_screencoded_message(41, "**** Can't load GEOS, boot as C64 ****");
698  cpu65.pc = memory[0x300] | (memory[0x301] << 8);
699  return 1; // do NOT execute the trap op
700 }
