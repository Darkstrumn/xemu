233 static Uint8 inject_prg ( void )
234 {
235  int addr;
236  if (!emufile_p)
237  return emurom_policy; // No loaded program, use the pre-defined policy instead
238  addr = emufile_p[0] | (emufile_p[1] << 8);
239  printf("LOAD: injecting program into the memory from $%04X" NL, addr);
240  memcpy(memory + addr, emufile_p + 2, emufile_size - 2);
241  memory[addr - 1] = 0;
242  memory[0x2b] = addr & 0xFF;
243  memory[0x2c] = addr >> 8;
244  addr += emufile_size - 2;
245  memory[0x2d] = addr & 0xFF;
246  memory[0x2e] = addr >> 8;
247  return 128;
248 }

261 // Need to be defined, if CPU65_TRAP_OPCODE is defined for the CPU emulator!
262 int cpu65_trap_callback ( Uint8 opcode )
263 {
264  if (cpu65.pc >= 0xA000 && opcode == CPU65_TRAP_OPCODE) { // cpu65.pc always meant to be the position _after_ the trap opcode!
265  Uint8 trap = memory[cpu65.pc];
266  if (is_our_rom() < 0)
267  FATAL("Unknown ROM/RAM code at $%04X caused trap!", cpu65.pc - 1);
268  switch (trap) {
269  case 0:
270  cpu65.a = inject_prg();
271  EMUPRINTF("\rMONITOR: SYS %d\r", 0xA009);
272  break;
273  case 1:
274  EMUPRINTF("** XVIC20 MONITOR/LGB\rBM=$%04X-%04X S=$%04X\rEXP %s\r",
275  memory[0x281] | (memory[0x282] << 8),
276  memory[0x283] | (memory[0x284] << 8),
277  memory[648] << 8,
278  vic20_get_memconfig_string()
279  );
280  cpu65.a = 1;
281  break;
282  case 2:
283  cpu65.a = 1; // by default, set the continue flag ...
284  execute_monitor_command();
285  break;
286  default:
287  FATAL("Unknown CPU trap (%d) at $%04X", trap, cpu65.pc - 1);
288  }
289  cpu65.pc++; // jump over the trap number byte ...
290  return 1; // you must return with the CPU cycles used, but at least with value of 1!
291  } else
292  return 0; // ignore trap!! Return with zero means, the CPU emulator should execute the opcode anyway
293 }

311 // Called by CPU emulation code when any kind of memory byte must be written.
312 // Note: optimization is used, to make the *most common* type of write access easy. Even if the whole function is more complex, or longer/slower this way for other accesses!
313 void cpu65_write_callback ( Uint16 addr, Uint8 data )
314 {
315  // Write optimization, handle the most common case first: memory byte to be written is not special, ie writable RAM, not I/O, etc
316  if (XEMU_LIKELY(is_kpage_writable[addr >> 10])) { // writable flag for every Kbytes of 64K is checked (for different memory configurations, faster "decoding", etc)
317  memory[addr] = data;
318  return;
319  }
320  // ELSE: other kind of address space is tried to be written ...
321  // TODO check if I/O devices are fully decoded or there can be multiple mirror ranges
322  if ((addr & 0xFF00) == 0x9000) { // VIC-I register is written (decoded for a full 256 bytes long area)
323  cpu_vic_reg_write(addr & 0xF, data);
324  return;
325  }
326  if ((addr & 0xFFF0) == 0x9110) { // VIA-1 register is written
327  via_write(&via1, addr & 0xF, data);
328  return;
329  }
330  if ((addr & 0xFFF0) == 0x9120) { // VIA-2 register is written
331  via_write(&via2, addr & 0xF, data);
332  return;
333  }
334  // if other memory areas tried to be written (hit this point), write will be simply ignored
335 }

351 // Called by CPU emulation code when any kind of memory byte must be read.
352 // Note: optimization is used, to make the *most common* type of read access easy. Even if the whole function is more complex, or longer/slower this way for other accesses!
353 Uint8 cpu65_read_callback ( Uint16 addr )
354 {
355  // Optimization: handle the most common case first!
356  // Check if our read is NOT about the (built-in) I/O area. If it's true, let's just use the memory array
357  // (even for undecoded areas, memory[] is intiailized with 0xFF values
358  if (XEMU_LIKELY((addr & 0xF800) != 0x9000))
359  return memory[addr];
360  // ELSE: it IS the I/O area or colour SRAM ... Let's see what we want!
361  // TODO check if I/O devices are fully decoded or there can be multiple mirror ranges
362  if ((addr & 0xFFF0) == 0x9000) // VIC-I register is read
363  return cpu_vic_reg_read(addr & 0xF);
364  if ((addr & 0xFFF0) == 0x9110) // VIA-1 register is read
365  return via_read(&via1, addr & 0xF);
366  if ((addr & 0xFFF0) == 0x9120) // VIA-2 register is read
367  return via_read(&via2, addr & 0xF);
368  if ((addr & 0xFC00) == 0x9400)
369  return memory[addr] | 0xF0; // colour SRAM is read, always return '1' for upper bits (only 4 bits "wide" memory chip!)
370  // if non of the above worked, let's just read our emulated address space
371  // which means some kind of ROM, or undecoded area (64K space is intialized with 0xFF bytes ...)
372  return memory[addr];
373 }

458 static void emulation_loop ( void )
459 {
460  for (;;) { // our emulation loop ...
461  int opcyc;
462  opcyc = cpu65_step(); // execute one opcode (or accept IRQ, etc), return value is the used clock cycles
463  via_tick(&via1, opcyc); // run VIA-1 tasks for the same amount of cycles as the CPU
464  via_tick(&via2, opcyc); // -- "" -- the same for VIA-2
465  cycles += opcyc;
466  if (cycles >= CYCLES_PER_SCANLINE) { // if [at least!] 71 (on PAL) CPU cycles passed then render a VIC-I scanline, and maintain scanline value + texture/SDL update (at the end of a frame)
467  // render one (scan)line. Note: this is INACCURATE, we should do rendering per dot clock/cycle or something,
468  // but for a simple emulator like this, it's already acceptable solultion, I think!
469  // Note about frameskip: we render only every second (half) frame, no interlace (PAL VIC), not so correct, but we also save some resources this way
470  if (!frameskip)
471  vic_render_line();
472  if (scanline == LAST_SCANLINE) {
473  update_emulator();
474  frameskip = !frameskip;
475  return;
476  } else
477  scanline++;
478  cycles -= CYCLES_PER_SCANLINE;
479  }
480  }
