68 // Called by CPU emulation code when any kind of memory byte must be written.
69 void cpu65_write_callback ( Uint16 addr, Uint8 data )
70 {
71  /**** 0000-9EFF: low RAM (fixed RAM) area ****/
72  if (addr < 0x9F00) {
73  lo_ram[addr] = data;
74  return;
75  }
76  /**** 9F00-9FFF: I/O area ****/
77  if (addr < 0xA000) {
78  switch ((addr >> 4) & 0xF) {
79  case 0x0:
80  case 0x1:
81  IODEBUGPRINT("IO_W: writing to reg $%04X (data=$%02X), audio controller @ PC=$%04X" NL, addr, data, cpu65.old_pc);
82  break;
83  case 0x2:
84  case 0x3:
85  IODEBUGPRINT("IO_W: writing to reg $%04X (data=$%02X), VERA video controller @ PC=$%04X" NL, addr, data, cpu65.old_pc);
86  vera_write_cpu_register(addr, data); // VERA masks the addr bits, so it's OK.
87  break;
88  case 0x6:
89  IODEBUGPRINT("IO_W: writing to reg $%04X (data=$%02X), VIA-1 @ PC=$%04X" NL, addr, data, cpu65.old_pc);
90  via_write(&via1, addr & 0xF, data);
91  break;
92  case 0x7:
93  IODEBUGPRINT("IO_W: writing to reg $%04X (data=$%02X), VIA-2 @ PC=$%04X" NL, addr, data, cpu65.old_pc);
94  via_write(&via2, addr & 0xF, data);
95  break;
96  case 0x8:
97  case 0x9:
98  IODEBUGPRINT("IO_W: writing to reg $%04X (data=$%02X), RTC @ PC=$%04X" NL, addr, data, cpu65.old_pc);
99  break;
100  default:
101  IODEBUGPRINT("IO_W: writing to reg $%04X (data=$%02X), UNKNOWN/RESERVED @ PC=$%04X" NL, addr, data, cpu65.old_pc);
102  break;
103  }
104  return;
105  }
106  /**** A000-BFFF: 8K pageable RAM area, "hi-RAM" ****/
107  if (XEMU_LIKELY(addr < 0xC000)) {
108  hi_ram[hi_ram_access_offset + addr - 0xA000] = data;
109  return;
110  }
111  // Others (ROM) can be ignored, since it's ROM (not writable anyway) ...
112 }

117 // Called by CPU emulation code when any kind of memory byte must be read.
118 Uint8 cpu65_read_callback ( Uint16 addr )
119 {
120  /**** 0000-9EFF: low RAM (fixed RAM) area ****/
121  if (addr < 0x9F00)
122  return lo_ram[addr];
123  /**** 9F00-9FFF: I/O area ****/
124  if (addr < 0xA000) {
125  Uint8 data = 0xFF;
126  switch ((addr >> 4) & 0xF) {
127  case 0x0:
128  case 0x1:
129  IODEBUGPRINT("IO_R: reading from reg $%04X (data=$%02X), audio controller @ PC=$%04X" NL, addr, data, cpu65.old_pc);
130  break;
131  case 0x2:
132  case 0x3:
133  data = vera_read_cpu_register(addr); // VERA masks the addr bits, so it's OK
134  IODEBUGPRINT("IO_R: reading from reg $%04X (data=$%02X), VERA video controller @ PC=$%04X" NL, addr, data, cpu65.old_pc);
135  break;
136  case 0x6:
137  data = via_read(&via1, addr & 0xF);
138  IODEBUGPRINT("IO_R: reading from reg $%04X (data=$%02X), VIA-1 @ PC=$%04X" NL, addr, data, cpu65.old_pc);
139  break;
140  case 0x7:
141  data = via_read(&via2, addr & 0xF);
142  IODEBUGPRINT("IO_R: reading from reg $%04X (data=$%02X), VIA-2 @ PC=$%04X" NL, addr, data, cpu65.old_pc);
143  break;
144  case 0x8:
145  case 0x9:
146  IODEBUGPRINT("IO_R: reading from reg $%04X (data=$%02X), RTC @ PC=$%04X" NL, addr, data, cpu65.old_pc);
147  break;
148  default:
149  IODEBUGPRINT("IO_R: reading from reg $%04X (data=$%02X), UNKNOWN/RESERVED @ PC=$%04X" NL, addr, data, cpu65.old_pc);
150  break;
151  }
152  return data;
153  }
154  /**** A000-BFFF: 8K pageable RAM area, "hi-RAM" ****/
155  if (addr < 0xC000)
156  return hi_ram[hi_ram_access_offset + addr - 0xA000];
157  /**** C000-FFFF: the rest, 16K pagable ROM area ****/
158  return rom[hi_rom_access_offset + addr - 0xC000];
159 }

186 static void init_ram ( int hi_ram_size )
187 {
188  if (hi_ram_size > 2048)
189  hi_ram_size = 2048;
190  if (hi_ram_size < 0)
191  hi_ram_size = 0;
192  hi_ram_banks = hi_ram_size >> 3;
193  DEBUGPRINT("Setting (hi-)RAM size memtop to %dK, %d banks." NL, hi_ram_size, hi_ram_banks);
194  memset(lo_ram, 0, sizeof lo_ram);
195  memset(hi_ram, 0xFF, sizeof hi_ram);
196  set_ram_bank(0xFF);
197 }

273 static void emulation_loop ( void )
274 {
275  for (;;) { // our emulation loop ...
276  int opcyc;
277  //printf("PC=%04X OPC=%02X\n", cpu65.pc, cpu65_read_callback(cpu65.pc));
278  opcyc = cpu65_step(); // execute one opcode (or accept IRQ, etc), return value is the used clock cycles
279  via_tick(&via1, opcyc); // run VIA-1 tasks for the same amount of cycles as the CPU
280  via_tick(&via2, opcyc); // -- "" -- the same for VIA-2
281  //opcyc <<= speed_shifter;
282  cycles += opcyc;
283  all_virt_cycles += opcyc; // FIXME: should be scaled for different CPU speeds, but then also CYCLES_PER_SECOND should be altered for the desired CPU speed!!!
284  if (cycles >= CYCLES_PER_SCANLINE) {
285  if (!frameskip) {
286  }
287  if (vera_render_line() == 0) { // start of a new frame that is ...
288  update_emulator();
289  //vera_vsync();
290  frameskip = !frameskip;
291  return;
292  }
293  cycles -= CYCLES_PER_SCANLINE;
294  }
295  }
296 }

338  xemucfg_define_switch_option("dumpmem", "Dump memory states on exit into files");
339  if (xemucfg_parse_all(argc, argv))
340  return 1;
341  /* Initiailize SDL - note, it must be before loading ROMs, as it depends on path info from SDL! */
342  if (xemu_post_init(
343  TARGET_DESC APP_DESC_APPEND, // window title
344  1, // resizable window
345  SCREEN_WIDTH, SCREEN_HEIGHT, // texture sizes
346  SCREEN_WIDTH, SCREEN_HEIGHT, // logical size
347  SCREEN_WIDTH, SCREEN_HEIGHT, // window size

351  NULL, // initialize palette into this stuff
352  RENDER_SCALE_QUALITY, // render scaling quality
353  USE_LOCKED_TEXTURE, // 1 = locked texture access
354  emulator_shutdown // shutdown function
355  ))
356  return 1;
357  hid_init(
358  x16_key_map,
359  VIRTUAL_SHIFT_POS,
360  SDL_ENABLE // enable HID joy events
361  );
362  // --- memory initialization ---
363  init_ram(xemucfg_get_num("hiramsize"));
364  vera_init();
365  if (
366  load_rom(xemucfg_get_str("rom"))
367  )
368  return 1;
369  // Continue with initializing ...
370  clear_emu_events(); // also resets the keyboard
371  cpu65_reset(); // reset CPU: it must be AFTER kernal is loaded at least, as reset also fetches the reset vector into PC ...
372  // Initiailize VIAs.

374  // it uses callback functions, which must be registered here, NULL values means unused functionality
375  via_init(&via1, "VIA-1",
376  via1_outa_ram_bank, // outa
377  via1_outb_rom_bank, // outb
378  NULL, // outsr
379  via1_ina_ram_bank, // ina
380  via1_inb_rom_bank, // inb
381  NULL, // insr
382  via1_setint
383  );
384  via_init(&via2, "VIA-2",
385  NULL, // outa [reg 1]
386  NULL, //via2_kbd_set_scan, // outb [reg 0], we wire port B as output to set keyboard scan, HOWEVER, we use ORB directly in get scan!
387  NULL, // outsr
388  via2_ina, // ina [reg 1], we wire port A as input to get the scan result, which was selected with port-A
389  NULL, //via2_inb,
390  NULL, // insr
391  via2_setint
392  );
393  // Without these, the first DDR register writes would cause problems, since not OR* (Output Register) is written first ...
394  //via1.ORA = 0xFF;
395  //via1.ORB = 0xFF;
396  //via2.ORA = 0xFF;
397  //via2.ORB = 0xFF;
398  cycles = 0;
399  xemu_set_full_screen(xemucfg_get_bool("fullscreen"));
400  if (!xemucfg_get_bool("syscon"))
401  sysconsole_close(NULL);
402  xemu_timekeeping_start(); // we must call this once, right before the start of the emulation
403  //vera_vsync();
404 #ifdef __EMSCRIPTEN__
405  // http://xemu-dist.lgb.hu/dist/x16/xemu-xcx16-sample.html?t=11
406  close(1);
407  close(2);
408 #endif
409  DEBUGPRINT("CPU: starting exection at $%04X" NL, cpu65.pc);
410  XEMU_MAIN_LOOP(emulation_loop, 30, 1);
411  return 0;
412 }
