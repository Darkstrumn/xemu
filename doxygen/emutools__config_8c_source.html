40 void xemucfg_define_option ( const char *optname, enum xemutools_option_type type, void *defval, const char *help )
41 {
42  struct xemutools_config_st *entry;
43  if (type == OPT_STR && check_string_size(defval))
44  FATAL("Xemu internal config error: too long default option for option '%s'", optname);
45  entry = xemu_malloc(sizeof(struct xemutools_config_st));
46  if (!config_head) {
47  config_head = entry;
48  entry->next = NULL;
49  } else {
50  struct xemutools_config_st *p = config_head, *p_prev = NULL;
51  for (;;) {
52  // we want to manage alphabet sorted list just for nice help output, and for checking re-definition assertion in one step as well
53  int ret = strcasecmp(optname, p->name);
54  if (!ret)
55  FATAL("Xemu internal config error: trying to re-define option '%s'", optname);
56  if (ret < 0) { // we want the first entry already later in alphabet than current one, insert new entry before that!
57  entry->next = p;
58  if (p_prev)
59  p_prev->next = entry;
60  else
61  config_head = entry;
62  break;
63  }
64  if (p->next) {
65  p_prev = p;
66  p = p->next;
67  } else {
68  p->next = entry;
69  entry->next = NULL;
70  break;
71  }
72 
73  }
74  }
75  entry->name = xemu_strdup(optname);
76  entry->type = type;
77  entry->value = (defval && type == OPT_STR) ? xemu_strdup(defval) : defval;
78  entry->help = help;
79 }

100 static struct xemutools_config_st *search_option ( const char *name )
101 {
102  struct xemutools_config_st *p = config_head;
103  char *s = strchr(name, '@');
104  int l = s ? s - name : strlen(name);
105  while (p)
106  if (!strncasecmp(name, p->name, l) && p->name[l] == 0 && (name[l] == 0 || name[l] == '@'))
107  break;
108  else
109  p = p->next;
110  return (p && s && p->type != OPT_PROC) ? NULL : p;
111 }

152 int xemucfg_parse_config_file ( const char *filename_in, int open_can_fail )
153 {
154  char *p, cfgmem[CONFIG_FILE_MAX_SIZE + 1];
155  int lineno = xemu_load_file(filename_in, cfgmem, 0, CONFIG_FILE_MAX_SIZE, open_can_fail ? NULL : "Cannot load specified configuration file.");
156  if (lineno < 0) {
157  if (open_can_fail)
158  return 1;
159  XEMUEXIT(1);
160  }
161  cfgmem[lineno] = 0; // terminate string
162  if (strlen(cfgmem) != lineno)
163  FATAL("Config: bad config file (%s), contains NULL character (not a text file)", xemu_load_filepath);
164  p = cfgmem;
165  lineno = 1; // line number counter in read config file from now
166  do {
167  char *p1, *pn;
168  // Skip white-spaces at the beginning of the line
169  while (*p == ' ' || *p == '\t')
170  p++;
171  // Search for end of line (relaxed check, too much mixed line-endings I've seen already within ONE config file failed with simple fgets() etc method ...)
172  p1 = strstr(p, "\r\n");
173  if (p1) pn = p1 + 2;
174  else {
175  p1 = strstr(p, "\n\r");
176  if (p1) pn = p1 + 2;
177  else {
178  p1 = strchr(p, '\n');
179  if (p1) pn = p1 + 1;
180  else {
181  p1 = strchr(p, '\r');
182  pn = p1 ? p1 + 1 : NULL;
183  }
184  }
185  }
186  if (p1) *p1 = 0; // terminate line string at EOL marker
187  p1 = strchr(p, '#');
188  if (p1) *p1 = 0; // comment - if any - is also excluded
189  // Chop white-spaces off from the end of the line
190  p1 = p + strlen(p);
191  while (p1 > p && (p1[-1] == '\t' || p1[-1] == ' '))
192  *(--p1) = 0;
193  // If line is not empty, separate key/val (if there is) and see what it means
194  if (*p) {
195  struct xemutools_config_st *o;
196  const char *s;
197  p1 = p;
198  while (*p1 && *p1 != '\t' && *p1 != ' ' && *p1 != '=')
199  p1++;
200  if (*p1) {
201  *(p1++) = 0;
202  while (*p1 == '\t' || *p1 == ' ' || *p1 == '=')
203  p1++;
204  if (!*p1)
205  p1 = NULL;
206  } else
207  p1 = NULL;
208  printf("Line#%d = \"%s\",\"%s\"" NL, lineno, p, p1 ? p1 : "<no-specified>");
209  o = search_option(p);
210  if (!o)
211  FATAL("Config file (%s) error at line %d: keyword '%s' is unknown.", xemu_load_filepath, lineno, p);
212  if (o->type != OPT_NO && !p1)
213  FATAL("Config file (%s) error at line %d: keyword '%s' requires a value.", xemu_load_filepath, lineno, p);
214  switch (o->type) {
215  case OPT_STR:
216  if (o->value)
217  free(o->value);
218  if (check_string_size(p1))
219  FATAL("Config file (%s) error at line %d: keyword '%s' has too long value", xemu_load_filepath, lineno, p);
220  o->value = xemu_strdup(p1);
221  break;
222  case OPT_BOOL:
223  s = set_boolean_value(p1, &o->value);
224  if (s)
225  FATAL("Config file (%s) error at line %d: keyword '%s' %s, but '%s' is detected.", xemu_load_filepath, lineno, p, s, p1);
226  break;
227  case OPT_NUM:
228  o->value = (void*)(intptr_t)atoi(p1);
229  break;
230  case OPT_NO:
231  if (p1)
232  FATAL("Config file (%s) error at line %d: keyword '%s' DOES NOT require any value, but '%s' is detected.", xemu_load_filepath, lineno, p, p1);
233  o->value = (void*)1;
234  break;
235  case OPT_PROC:
236  s = (*(xemucfg_parser_callback_func_t)(o->value))(o, p, p1);
237  if (s)
238  FATAL("Config file (%s) error at line %d: keyword's '%s' parameter '%s' is invalid: %s", xemu_load_filepath, lineno, p, p1, s);
239  break;
240  }
241  }
242  // Prepare for next line
243  p = pn; // start of next line (or EOF if NULL)
244  lineno++;
245  } while (p);
246  return 0;
247 }

251 static int xemucfg_parse_commandline ( int argc, char **argv, const char *only_this )
252 {
253  // Skip arg-0, which is program name ...
254  argc--;
255  argv++;
256 #ifdef XEMU_ARCH_MAC
257  // Oh no, another MacOS miss-feature :-O it seems Finder passes a strange parameter to EVERY app it starts!!
258  // Skip that, otherwise the user will experience an error box about unknown option and Xemu exits ...
259  if (argc && !strncmp(argv[0], "-psn_", 5)) {
260  argc--;
261  argv++;
262  macos_gui_started = 1;
263  }
264 #endif
265  while (argc) {
266  struct xemutools_config_st *o;
267  if (*argv[0] != '/' && *argv[0] != '-')
268  OPT_ERROR_CMDLINE("Invalid option '%s', must start with '-'", *argv);
269  if (!strcasecmp(*argv + 1, "h") || !strcasecmp(*argv + 1, "help") || !strcasecmp(*argv + 1, "-help") || !strcasecmp(*argv + 1, "?")) {
270  if (!only_this || (only_this && !strcasecmp(only_this, "help"))) {
271  dump_help();
272  return 1;
273  }
274  argc--;
275  argv++;
276  continue;
277  }
278  o = search_option(*argv + 1);
279  if (!o)
280  OPT_ERROR_CMDLINE("Unknown option '%s'", *argv);
281  argc--;
282  argv++;
283  if (o->type == OPT_NO) {
284  if (only_this && strcasecmp(only_this, o->name))
285  continue;
286  o->value = (void*)1;
287  } else {
288  const char *s;
289  if (!argc)
290  OPT_ERROR_CMDLINE("Option '%s' requires a parameter, but end of command line detected.", argv[-1]);
291  if (only_this && strcasecmp(only_this, o->name))
292  goto do_not;
293  switch (o->type) {
294  case OPT_STR:
295  if (o->value)
296  free(o->value);
297  if (check_string_size(*argv))
298  OPT_ERROR_CMDLINE("Option '%s' has too long value.", argv[1]);
299  o->value = xemu_strdup(*argv);
300  break;
301  case OPT_BOOL:
302  s = set_boolean_value(*argv, &o->value);
303  if (s)
304  OPT_ERROR_CMDLINE("Option '%s' %s, but '%s' is detected.", argv[-1], s, *argv);
305  break;
306  case OPT_NUM:
307  o->value = (void*)(intptr_t)atoi(*argv);
308  break;
309  case OPT_PROC:
310  s = (*(xemucfg_parser_callback_func_t)(o->value))(o, argv[-1] + 1, *argv);
311  if (s)
312  OPT_ERROR_CMDLINE("Option's '%s' parameter '%s' is invalid: %s", argv[-1], *argv, s);
313  break;
314  case OPT_NO:
315  break; // make GCC happy to handle all cases ...
316  }
317  do_not:
318  argc--;
319  argv++;
320  }
321  }
322  return 0;
323 }
