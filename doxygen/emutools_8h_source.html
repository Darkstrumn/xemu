78 #define _REPORT_WINDOW_(sdlflag, str, ...) do { \
79  char _buf_for_win_msg_[4096]; \
80  CHECK_SNPRINTF(snprintf(_buf_for_win_msg_, sizeof _buf_for_win_msg_, __VA_ARGS__), sizeof _buf_for_win_msg_); \
81  fprintf(stderr, str ": %s" NL, _buf_for_win_msg_); \
82  if (debug_fp) \
83  fprintf(debug_fp, str ": %s" NL, _buf_for_win_msg_); \
84  if (sdl_win) { \
85  save_mouse_grab(); \
86  MSG_POPUP_WINDOW(sdlflag, sdl_window_title, _buf_for_win_msg_, sdl_win); \
87  clear_emu_events(); \
88  xemu_drop_events(); \
89  SDL_RaiseWindow(sdl_win); \
90  restore_mouse_grab(); \
91  xemu_timekeeping_start(); \
92  } else \
93  MSG_POPUP_WINDOW(sdlflag, sdl_window_title, _buf_for_win_msg_, sdl_win); \
94 } while (0)
