150 // Call this, if "enable NMI" or VBLANK signals change!
151 static void manage_nmi ( void )
152 {
153  // NMI signal on the CPU in the Primo is created from VBLANK, and a gate signal to allow to mask the unmaskable :) interrupt (NMI)
154  // VBLANK is a looong signal, the reason that we can use this way, that Z80 has an edge triggered NMI input, not level (unlike INT)
155  int new_status = nmi_enabled && vblank;
156  if (new_status && !nmi_status) { // rising edge (note: in the emulator we only use high-active signal, which can be different than the real hw)
157  emu_loop_notification |= EMU_LOOP_NMI_NOTIFY;
158  DEBUG("NMI: triggering edge" NL);
159  }
160  nmi_status = new_status;
161 }

166 void z80ex_pwrite_cb ( Z80EX_WORD port16, Z80EX_BYTE value )
167 {
168  if ((port16 & 0xFF) < 0x40) {
169  primo_screen = (value & 8) ? 0x2000 : 0x0000;
170  nmi_enabled = value & 128;
171  manage_nmi();
172  if ((value & 16) != beeper) {
173  static int rounding_error = 0;
174  Uint64 no_of_samples = (rounding_error + all_cycles_spent - beeper_last_changed) / cpu_clocks_per_audio_sample;
175  rounding_error = (rounding_error + all_cycles_spent - beeper_last_changed) % cpu_clocks_per_audio_sample;
176  if (no_of_samples <= AUDIO_PULSE_SAMPLES_MAX_PASS && no_of_samples > 0 && audio) {
177  Uint8 samples[no_of_samples];
178  memset(samples, value & 16 ? 0x80 + VOLUME8 : 0x80 - VOLUME8, no_of_samples);
179  int ret = SDL_QueueAudio(audio, samples, no_of_samples); // last param are (number of) BYTES not samles. But we use 1 byte/samle audio ...
180  if (ret)
181  DEBUGPRINT("AUDIO: DATA: ERROR: %s" NL, SDL_GetError());
182  else
183  DEBUG("AUDIO: DATA: queued" NL);
184  } else
185  DEBUG("AUDIO: DATA: rejected! samples=%d" NL, (int)no_of_samples);
186  beeper = value & 16;
187  beeper_last_changed = all_cycles_spent;
188  }
189  if ((value & 64) != joy.clock) {
190  joy.clock = value & 64;
191  if (joy.clock) {
192  joy.step = (all_cycles_spent - joy.last_clocked >= joy.clocking_timeout) ? 0 : (joy.step + 1) & 7;
193  joy.last_clocked = all_cycles_spent;
194  if (joy.step > 4) // FIXME: should we do this??? or real HW would always depends on timed reset of counter?
195  joy.step = 0;
196  }
197  }
198  } else if ((port16 & 0xFF) < 0x80) {
199  serial = value & (2 | 8 | 16 | 32);
200  } else if ((port16 & 0xFF) == 0xFD) {
201  static Uint8* const rd_tab[] = { memory.main, memory.expram, memory.exprom, memory.exprom + 0x4000 };
202  static Uint8* const wr_tab[] = { memory.wrwaste, memory.expram, memory.wrwaste , memory.wrwaste };
203  memory.rd_p[0] = rd_tab[value & 3];
204  memory.wr_p[0] = wr_tab[value & 3];
205  }
206 }

450 static int pri_apply ( Uint8 *data, int size, int wet_run )
451 {
452  int i = 0;
453  while (i < size) {
454  int btype = data[i];
455  DEBUGPRINT("PRI: block_type=$%02X @ %d" NL, btype, i);
456  if (btype == 0xC9)
457  return 0; // return, OK, no Z80 PC specified
458  if (i >= size - 2)
459  break;
460  int baddr = data[i + 1] | (data[i + 2] << 8);
461  if (btype == 0xC3)
462  return baddr; // return OK, Z80 PC specified!
463  if (i >= size - 4)
464  break;
465  int bsize = data[i + 3] | (data[i + 4] << 8);
466  if (btype == 0xD1) {
467  baddr += memory.main[0x40A4] | (memory.main[0x40A5] << 8);
468  if (wet_run) {
469  memory.main[0x40F9] = (baddr + bsize + 1) & 0xFF;
470  memory.main[0x40FA] = (baddr + bsize + 1) >> 8;
471  }
472  } else if (btype != 0xD5 && btype != 0xD9) {
473  ERROR_WINDOW("Invalid PRI file, unknown block type $%02X @ %d", btype, i);
474  return -1;
475  }
476  if (i >= size - bsize)
477  break;
478  DEBUGPRINT(" ... ADDR=%04X SIZE=%04X" NL, baddr, bsize);
479  i += 5;
480  while (bsize--) {
481  if (wet_run) // I don't use direct memory.main access here, as PRI could overwrite ROM etc ...
482  z80ex_mwrite_cb(baddr++, data[i]);
483  i++;
484  }
485 #if 0
486  if (wet_run)
487  memcpy(memory.main + baddr, data + i + 5, bsize);
488  i += 5 + bsize;
489 #endif
490  }
491  ERROR_WINDOW("Invalid PRI file, probably truncated");
492  return -1;
493 }

511 static int set_cpu_hz ( int hz )
512 {
513  if (hz < 1000000)
514  hz = 1000000;
515  else if (hz > 8000000)
516  hz = 8000000;
517  cpu_clocks_per_scanline = (hz / PAL_LINE_FREQ) & ~1; // 15625 Hz = 312.5 * 50, PAL "scanline frequency", how many Z80 cycles we need for that. Also make it to an even number
518  cpu_clock = cpu_clocks_per_scanline * PAL_LINE_FREQ; // to reflect the possible situation when it's not a precise divider above
519  DEBUGPRINT("CLOCK: CPU: clock speed set to %.2f MHz (%d CPU cycles per scanline)" NL, cpu_clock / 1000000.0, cpu_clocks_per_scanline);
520  joy.clocking_timeout = (int)(((double)JOY_CLOCKING_TIMEOUT_MICROSECS / 1000000.0) * (double)cpu_clock);
521  DEBUGPRINT("CLOCK: JOY: clocking timeout is %d microseconds, %d CPU cycles" NL, JOY_CLOCKING_TIMEOUT_MICROSECS, (int)joy.clocking_timeout);
522  cpu_clocks_per_audio_sample = cpu_clock / AUDIO_SAMPLING_FREQ;
523  DEBUGPRINT("CLOCK: AUDIO: %d CPU clocks per audio sample at sampling frequency of %d Hz" NL, cpu_clocks_per_audio_sample, AUDIO_SAMPLING_FREQ);
524  return cpu_clock;
525 }

544 static void emulation_loop ( void )
545 {
546  static int cycles = 0;
547  static int frameskip = 0;
548  Uint64 all_cycles_old = all_cycles_spent;
549  goto first_shot;
550  for (;;) { // our emulation loop ...
551  int op_cycles = z80ex_step();
552  cycles -= op_cycles;
553  all_cycles_spent += op_cycles;
554  first_shot:
555  if (XEMU_UNLIKELY(emu_loop_notification)) {
556  DEBUG("EMU: notification" NL);
557  // DESIGN: for fast emulation loop, we have a "notification" system. In general,
558  // there are no notifications. So this way we have to check only a single variable,
559  // and do more "if"-s only if we have something. So the usual scenario (not having
560  // notification) would require to check a single condition -> faster emulation. Still,
561  // we can expand the range of notifications without affecting the emulation speed
562  // this way, if there are no notifications, which should be the majority of the emulator
563  // time. NOTE: debugger support/etc, should use notification system as well!
564  if (emu_loop_notification & EMU_LOOP_NMI_NOTIFY) { // Notification for triggering NMI
565  // check if Z80ex accepted NMI. Though a real Z80 should always do this (NMI is not maskable),
566  // Z80ex is constructed in a way, that it actually *returns* after z80ex_step() even if it was
567  // a prefix byte, like FD,DD,CB,ED ... Surely, Z80 would not allow this, that's why z80ex_nmi()
568  // may report zero, that is NMI *now* cannot be accepted. Then we simply try again on the next
569  // run. Note: the construction of this z80ex_step() function actually GOOD because of more fine
570  // grained timing, so it's not only "a problem"! Actually it's IMPOSSIBLE to write an accurate
571  // Z80 emulation without this behaviour! Since, a real Z80 would react for a loong opocde
572  // sequence of FD/DD bytes to block even NMIs till its end, which would block the execution
573  // of the emulator updates etc as well, probably.
574  op_cycles = z80ex_nmi();
575  if (op_cycles) {
576  DEBUG("NMI accepted" NL);
577  cycles -= op_cycles;
578  all_cycles_spent += op_cycles;
579  emu_loop_notification &= ~EMU_LOOP_NMI_NOTIFY; // clear notification, it's done
580  } else {
581  DEBUG("NMI not accepted" NL);
582  }
583  }
584  if (emu_loop_notification & EMU_LOOP_LOAD_NOTIFY) { // Notification for checking PC to trigger PRI loading!
585  if (Z80_PC == ROM_Z80_PC_LOAD_TRIGGER) {
586  emu_loop_notification &= ~EMU_LOOP_LOAD_NOTIFY; // clear notification, it's done
587  int ret = pri_load(pri_name, 1);
588  if (ret > 0) {
589  DEBUGPRINT("PRI: loaded, Z80 PC change $%04X -> $%04X" NL, Z80_PC, ret);
590  Z80_PC = ret;
591  } else if (!ret)
592  DEBUGPRINT("PRI: loaded, no Z80 PC change requested" NL);
593  else
594  DEBUGPRINT("PRI: cannot load program" NL);
595  }
596  }
597  // This is not very much used currently, but maybe in the future:
598  if (emu_loop_notification & EMU_LOOP_UPDATE_NOTIFY) { // update notification for the emulator core
599  emu_loop_notification &= ~EMU_LOOP_UPDATE_NOTIFY; // clear notification, it's done
600  break; // end of the emulation loop!
601  }
602  if (emu_loop_notification & EMU_LOOP_DISASM_NOTIFY) {
603  char buf[256];
604  int t1, t2;
605  z80ex_dasm(buf, sizeof buf, 0, &t1, &t2, disasm_read_callback, Z80_PC);
606  DEBUGPRINT("%04X %s" NL, Z80_PC, buf);
607  }
608  }
609  if (XEMU_UNLIKELY(cycles <= 0)) {
610  // We run at least enough CPU cycles for a scanline, by decrementing cycles with the number
611  // of cycles Z80 spent to execute code at each step. The value "cycles" must be intialized
612  // with the desired number of CPU cycles for a scanline, which is calculated by the set_cpu_hz()
613  // function to configure CPU clock speed.
614  if (scanline == 311) {
615  cycles += cpu_clocks_per_scanline / 2; // last scanline is only a "half" according to the PAL standard. Or such ...
616  scanline++;
617  } else if (scanline == 312) { // we were at the last scanline of a half-frame.
618  cycles += cpu_clocks_per_scanline;
619  scanline = 0;
620  vblank = VBLANK_OFF;
621  manage_nmi();
622  //DEBUGPRINT("VIDEO: new frame!" NL);
623  frameskip = !frameskip;
624  if (!frameskip)
625  break; // end of the emulation loop!
626  } else {
627  cycles += cpu_clocks_per_scanline;
628  scanline++;
629  if (scanline == VBLANK_START_SCANLINE) {
630  vblank = VBLANK_ON;
631  manage_nmi();
632  }
633  }
634  if (XEMU_UNLIKELY(cycles <= 0)) {
635  FATAL("Emulation problem, underflow of cpu cycles / scanline tracking counter!");
636  }
637  }
638  }
639  renderer();
640  hid_handle_all_sdl_events();
641  xemugui_iteration();
642  xemu_timekeeping_delay((Uint64)(1000000L * (Uint64)(all_cycles_spent - all_cycles_old) / (Uint64)cpu_clock));
643 }
