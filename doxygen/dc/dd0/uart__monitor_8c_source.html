55 static char *parse_hex_arg ( char *p, int *val, int min, int max )
56 {
57  int r;
58  while (*p == 32)
59  p++;
60  *val = -1;
61  if (!*p) {
62  umon_printf(SYNTAX_ERROR "unexpected end of command (no parameter)");
63  return NULL;
64  }
65  for (r = 0;;) {
66  if (*p >= 'a' && *p <= 'f')
67  r = (r << 4) | (*p - 'a' + 10);
68  else if (*p >= 'A' && *p <= 'F')
69  r = (r << 4) | (*p - 'A' + 10);
70  else if (*p >= '0' && *p <= '9')
71  r = (r << 4) | (*p - '0');
72  else if (*p == 32 || *p == 0)
73  break;
74  else {
75  umon_printf(SYNTAX_ERROR "invalid data as hex digit '%c'", *p);
76  return NULL;
77  }
78  p++;
79  }
80  *val = r;
81  if (r < min || r > max) {
82  umon_printf(SYNTAX_ERROR "command parameter's value is outside of the allowed range for this command %X (%X...%X)", r, min, max);
83  return NULL;
84  }
85  return p;
86 }

103 static void setmem28(char *param, int addr)
104 {
105  addr &= 0xFFFFFFF;
106  Uint8* vals = NULL;
107  char* orig_param = param;
108  int cnt = 0;
109  int val;
110 
111  // get param count
112  while (param && !check_end_of_command(param, 0))
113  {
114  param = parse_hex_arg(param, &val, 0, 0xFF);
115  cnt++;
116  }
117 
118  vals = calloc(cnt, sizeof(Uint8));
119  param = orig_param;
120 
121  int idx = 0;
122  while (param && !check_end_of_command(param, 0))
123  {
124  param = parse_hex_arg(param, &val, 0, 0xFF);
125  vals[idx++] = (Uint8)val;
126  }
127 
128  m65mon_setmem28(addr, cnt, vals);
129  free(vals);

132 static void execute_command ( char *cmd )
133 {
134  int par1;
135  char *p = cmd;
136  while (*p)
137  if (p == cmd && (*cmd == 32 || *cmd == '\t' || *cmd == 8))
138  cmd = ++p;
139  else if (*p == '\t')
140  *(p++) = 32;
141  else if (*p == 8)
142  memmove(p - 1, p + 1, strlen(p + 1) + 1);
143  else if (*p > 127 || *p < 32) {
144  umon_printf(SYNTAX_ERROR "invalid character in the command (ASCII=%d)", *p);
145  return;
146  } else
147  p++;
148  p--;
149  while (p >= cmd && *p <= 32)
150  *(p--) = 0;
151  DEBUG("UARTMON: command got \"%s\" (%d bytes)." NL, cmd, (int)strlen(cmd));
152  switch (*(cmd++)) {
153  case 'h':
154  case 'H':
155  case '?':
156  if (check_end_of_command(cmd, 1))
157  umon_printf("Xemu/Mega65 Serial Monitor\r\nWarning: not 100%% compatible with UART monitor of a *real* Mega65 ...");
158  break;
159  case 'r':
160  case 'R':
161  if (check_end_of_command(cmd, 1))
162  m65mon_show_regs();
163  break;
164  case 'd':
165  cmd = parse_hex_arg(cmd, &par1, 0, 0xFFFF);
166  if (cmd && check_end_of_command(cmd, 1))
167  m65mon_dumpmem16(par1);
168  break;
169  case 'm':
170  cmd = parse_hex_arg(cmd, &par1, 0, 0xFFFFFFF);
171  if (cmd && check_end_of_command(cmd, 1))
172  m65mon_dumpmem28(par1);
173  break;
174  case 's':
175  cmd = parse_hex_arg(cmd, &par1, 0, 0xFFFFFFF);
176  setmem28(cmd, par1);
177  break;
178  case 't':
179  if (!*cmd)
180  m65mon_do_trace();
181  else if (*cmd == 'c') {
182  if (check_end_of_command(cmd, 1))
183  m65mon_do_trace_c();
184  } else {
185  cmd = parse_hex_arg(cmd, &par1, 0, 1);
186  if (cmd && check_end_of_command(cmd, 1))
187  m65mon_set_trace(par1);
188  }
189  break;
190  case 'b':
191  cmd = parse_hex_arg(cmd, &par1, 0, 0xFFFF);
192  if (cmd && check_end_of_command(cmd, 1))
193  m65mon_breakpoint(par1);
194  break;
195  case 0:
196  m65mon_empty_command(); // emulator can use it, if it wants
197  break;
198  default:
199  umon_printf(SYNTAX_ERROR "unknown (or not implemented) command '%c'", cmd[-1]);
200  break;
201  }
202 }

209 static int set_nonblock ( int fd )
210 {
211  int flags = fcntl(fd, F_GETFL, 0);
212  if (flags == -1)
213  return 1;
214  if (fcntl(fd, F_SETFL, flags | O_NONBLOCK))
215  return 1;
216  return 0;
217 }

221 int uartmon_init ( const char *fn )
222 {
223  struct sockaddr_un sock_st;
224  int sock;
225  sock_server = -1;
226  if (!fn || !*fn) {
227  DEBUGPRINT("UARTMON: disabled, no name is specified to bind to." NL);
228  return 0;
229  }
230  sock = socket(AF_UNIX, SOCK_STREAM, 0);
231  if (sock < 0) {
232  ERROR_WINDOW("Cannot create named socket %s, UART monitor cannot be used: %s\n", fn, strerror(errno));
233  return 1;
234  }
235  sock_st.sun_family = AF_UNIX;
236  strcpy(sock_st.sun_path, fn);
237  unlink(sock_st.sun_path);
238  if (bind(sock, (struct sockaddr*)&sock_st, sizeof(struct sockaddr_un))) {
239  ERROR_WINDOW("Cannot bind named socket %s, UART monitor cannot be used: %s\n", fn, strerror(errno));
240  close(sock);
241  return 1;
242  }
243  if (listen(sock, 5)) {
244  ERROR_WINDOW("Cannot listen with named socket %s, UART monitor cannot be used: %s\n", fn, strerror(errno));
245  close(sock);
246  return 1;
247  }
248  if (set_nonblock(sock)) {
249  ERROR_WINDOW("Cannot set named socket %s into non-blocking mode, UART monitor cannot be used: %s\n", fn, strerror(errno));
250  close(sock);
251  return 1;
252  }
253  DEBUG("UARTMON: monitor is listening on socket %s" NL, fn);
254  sock_client = -1; // no client connection yet
255  sock_server = sock; // now set the socket
256  umon_echo = 1;
257  umon_send_ok = 1;
258  return 0;
259 }

276 void uartmon_finish_command ( void )
277 {
278  umon_send_ok = 1;
279  if (umon_write_buffer[umon_write_size - 1] != '\n') {
280  // if generated message wasn't closed with CRLF (well, only LF is checked), we do so here
281  umon_write_buffer[umon_write_size++] = '\r';
282  umon_write_buffer[umon_write_size++] = '\n';
283  }
284  // umon_trigger_end_of_answer = 1;
285  umon_write_buffer[umon_write_size++] = '.'; // add the 'dot prompt'! (m65dbg seems to check LF + dot for end of the answer)
286  umon_read_pos = 0;
287  umon_echo = 1;

292 // Non-blocky I/O for UART monitor emulation.
293 // Note: you need to call it "quite often" or it will be terrible slow ...
294 // From emulator main update, aka etc 25Hz rate should be Okey ...
295 void uartmon_update ( void )
296 {
297  int ret;
298  // If there is no server socket, we can't do anything!
299  if (sock_server < 0)
300  return;
301  // Try to accept new connection, if not yet have one (we handle only *ONE* connection!!!!)
302  if (sock_client < 0) {
303  struct sockaddr_un sock_st;
304  socklen_t len = sizeof(struct sockaddr_un);
305  ret = accept(sock_server, (struct sockaddr *)&sock_st, &len);
306  if (ret >=0 || (errno != EAGAIN && errno != EWOULDBLOCK))
307  DEBUG("UARTMON: accept()=%d error=%s" NL,
308  ret,
309  ret >= 0 ? "OK" : strerror(errno)
310  );
311  if (ret >= 0) {
312  if (set_nonblock(ret)) {
313  close(ret);
314  } else {
315  sock_client = ret; // "publish" new client socket
316  // Reset reading/writing information
317  umon_write_size = 0;
318  umon_read_pos = 0;
319  DEBUGPRINT("UARTMON: new connection established on socket %d" NL, sock_client);
320  }
321  }
322  }
323  // If no established connection, return
324  if (sock_client < 0)
325  return;
326  // If there is data to write, try to write
327  if (umon_write_size) {
328  if (!umon_send_ok)
329  return;
330  ret = write(sock_client, umon_write_buffer + umon_write_pos, umon_write_size);
331  if (ret >=0 || (errno != EAGAIN && errno != EWOULDBLOCK))
332  DEBUG("UARTMON: write(%d,buffer+%d,%d)=%d (%s)" NL,
333  sock_client, umon_write_pos, umon_write_size,
334  ret, ret < 0 ? strerror(errno) : "OK"
335  );
336  if (ret == 0) { // client socket closed
337  close(sock_client);
338  sock_client = -1;
339  DEBUGPRINT("UARTMON: connection closed by peer while writing" NL);
340  return;
341  }
342  if (ret > 0) {
343  //debug_buffer_slice(umon_write_buffer + umon_write_pos, ret);
344  umon_write_pos += ret;
345  umon_write_size -= ret;
346  if (umon_write_size < 0)
347  FATAL("FATAL: negative umon_write_size!");
348  }
349  if (umon_write_size)
350  return; // if we still have bytes to write, return and leave the work for the next update
351  }
352  umon_write_pos = 0;
353  // Try to read data
354  ret = read(sock_client, umon_read_buffer + umon_read_pos, sizeof(umon_read_buffer) - umon_read_pos - 1);
355  if (ret >=0 || (errno != EAGAIN && errno != EWOULDBLOCK))
356  DEBUG("UARTMON: read(%d,buffer+%d,%d)=%d (%s)" NL,
357  sock_client, umon_read_pos, (int)sizeof(umon_read_buffer) - umon_read_pos - 1,
358  ret, ret < 0 ? strerror(errno) : "OK"
359  );
360  if (ret == 0) { // client socket closed
361  close(sock_client);
362  sock_client = -1;
363  DEBUGPRINT("UARTMON: connection closed by peer while reading" NL);
364  return;
365  }
366  if (ret > 0) {
367  /* ECHO: provide echo for the client */
368  if (umon_echo) {
369  char*p = umon_read_buffer + umon_read_pos;
370  int n = ret;
371  while (n--)
372  if (*p != 13 && *p != 10) {
373  umon_write_buffer[umon_write_size++] = *(p++);
374  } else {
375  umon_echo = 0; // setting to zero avoids more input to echo, and also signs a complete command
376  *p = 0; // terminate string in read buffer
377  break;
378  }
379  }
380  /* ECHO: end */
381  umon_read_pos += ret;
382  umon_read_buffer[umon_read_pos] = 0;
383  //debug_buffer(umon_read_buffer);
384  if (!umon_echo || sizeof(umon_read_buffer) - umon_read_pos - 1 == 0) {
385  umon_read_buffer[sizeof(umon_read_buffer) - 1] = 0; // just in case of a "mega long command" with filled rx buffer ...
386  umon_write_buffer[umon_write_size++] = '\r';
387  umon_write_buffer[umon_write_size++] = '\n';
388  umon_send_ok = 1; // by default, command is finished after the execute_command()
389  execute_command(umon_read_buffer); // Execute our command!
390  // command may delay (like with trace) the finish of the command with
391  // setting umon_send_ok to zero. In this case, some need to call
392  // uartmon_finish_command() some time otherwise the monitor connection
393  // will just hang!
394  if (umon_send_ok)
395  uartmon_finish_command();
396  }
397  }
398 }
