92 // Called in case of an RMW (read-modify-write) opcode write access.
93 // Original NMOS 6502 would write the old_data first, then new_data.
94 // It has no inpact in case of normal RAM, but it *does* with an I/O register in some cases!
95 // CMOS line of 65xx (probably 65CE02 as well?) seems not write twice, but read twice.
96 // However this leads to incompatibilities, as some software used the RMW behavour by intent.
97 // Thus Mega65 fixed the problem to "restore" the old way of RMW behaviour.
98 // I also follow this path here, even if it's *NOT* what 65CE02 would do actually!
99 CPU_CUSTOM_FUNCTIONS_INLINE_DECORATOR void cpu65_write_rmw_callback ( Uint16 addr, Uint8 old_data, Uint8 new_data ) {
100  cpu_rmw_old_data = old_data;
101  // It's the backend's (which realizes the op) responsibility to handle or not handle the RMW behaviour,
102  // based on the fact if cpu_rmw_old_data is non-negative (being an int type) when it holds the "old_data".
103  CALL_MEMORY_WRITER(addr >> 8, addr, new_data);
104  cpu_rmw_old_data = -1;
105 }
106 CPU_CUSTOM_FUNCTIONS_INLINE_DECORATOR void cpu65_write_rmw_paged_callback ( Uint8 page, Uint8 addr8, Uint8 old_data, Uint8 new_data ) {
107  cpu_rmw_old_data = old_data;
108  CALL_MEMORY_WRITER_PAGED(page, addr8, new_data);
109  cpu_rmw_old_data = -1;
110 }
