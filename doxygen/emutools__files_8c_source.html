38 int xemuexec_run ( char *const args[] )
39 {
40  pid_t pid = fork();
41  if (pid == -1) {
42  DEBUGPRINT("EXEC: fork() failed: %s" NL, strerror(errno));
43  return -1; // fork failed?
44  }
45  if (!pid) { // the child's execution process after fork()
46  int a;
47  for(a = 3; a < 1024; a++)
48  close(a);
49  close(0);
50  execvp(args[0], args);
51  // exec won't return in case if it's OK. so if we're still here, there was a problem with the exec func ...
52  printf("EXEC: execution of \"%s\" failed: %s" NL, args[0], strerror(errno));
53  _exit(127); // important to call this and not plain exit(), as we don't want to run atexit() registered stuffs and so on!!!
54  }
55  return pid; // the parent's execution process after fork()
56 }

59 int xemuexec_check_status ( pid_t pid, int wait )
60 {
61  int status;
62  pid_t ret;
63  if (pid <= 0)
64  return 127;
65  do {
66  ret = waitpid(pid, &status, wait ? 0 : WNOHANG);
67  if (ret == -1) {
68  if (errno == EINTR)
69  continue;
70  DEBUGPRINT("EXEC: WAIT: waitpid(%d) returned error: %s" NL, pid, strerror(errno));
71  return -1;
72  }
73  } while (ret < 0);
74  if (ret != pid) {
75  DEBUGPRINT("EXEC: still running" NL);
76  return XEMUEXEC_STILL_RUNNING; // still running?
77  }
78  if (WIFEXITED(status))
79  return WEXITSTATUS(status);
80  return 127; // we have not so much a standard status (not exited normally?!) so fake one ...

260 static int download_file ( int size )
261 {
262  int ret;
263  char path_final[PATH_MAX];
264  const char **execargs_p = downloader_utility_spec_start_p;
265  struct stat st;
266  strcpy(path_final, installer_store_to);
267  strcat(installer_store_to, ".temp");
268  DEBUGPRINT("INSTALLER: goal: %s -> %s -> %s (%d bytes)" NL,
269  installer_fetch_url, installer_store_to, path_final, size
270  );
271  if (unlink(path_final) && errno != ENOENT) {
272  ERROR_WINDOW("Installer: cannot delete already existing final file");
273  return -1;
274  }
275  do {
276  xemuexec_process_t proc;
277  printf("Tryning: %s\n", *execargs_p);
278  if (unlink(installer_store_to) && errno != ENOENT) {
279  ERROR_WINDOW("Installer: cannot delete already exisiting temp file");
280  return -1;
281  }
282  proc = xemuexec_run((char* const*)execargs_p);
283  ret = xemuexec_check_status(proc, 1);
284  printf("Exit status: %d\n", ret);
285  if (!ret || downloader_utility_selected)
286  break;
287  while (*(execargs_p++))
288  ;
289  } while (*execargs_p);
290  if (ret) {
291  ERROR_WINDOW("Installer: cannot download and/or no working download utility can be used");
292  unlink(installer_store_to);
293  return -1;
294  }
295  if (stat(installer_store_to, &st)) {
296  ERROR_WINDOW("Installer: cannot stat file (not downloaded at all?)");
297  return -1;
298  }
299  printf("File size = %d\n", (int)st.st_size);
300  if (st.st_size != size) {
301  unlink(installer_store_to);
302  ERROR_WINDOW("Installer: downloaded file has wrong size (%d, wanted: %d)", (int)st.st_size, size);
303  return -1;
304  }
305  if (rename(installer_store_to, path_final)) {
306  ERROR_WINDOW("Installer: cannot rename to final");
307  return -1;
308  }
309  if (!downloader_utility_selected) {
310  downloader_utility_spec_start_p = execargs_p;
311  downloader_utility_selected = 1;
312  DEBUGPRINT("INSTALLER: setting \"%s\" as the default downloader utility for this session." NL, *execargs_p);
313  }
314  return 0;

319 static int download_file_by_db ( const char *filename, const char *storepath )
320 {
321  char *p;
322  int i;
323  if (!xemu_installer_db)
324  return -1;
325  strcpy(installer_store_to, storepath);
326  p = xemu_installer_db;
327  i = strlen(filename);
328  while (*p) {
329  while (*p && *p <= 32)
330  p++;
331  if (!strncmp(filename, p, i) && (p[i] == '\t' || p[i] == ' ')) {
332  p += i + 1;
333  while (*p == ' ' || *p == '\t')
334  p++;
335  if (*p > 32) {
336  long int sizereq;
337  char *q;
338  if (strncasecmp(p, "http://", 7) && strncasecmp(p, "https://", 8) && strncasecmp(p, "ftp://", 6)) {
339  ERROR_WINDOW("Bad download descriptor file at URL field (bar protocol) for record \"%s\"", filename);
340  return -1;
341  }
342  q = installer_fetch_url;
343  sizereq = 0;
344  while (*p > 32) {
345  if (sizereq == sizeof(installer_fetch_url) - 2) {
346  ERROR_WINDOW("Bad download descriptor file at URL field (too long) for record \"%s\"", filename);
347  return -1;
348  }
349  *q++ = *p++;
350  sizereq++;
351  }
352  *q = 0;
353  sizereq = strtol(p, &p, 0);
354  if (*p > 32)
355  sizereq = -1;
356  if (sizereq > 0 && sizereq <= 4194304) {
357  int ret;
358  char msgbuffer[sizeof(installer_fetch_url) + 256];
359  if (legal_warning) {
360  INFO_WINDOW("Legal-warning blah-blah ...");
361  legal_warning = 0;
362  }
363  sprintf(msgbuffer, "Downloading file \"%s\". Do you agree?\nSource: %s", filename, installer_fetch_url);
364  if (QUESTION_WINDOW("YES|NO", msgbuffer))
365  return -1;
366  ret = download_file(sizereq);
367  if (!ret)
368  INFO_WINDOW("File %s seems to be downloaded nicely with %s", filename, *downloader_utility_spec_start_p);
369  return ret;
370  } else {
371  ERROR_WINDOW("Bad download descriptor file at size field for record \"%s\" (or this file is not auto-installable)", filename);
372  return -1;
373  }
374  }
375  }
376  while (*p && *p != '\n' && *p != '\r')
377  p++;
378  }
379  DEBUGPRINT("INSTALLER: file-key %s cannot be found in the download description file" NL, filename);
380  return -1;
381 }

418 /* Open a file, returning a file descriptor, or negative value in case of failure.
419  * Input parameters:
420  * filename: name of the file
421  * - if it begins with '@' the file is meant to relative to the SDL preferences directory (ie: @thisisit.rom, no need for dirsep!)
422  * - if it begins with '#' the file is meant for 'data directory' which is probed then multiple places, depends on the OS as well
423  * - Note: in this case, if installer is enabled and file not found, Xemu can try to download the file. For this, see above the "installer" part of this source
424  * - otherwise it's simply a file name, passed as-is
425  * mode: actually the flags parameter for open (O_RDONLY, etc)
426  * - O_BINARY is used automatically in case of Windows, no need to specify as input data
427  * - you can even use creating file effect with the right value here
428  * *mode2: pointer to an int, secondary open mode set
429  * - if it's NULL pointer, it won't be used ever
430  * - if it's not NULL, open() is also tried with the pointed flags for open() after trying (and failed!) open() with the 'mode' par
431  * - if mode2 pointer is not NULL, the pointed value will be zeroed by this func, if NOT *mode2 is used with successfull open
432  * - the reason for this madness: opening a disk image which neads to be read/write access, but also works for read-only, however
433  * then the caller needs to know if the disk emulation is about r/w or ro only ...
434  * *filepath_back: if not null, actually tried path will be placed here (even in case of non-successfull call, ie retval is negative)
435  * - in case of multiple-path tries (# prefix) the first (so the most relevant, hopefully) is passed back
436  * - note: if no prefix (@ and #) the filename will be returned as is, even if didn't hold absolute path (only relative) or no path as all (both: relative to cwd)!
437  *
438  */
439 int xemu_open_file ( const char *filename, int mode, int *mode2, char *filepath_back )
440 {
441  char paths[16][PATH_MAX];
442  int a, max_paths;
443  if (!filename)
444  FATAL("Calling xemu_open_file() with NULL filename!");
445  if (!*filename)
446  FATAL("Calling xemu_open_file() with empty filename!");
447  max_paths = 0;
448 #ifdef XEMU_ARCH_HTML
449  sprintf(paths[max_paths++], "%s%s", EMSCRIPTEN_SDL_BASE_DIR, (filename[0] == '@' || filename[0] == '#') ? filename + 1 : filename);
450 #else
451  if (*filename == '@') {
452  sprintf(paths[max_paths++], "%s%s", sdl_pref_dir, filename + 1);
453  } else if (*filename == '#') {
454  sprintf(paths[max_paths++], "%s%s", sdl_inst_dir, filename + 1);
455  sprintf(paths[max_paths++], "%s%s", sdl_pref_dir, filename + 1);
456  sprintf(paths[max_paths++], "%srom" DIRSEP_STR "%s", sdl_base_dir, filename + 1);
457  sprintf(paths[max_paths++], "%s%s", sdl_base_dir, filename + 1);
458 #ifndef XEMU_ARCH_WIN
459  sprintf(paths[max_paths++], "/usr/local/share/xemu/%s", filename + 1);
460  sprintf(paths[max_paths++], "/usr/local/lib/xemu/%s", filename + 1);
461  sprintf(paths[max_paths++], "/usr/share/xemu/%s", filename + 1);
462  sprintf(paths[max_paths++], "/usr/lib/xemu/%s", filename + 1);
463 #endif
464 #ifdef HAVE_XEMU_INSTALLER
465  sprintf(paths[max_paths++], "%s%s%s", installer_marker_prefix, sdl_inst_dir, filename + 1);
466 #endif
467  } else
468  strcpy(paths[max_paths++], filename);
469 // End of #ifdef XEMU_ARCH_HTML
470 #endif
471  a = 0;
472  do {
473  int fd;
474  const char *filepath = paths[a];
475 #ifdef HAVE_XEMU_INSTALLER
476  if (!strncmp(paths[a], installer_marker_prefix, strlen(installer_marker_prefix))) {
477  filepath += strlen(installer_marker_prefix);
478  download_file_by_db(filename + 1, filepath);
479  }
480 #endif
481  // Notes:
482  // 1. O_BINARY is a windows stuff. However, since we define O_BINARY as zero for non-win archs, it's OK
483  // 2. 0666 mask is needed as it can be also a creat() function basically ...
484  fd = open(filepath, mode | O_BINARY, 0666);
485  if (filepath_back)
486  strcpy(filepath_back, filepath);
487  if (fd >= 0) {
488  if (mode2)
489  *mode2 = 0;
490  DEBUGPRINT("FILE: file %s opened as %s with base mode-set as fd=%d" NL, filename, paths[a], fd);
491  return fd;
492  }
493  if (mode2) {
494  fd = open(filepath, *mode2 | O_BINARY, 0666); // please read the comments at the previous open(), above
495  if (fd >= 0) {
496  DEBUGPRINT("FILE: file %s opened as %s with *second* mode-set as fd=%d" NL, filename, paths[a], fd);
497  return fd;
498  }
499  }
500  } while (++a < max_paths);
501  // if not found, copy the first try so the report to user is more sane
502  if (filepath_back)
503  strcpy(filepath_back, paths[0]);
504  DEBUGPRINT("FILE: %s cannot be open, tried path(s): ", filename);
505  for (a = 0; a < max_paths; a++)
506  DEBUGPRINT(" %s", paths[a]);
507  DEBUGPRINT(NL);
508  return -1;
509 }

513 ssize_t xemu_safe_read ( int fd, void *buffer, size_t length )
514 {
515  ssize_t loaded = 0;
516  while (length > 0) {
517  ssize_t r = read(fd, buffer, length);
518  if (r < 0) { // I/O error on read
519  if (errno == EAGAIN || errno == EWOULDBLOCK || errno == EINTR)
520  continue;
521  return -1;
522  }
523  if (r == 0) // end of file
524  break;
525  loaded += r;
526  length -= r;
527  buffer += r;
528  }
529  return loaded;
530 }

626 /* Loads a file, probably ROM image etc. It uses xemu_open_file() - see above - for opening it.
627  * Return value:
628  * - non-negative: given mumber of bytes loaded
629  * - negative, error: -1 file open error, -2 file read error, -3 limit constraint violation
630  * Input parameters:
631  * * filename: see comments at xemu_open_file()
632  * * store_to: pointer to the store buffer
633  * - note: the buffer will be filled only in case of success, no partial modification can be
634  * - if store_to is NULL, then the load buffer is NOT free'd and the buffer pointer is assigned to xemu_load_buffer_p
635  * * min_size,max_size: in bytes, the minimal needed and the maximal allowed file size (can be the same)
636  * - note: limit contraint violation, if this does not meet during the read ("load") process
637  * * cry: character message, to 'cry' (show an error window) in case of a problem. if NULL = no dialog box
638  */
639 int xemu_load_file ( const char *filename, void *store_to, int min_size, int max_size, const char *cry )
640 {
641  int fd = xemu_open_file(filename, O_RDONLY, NULL, xemu_load_filepath);
642  if (fd < 0) {
643  if (cry) {
644  ERROR_WINDOW("Cannot open file requested by %s: %s\nTried as: %s\n%s%s", filename, strerror(errno), xemu_load_filepath,
645  (*filename == '#') ? "(# prefixed, multiple paths also tried)\n" : "",
646  cry
647  );
648  }
649  return -1;
650  } else {
651  int load_size;
652  xemu_load_buffer_p = xemu_malloc(max_size + 1); // try to load one byte more than the max allowed, to detect too large file scenario
653  load_size = xemu_safe_read(fd, xemu_load_buffer_p, max_size + 1);
654  if (load_size < 0) {
655  ERROR_WINDOW("Cannot read file %s: %s\n%s", xemu_load_filepath, strerror(errno), cry ? cry : "");
656  free(xemu_load_buffer_p);
657  xemu_load_buffer_p = NULL;
658  close(fd);
659  return -2;
660  }
661  close(fd);
662  if (load_size < min_size) {
663  free(xemu_load_buffer_p);
664  xemu_load_buffer_p = NULL;
665  if (cry)
666  ERROR_WINDOW("File (%s) is too small (%d bytes), %d bytes needed.\n%s", xemu_load_filepath, load_size, min_size, cry);
667  else
668  DEBUGPRINT("FILE: file (%s) is too small (%d bytes), %d bytes needed." NL, xemu_load_filepath, load_size, min_size);
669  return -3;
670  }
671  if (load_size > max_size) {
672  free(xemu_load_buffer_p);
673  xemu_load_buffer_p = NULL;
674  if (cry)
675  ERROR_WINDOW("File (%s) is too large, larger than %d bytes.\n%s", xemu_load_filepath, max_size, cry);
676  else
677  DEBUGPRINT("FILE: file (%s) is too large, larger than %d bytes needed." NL, xemu_load_filepath, max_size);
678  return -3;
679  }
680  if (store_to) {
681  memcpy(store_to, xemu_load_buffer_p, load_size);
682  free(xemu_load_buffer_p);
683  xemu_load_buffer_p = NULL;
684  } else
685  xemu_load_buffer_p = xemu_realloc(xemu_load_buffer_p, load_size);
686  DEBUGPRINT("FILE: %d bytes loaded from file: %s" NL, load_size, xemu_load_filepath);
687  return load_size;
688  }
689 }
