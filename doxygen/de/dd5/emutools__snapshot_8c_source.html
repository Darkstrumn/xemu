95 int xemusnap_read_block_header ( struct xemu_snapshot_block_st *block )
96 {
97  Uint8 buffer[256];
98  // Read fixed size portion of the block header
99  int ret = xemusnap_read_file(buffer, XEMUSNAP_FIXED_HEADER_SIZE);
100  if (ret)
101  return ret;
102  // Check
103  if (memcmp(buffer, block_framing_id, 8))
104  return XSNAPERR_FORMAT;
105  if (P_AS_BE32(buffer + 8) != XEMUSNAP_FRAMING_VERSION)
106  return XSNAPERR_FORMAT;
107  if (P_AS_BE32(buffer + 12))
108  return XSNAPERR_FORMAT;
109  block->block_version = P_AS_BE32(buffer + 16);
110  block->header_size = buffer[20];
111  block->idlen = block->header_size - XEMUSNAP_FIXED_HEADER_SIZE;
112  if (block->idlen < 1 || block->idlen > XEMUSNAP_MAX_IDENT_LENGTH)
113  return XSNAPERR_FORMAT;
114  // Read the block ident string part
115  ret = xemusnap_read_file(block->idstr, block->idlen);
116  if (ret)
117  return ret;
118  block->idstr[block->idlen] = 0;
119  return 0;
120 }

123 int xemusnap_write_block_header ( const char *ident, Uint32 version )
124 {
125  int len = strlen(ident);
126  Uint8 buffer[len + XEMUSNAP_FIXED_HEADER_SIZE];
127  memcpy(buffer, block_framing_id, 8);
128  U32_AS_BE(buffer + 8, XEMUSNAP_FRAMING_VERSION);
129  U32_AS_BE(buffer + 12, 0);
130  U32_AS_BE(buffer + 16, version);
131  buffer[20] = len + XEMUSNAP_FIXED_HEADER_SIZE;
132  memcpy(buffer + XEMUSNAP_FIXED_HEADER_SIZE, ident, len);
133  last_sub_block_size_written = -1;
134  return xemusnap_write_file(buffer, len + XEMUSNAP_FIXED_HEADER_SIZE);

192 static int load_from_open_file ( void )
193 {
194  struct xemu_snapshot_block_st block;
195  const struct xemu_snapshot_definition_st *def;
196  int ret;
197  if (!snapdef)
198  RETURN_XSNAPERR("Xemu error: snapshot format is not defined!");
199  block.counter = 0;
200  block.sub_counter = -1;
201  for (;;) {
202  ret = xemusnap_read_block_header(&block);
203  if ((ret == XSNAPERR_NODATA) && block.counter)
204  goto end_of_load;
205  block.sub_counter = -1;
206  handle_error:
207  switch (ret) {
208  case XSNAPERR_CALLBACK:
209  RETURN_XSNAPERR("Error while loading snapshot block \"%s\" (%d/%d): %s", block.idstr, block.counter, block.sub_counter, xemusnap_user_error_buffer);
210  case XSNAPERR_NODATA:
211  case XSNAPERR_TRUNCATED:
212  RETURN_XSNAPERR("Truncated file, unexpected end of file (~ %d/%d)", block.counter, block.sub_counter);
213  case XSNAPERR_FORMAT:
214  RETURN_XSNAPERR("Snapshot format error, maybe not a snapshot file, or version mismatch");
215  case XSNAPERR_IO:
216  RETURN_XSNAPERR("File I/O error while reading snapshot: %s", strerror(errno));
217  case 0:
218  block.is_ident = !memcmp(block.idstr, emu_ident, 6);
219  def = snapdef;
220  if (block.is_ident) { // is it an ident block?
221  if (block.counter)
222  goto end_of_load; // ident block other than the first one also signals end of snapshot, regardless of the rest of ident string
223  // check the ident string if it's really our one
224  if (strcmp(block.idstr + 6, emu_ident + 6))
225  RETURN_XSNAPERR("Not our snapshot file, format is \"%s\", expected: \"%s\"", block.idstr + 6, emu_ident + 6);
226  } else {
227  if (!block.counter)
228  RETURN_XSNAPERR("Invalid snapshot file, first block must be the ident block");
229  for (;;) {
230  if (!def->idstr)
231  RETURN_XSNAPERR("Unknown block type: \"%s\"", block.idstr);
232  if (!strcmp(def->idstr, block.idstr))
233  break;
234  def++;
235  }
236  }
237  block.sub_counter = 0;
238  for (;;) {
239  ret = xemusnap_read_be32(&block.sub_size);
240  if (ret) goto handle_error;
241  if (!block.sub_size)
242  break;
243  if (block.is_ident) {
244  ret = xemusnap_skip_sub_blocks(0);
245  if (ret) goto handle_error;
246  } else {
247  // Block can be "save only", ie no LOAD callback
248  if (def->load) {
249  strcpy(xemusnap_user_error_buffer, "?");
250  ret = def->load(def, &block);
251  if (ret) goto handle_error;
252  } else {
253  ret = xemusnap_skip_sub_blocks(0);
254  if (ret) goto handle_error;
255  }
256  }
257  block.sub_counter++;
258  }
259  break;
260  default:
261  FATAL("Xemu snapshot load internal error: unknown xemusnap_read_block() answer: %d", ret);
262  }
263  block.counter++;
264  }
265 end_of_load:
266  // the last entry given in the snapshot def table can hold the "finalizer" callback, call that too.
267  // for this, find the end of the definition first
268  def = snapdef;
269  while (def->idstr)
270  def++;
271  if (def->load) { // and call it, if it's not NULL
272  strcpy(xemusnap_user_error_buffer, "?");
273  ret = def->load(def, NULL);
274  if (ret) goto handle_error;
275  }
276  return 0;
277 }

295 static int save_to_open_file ( void )
296 {
297  const struct xemu_snapshot_definition_st *def = snapdef;
298  int ret;
299  if (!snapdef)
300  RETURN_XSNAPERR("Xemu error: snapshot format is not defined!");
301  // Ident block
302  ret = xemusnap_write_block_header(emu_ident, 0);
303  if (ret) goto handle_error;
304  ret = xemusnap_write_sub_block(NULL, 0);
305  if (ret) goto handle_error;
306  // Walk on blocks, use saver callback to save them all
307  while (def->idstr) {
308  if (def->save) {
309  // Block can be "load only", ie no SAVE callback
310  strcpy(xemusnap_user_error_buffer, "?");
311  ret = def->save(def);
312  if (ret) goto handle_error;
313  ret = xemusnap_write_sub_block(NULL, 0); // close block with writing a zero-sized sub-block
314  if (ret) goto handle_error;
315  }
316  def++;
317  }
318  // the last entry given in the snapshot def table can hold the "finalizer" callback, call that too.
319  if (def->save) {
320  strcpy(xemusnap_user_error_buffer, "?");
321  ret = def->save(def);
322  if (ret) goto handle_error;
323  }
324  return 0;
325 handle_error:
326  switch (ret) {
327  case XSNAPERR_NODATA:
328  case XSNAPERR_TRUNCATED:
329  RETURN_XSNAPERR("Cannot write file");
330  case XSNAPERR_FORMAT:
331  RETURN_XSNAPERR("Format violation");
332  case XSNAPERR_IO:
333  RETURN_XSNAPERR("File I/O error while writing snapshot: %s", strerror(errno));
334  case XSNAPERR_CALLBACK:
335  RETURN_XSNAPERR("Error while saving snapshot block \"%s\": %s", def->idstr, xemusnap_user_error_buffer);
336  default:
337  FATAL("Xemu snapshot save internal error: unknown error code: %d", ret);
338  }
339 }
