51 static XEMU_INLINE Uint8 c64_keyboard_read_on_CIA1_B ( Uint8 kbsel_a, Uint8 effect_b, Uint8 joy_state
52 #ifdef C65_KEYBOARD
53  , int kbsel_c65_special
54 #endif
55 )
56 {
57  // selected line(s) for scan: LOW pin state on port A, while reading in port B
58  // CIA uses pull-ups, output can be LOW only, if port data is zero, and ddr is output (thus being 1)
59  // so the "caller" of this func will do something like this:
60  // kbsel_a = cia1.PRA | (~cia1.DDRA)
61  // effect_b = cia1.PRB | (~cia1.DDRB)
62  // the second is needed: the read value is still low, if the port we're reading on is configured for output with port value set 0
63  //DEBUGPRINT("USING#1, extra row = %02X" NL, kbd_matrix[(C65_KEYBOARD_EXTRA_POS) >> 4]);
64  return
65 #ifdef C65_KEYBOARD
66  ((kbsel_c65_special) ? 0xFF : kbd_matrix[(C65_KEYBOARD_EXTRA_POS) >> 4]) &
67 #endif
68  ((kbsel_a & 1) ? 0xFF : kbd_matrix[0]) &
69  ((kbsel_a & 2) ? 0xFF : kbd_matrix[1]) &
70  ((kbsel_a & 4) ? 0xFF : kbd_matrix[2]) &
71  ((kbsel_a & 8) ? 0xFF : kbd_matrix[3]) &
72  ((kbsel_a & 16) ? 0xFF : kbd_matrix[4]) &
73  ((kbsel_a & 32) ? 0xFF : kbd_matrix[5]) &
74  ((kbsel_a & 64) ? 0xFF : kbd_matrix[6]) &
75  ((kbsel_a & 128) ? 0xFF : kbd_matrix[7]) &
76  joy_state &
77  effect_b
78  ;
79 }

81 static XEMU_INLINE Uint8 c64_keyboard_read_on_CIA1_A ( Uint8 kbsel_b, Uint8 effect_a, Uint8 joy_state )
82 {
83  // For description on this logic, please see comments at c64_keyboard_read_on_CIA1_B() above.
84  // The theory is the same, however here we use the negated value. it's not because it's different by hardware
85  // but because we handle kbsel_b value differently than with c64_keyboard_read_on_CIA1_B(), that's all!
86  kbsel_b = ~kbsel_b;
87  return (
88 #ifdef C65_KEYBOARD
89 #endif
90  (((kbd_matrix[0] & kbsel_b) == kbsel_b) ? 1 : 0) |
91  (((kbd_matrix[1] & kbsel_b) == kbsel_b) ? 2 : 0) |
92  (((kbd_matrix[2] & kbsel_b) == kbsel_b) ? 4 : 0) |
93  (((kbd_matrix[3] & kbsel_b) == kbsel_b) ? 8 : 0) |
94  (((kbd_matrix[4] & kbsel_b) == kbsel_b) ? 16 : 0) |
95  (((kbd_matrix[5] & kbsel_b) == kbsel_b) ? 32 : 0) |
96  (((kbd_matrix[6] & kbsel_b) == kbsel_b) ? 64 : 0) |
97  (((kbd_matrix[7] & kbsel_b) == kbsel_b) ? 128 : 0)
98  ) &
99  joy_state &
100  effect_a
101  ;
102 }
