83 struct virtdisk_chunk_st {
84  struct virtdisk_chunk_st *next; // pointer to the next chunk, or NULL, if it was the last (in that case vdisk.tail should point to this chunk)
85  int used_blocks; // number of used blocks in this chunk (up to vidsk.blocks_per_chunk)
86  Uint32 block_no_min; // optimization: lowest numbered block inside _this_ chunk of blocks
87  Uint32 block_no_max; // optimization: highest numbered block inside _this_ chunk of blocks
88  Uint8 *base; // base pointer of the data area on this chunk, 512 bytes per block then
89  Uint32 list[]; // block number list for this chunk (see: used_blocks) [C99/C11 flexible array members syntax]
90 };
91 struct virtdisk_st {
92  struct virtdisk_chunk_st *head, *tail; // pointers to the head and tail of the linked list of chunks, or NULL for both, if there is nothing yet
93  int blocks_per_chunk; // number of blocks handled by a chunk
94  int allocation_size; // pre-calculated to have number of bytes allocated for a chunk, including the data area + struct itself
95  int base_offset; // pre-calculated to have the offset to the data area inside a chunk
96  int all_chunks; // number of all currently allocated chunks
97  int all_blocks; // number of all currently USED blocks within all chunks
98  Uint8 range_map[RANGE_MAP_SIZE]; // optimization: store here if a given range is stored in memory, eliminating the need to search all chunks over, if not
99  Uint32 range_map_divisor; // optimization: for the above, to form range number from block number
100  int mode; // what mode is used, 0=no virtual disk
101 };

105 static void virtdisk_destroy ( void )
106 {
107  if (vdisk.head) {
108  struct virtdisk_chunk_st *v = vdisk.head;
109  int ranges = 0;
110  for (unsigned int a = 0; a < RANGE_MAP_SIZE; a++)
111  for (Uint8 m = 0x80; m; m >>= 1)
112  if ((vdisk.range_map[a] & m))
113  ranges++;
114  DEBUGPRINT("SDCARD: VDISK: destroying %d chunks (active data: %d blocks, %dKbytes, %d%%, ranges: %d/%d) of storage." NL,
115  vdisk.all_chunks, vdisk.all_blocks, vdisk.all_blocks >> 1,
116  100 * vdisk.all_blocks / (vdisk.all_chunks * vdisk.blocks_per_chunk),
117  ranges, RANGE_MAP_SIZE * 8
118  );
119  while (v) {
120  struct virtdisk_chunk_st *next = v->next;
121  free(v);
122  v = next;
123  }
124  }
125  vdisk.head = NULL;
126  vdisk.tail = NULL;
127  vdisk.all_chunks = 0;
128  vdisk.all_blocks = 0;
129  memset(vdisk.range_map, 0, RANGE_MAP_SIZE);
130 }

132 static void virtdisk_init ( int blocks_per_chunk, Uint32 total_number_of_blocks )
133 {
134  virtdisk_destroy();
135  vdisk.blocks_per_chunk = blocks_per_chunk;
136  vdisk.base_offset = sizeof(struct virtdisk_chunk_st) + (sizeof(Uint32) * blocks_per_chunk);
137  vdisk.allocation_size = (blocks_per_chunk << 9) + vdisk.base_offset;
138  vdisk.range_map_divisor = (total_number_of_blocks / (RANGE_MAP_SIZE * 8)) + 1;
139  DEBUGPRINT("SDCARD: VDISK: %d blocks (%dKbytes) per chunk, range-divisor is %u" NL, blocks_per_chunk, blocks_per_chunk >> 1, vdisk.range_map_divisor);
140 }

142 // Note: you must take care to call this function with "block" not outside of the desired capacity or some logic (range_map) will cause crash.
143 // That's not a problem, as this function is intended to use as storage backend, thus boundary checking should be done BEFORE you call this!
144 static Uint8 *virtdisk_search_block ( Uint32 block, int do_allocate )
145 {
146  struct virtdisk_chunk_st *v;
147  Uint32 range_index = block / vdisk.range_map_divisor;
148  Uint8 range_mask = 1U << (range_index & 7U);
149  range_index >>= 3U;
150  if (XEMU_LIKELY(vdisk.head && (vdisk.range_map[range_index] & range_mask))) {
151  v = vdisk.head;
152  do {
153  if (block >= v->block_no_min && block <= v->block_no_max)
154  for (unsigned int a = 0; a < v->used_blocks; a++)
155  if (v->list[a] == block)
156  return v->base + (a << 9);
157  v = v->next;
158  } while (v);
159  }
160  // if we can't found the block, and do_allocate is false, we return with zero
161  // otherwise we continue to allocate a block
162  if (!do_allocate)
163  return NULL;
164  //DEBUGPRINT("RANGE-INDEX=%d RANGE-MASK=%d" NL, range_index, range_mask);
165  // We're instructed to allocate block if cannot be found already elsewhere
166  // This condition checks if we have room in "tail" or there is any chunk already at all (tail is not NULL)
167  if (vdisk.tail && (vdisk.tail->used_blocks < vdisk.blocks_per_chunk)) {
168  v = vdisk.tail;
169  // OK, we had room in the tail, so put the block there
170  vdisk.all_blocks++;
171  if (block < v->block_no_min)
172  v->block_no_min = block;
173  if (block > v->block_no_max)
174  v->block_no_max = block;
175  vdisk.range_map[range_index] |= range_mask;
176  v->list[v->used_blocks] = block;
177  return v->base + ((v->used_blocks++) << 9);
178  }
179  // we don't have room in the tail, or not any chunk yet AT ALL!
180  // Let's allocate a new chunk, and use the first block from it!
181  v = xemu_malloc(vdisk.allocation_size); // xemu_malloc() is safe, it malloc()s space or abort the whole program if it cannot ...
182  v->next = NULL;
183  v->base = (Uint8*)v + vdisk.base_offset;
184  vdisk.all_chunks++;
185  if (vdisk.tail)
186  vdisk.tail->next = v;
187  vdisk.tail = v;
188  if (!vdisk.head)
189  vdisk.head = v;
190  v->list[0] = block;
191  v->used_blocks = 1;
192  v->block_no_min = block;
193  v->block_no_max = block;
194  vdisk.range_map[range_index] |= range_mask;
195  vdisk.all_blocks++;
196  return v->base; // no need to add block offset in storage, always the first block is served in a new chunk!
197 }

263 static int detect_compressed_image ( int fd )
264 {
265  static const char compressed_marker[] = "XemuBlockCompressedImage000";
266  Uint8 buf[512];
267  if (lseek(sdfd, 0, SEEK_SET) == OFF_T_ERROR || xemu_safe_read(fd, buf, 512) != 512)
268  return -1;
269  if (memcmp(buf, compressed_marker, sizeof compressed_marker)) {
270  DEBUGPRINT("SDCARD: image is not compressed" NL);
271  return 0;
272  }
273  if (((buf[0x1C] << 16) | (buf[0x1D] << 8) | buf[0x1E]) != 3) {
274  ERROR_WINDOW("Invalid/unknown compressed image format");
275  return -1;
276  }
277  sdcard_size_in_blocks = (buf[0x1F] << 16) | (buf[0x20] << 8) | buf[0x21];
278  DEBUGPRINT("SDCARD: compressed image with %u blocks" NL, sdcard_size_in_blocks);
279  sd_bdata_start = 3 * sdcard_size_in_blocks + 0x22;
280  sd_is_read_only = O_RDONLY;
281  return 1;
282 }
283 #endif

286 int sdcard_init ( const char *fn, const char *extd81fn, int virtsd_flag )
287 {
288  char fnbuf[PATH_MAX + 1];
289 #ifdef VIRTUAL_DISK_IMAGE_SUPPORT
290  if (virtsd_flag) {
291  virtdisk_init(VIRTUAL_DISK_BLOCKS_PER_CHUNK, VIRTUAL_DISK_SIZE_IN_BLOCKS);
292  vdisk.mode = 1;
293  } else
294  vdisk.mode = 0;
295 #endif
296  sdcard_set_external_d81_name(extd81fn);
297  d81access_init();
298  atexit(sdcard_shutdown);
299  KEEP_BUSY(0);
300  sd_status = 0;
301  fd_mounted = 0;
302  memset(sd_sector_registers, 0, sizeof sd_sector_registers);
303  memset(sd_d81_img1_start, 0, sizeof sd_d81_img1_start);
304  memset(sd_fill_buffer, sd_fill_value, 512);
305 #ifdef VIRTUAL_DISK_IMAGE_SUPPORT
306  if (vdisk.mode) {
307  sdfd = -1;
308  sd_is_read_only = 0;
309  sdcard_size_in_blocks = VIRTUAL_DISK_SIZE_IN_BLOCKS;
310 #ifdef COMPRESSED_SD
311  sd_compressed = 0;
312 #endif
313  DEBUGPRINT("SDCARD: card init done (VDISK!), size=%u Mbytes, virtsd_flag=%d" NL, sdcard_size_in_blocks >> 11, virtsd_flag);
314 #ifdef SD_CONTENT_SUPPORT
315  sdcontent_handle(sdcard_size_in_blocks, fn, SDCONTENT_FORCE_FDISK);
316 #endif
317  return 0;
318  }
319 #endif
320 retry:
321  sd_is_read_only = O_RDONLY;
322  sdfd = xemu_open_file(fn, O_RDWR, &sd_is_read_only, fnbuf);
323  if (sdfd < 0) {
324  int r = errno;
325  ERROR_WINDOW("Cannot open SD-card image %s, SD-card access won't work! ERROR: %s", fnbuf, strerror(r));
326  DEBUG("SDCARD: cannot open image %s" NL, fn);
327  if (r == ENOENT && !strcmp(fn, SDCARD_NAME)) {
328  r = QUESTION_WINDOW(
329  "No, thank you, I give up :(|Yes, create it for me :)"
330  ,
331  "Default SDCARD image does not exist. Would you like me to create one for you?\n"
332  "Note: it will be a 4Gbytes long file, since this is the minimal size for an SDHC card,\n"
333  "what MEGA65 needs. Do not worry, it's a 'sparse' file on most modern OSes which does\n"
334  "not takes as much disk space as its displayed size suggests.\n"
335  "This is unavoidable to emulate something uses an SDHC-card."
336  );
337  if (r) {
338  r = xemu_create_sparse_file(fnbuf, 4294967296UL);
339  if (r) {
340  ERROR_WINDOW("Couldn't create: %s", strerror(r));
341  } else {
342  goto retry;
343  }
344  }
345  }
346  } else {
347  if (sd_is_read_only)
348  INFO_WINDOW("Image file %s could be open only in R/O mode", fnbuf);
349  else
350  DEBUG("SDCARD: image file re-opened in RD/WR mode, good" NL);
351  // Check size!
352  DEBUG("SDCARD: cool, SD-card image %s (as %s) is open" NL, fn, fnbuf);
353  off_t size_in_bytes = xemu_safe_file_size_by_fd(sdfd);
354  if (size_in_bytes == OFF_T_ERROR) {
355  ERROR_WINDOW("Cannot query the size of the SD-card image %s, SD-card access won't work! ERROR: %s", fn, strerror(errno));
356  close(sdfd);
357  sdfd = -1;
358  return sdfd;
359  }
360 #ifdef COMPRESSED_SD
361  sd_compressed = detect_compressed_image(sdfd);
362  if (sd_compressed < 0) {
363  ERROR_WINDOW("Error while trying to detect compressed SD-image");
364  sdcard_size_in_blocks = 0; // just cheating to trigger error handling later
365  }
366 #endif
367  sdcard_size_in_blocks = size_in_bytes >> 9;
368  DEBUG("SDCARD: detected size in Mbytes: %d" NL, (int)(size_in_bytes >> 20));
369  if (size_in_bytes < 67108864UL) {
370  ERROR_WINDOW("SD-card image is too small! Min required size is 64Mbytes!");
371  close(sdfd);
372  sdfd = -1;
373  return sdfd;
374  }
375  if (size_in_bytes & (off_t)511) {
376  ERROR_WINDOW("SD-card image size is not multiple of 512 bytes!!");
377  close(sdfd);
378  sdfd = -1;
379  return sdfd;
380  }
381  if (size_in_bytes > 34359738368UL) {
382  ERROR_WINDOW("SD-card image is too large! Max allowed size is 32Gbytes!");
383  close(sdfd);
384  sdfd = -1;
385  return sdfd;
386  }
387  }
388  if (sdfd >= 0) {
389  DEBUGPRINT("SDCARD: card init done, size=%u Mbytes, virtsd_flag=%d" NL, sdcard_size_in_blocks >> 11, virtsd_flag);
390 #ifdef SD_CONTENT_SUPPORT_DANGEROUS
391  sdcontent_handle(sdcard_size_in_blocks, NULL, SDCONTENT_ASK_FDISK | SDCONTENT_ASK_FILES);
392 #endif
393  }
394  return sdfd;
395 }

404 static int host_seek ( Uint32 block )
405 {
406  if (sdfd < 0)
407  FATAL("host_seek is called with invalid sdfd!"); // FIXME: this check can go away, once we're sure it does not apply!
408  off_t offset;
409 #ifdef COMPRESSED_SD
410  if (sd_compressed) {
411  offset = block * 3 + 0x22;
412  if (lseek(sdfd, offset, SEEK_SET) != offset)
413  FATAL("SDCARD: SEEK: compressed image host-OS seek failure: %s", strerror(errno));
414  Uint8 buf[3];
415  if (xemu_safe_read(sdfd, buf, 3) != 3)
416  FATAL("SDCARD: SEEK: compressed image host-OK pre-read failure: %s", strerror(errno));
417  compressed_block = (buf[0] & 0x80);
418  buf[0] &= 0x7F;
419  offset = ((off_t)((buf[0] << 16) | (buf[1] << 8) | buf[2]) << 9) + sd_bdata_start;
420  //DEBUGPRINT("SD-COMP: got address: %d" NL, (int)offset);
421  } else {
422  offset = (off_t)block << 9;
423  }
424 #else
425  offset = (off_t)block << 9;
426 #endif
427  if (lseek(sdfd, offset, SEEK_SET) != offset)
428  FATAL("SDCARD: SEEK: image seek host-OS failure: %s", strerror(errno));
429  return 0;
430 }

511  is_write ? "writing" : "reading",
512  block, cpu65.pc
513  );
514  if (XEMU_UNLIKELY(sd_status & SD_ST_EXT_BUS)) {
515  DEBUGPRINT("SDCARD: bus #1 is empty" NL);
516  // FIXME: what kind of error we should create here?????

518  KEEP_BUSY(1);
519  return;
520  }

522  int ret = is_write ? sdcard_write_block(block, buffer) : sdcard_read_block(block, buffer);

562 static void sdcard_command ( Uint8 cmd )
563 {
564  static Uint32 multi_io_block;
565  static Uint8 sd_last_ok_cmd;
566  DEBUG("SDCARD: writing command register $D680 with $%02X PC=$%04X" NL, cmd, cpu65.pc);
567  sd_status &= ~(SD_ST_BUSY1 | SD_ST_BUSY0); // ugly hack :-@
568  KEEP_BUSY(0);
569  switch (cmd) {
570  case 0x00: // RESET SD-card
571  case 0x10: // RESET SD-card with flags specified [FIXME: I don't know what the difference is ...]
572  sd_status = SD_ST_RESET | (sd_status & SD_ST_EXT_BUS); // clear all other flags, but not the bus selection, FIXME: bus selection should not be touched?
573  memset(sd_sector_registers, 0, sizeof sd_sector_registers);
574  sdhc_mode = 1;
575  break;
576  case 0x01: // END RESET
577  case 0x11: // ... [FIXME: again, I don't know what the difference is ...]
578  sd_status &= ~(SD_ST_RESET | SD_ST_ERROR | SD_ST_FSM_ERROR);
579  break;
580  case 0x02: // read block
581  sdcard_block_io(U8A_TO_U32(sd_sector_registers), 0);
582  break;
583  case 0x03: // write block
584  sdcard_block_io(U8A_TO_U32(sd_sector_registers), 1);
585  break;
586  case 0x04: // multi sector write - first sector
587  if (sd_last_ok_cmd != 0x04) {
588  multi_io_block = U8A_TO_U32(sd_sector_registers);
589  sdcard_block_io(multi_io_block, 1);
590  } else {
591  DEBUGPRINT("SDCARD: bad multi-command sequence command $%02X after command $%02X" NL, cmd, sd_last_ok_cmd);
592  sd_status |= SD_ST_ERROR | SD_ST_FSM_ERROR;
593  }
594  break;
595  case 0x05: // multi sector write - not the first, neither the last sector
596  if (sd_last_ok_cmd == 0x04 || sd_last_ok_cmd == 0x05) {
597  multi_io_block++;
598  sdcard_block_io(multi_io_block, 1);
599  } else {
600  DEBUGPRINT("SDCARD: bad multi-command sequence command $%02X after command $%02X" NL, cmd, sd_last_ok_cmd);
601  sd_status |= SD_ST_ERROR | SD_ST_FSM_ERROR;
602  }
603  break;
604  case 0x06: // multi sector write - last sector
605  if (sd_last_ok_cmd == 0x04 || sd_last_ok_cmd == 0x05) {
606  multi_io_block++;
607  sdcard_block_io(multi_io_block, 1);
608  } else {
609  DEBUGPRINT("SDCARD: bad multi-command sequence command $%02X after command $%02X" NL, cmd, sd_last_ok_cmd);
610  sd_status |= SD_ST_ERROR | SD_ST_FSM_ERROR;
611  }
612  break;
613  case 0x0C: // request flush of the SD-card [currently does nothing in Xemu ...]
614  break;
615  case 0x40: // SDHC mode OFF - Not supported on newer M65s!
616  sd_status &= ~SD_ST_SDHC;
617  sdhc_mode = 0;
618  break;
619  case 0x41: // SDHC mode ON - Not supported on newer M65s!
620  sd_status |= SD_ST_SDHC;
621  sdhc_mode = 1;
622  break;
623  case 0x44: // sd_clear_error <= '0' FIXME: what is this?
624  break;
625  case 0x45: // sd_clear_error <= '1' FIXME: what is this?
626  break;
627  case 0x81: // map SD-buffer
628  sd_status |= SD_ST_MAPPED;
629  sd_status &= ~(SD_ST_ERROR | SD_ST_FSM_ERROR);
630  break;
631  case 0x82: // unmap SD-buffer
632  sd_status &= ~(SD_ST_MAPPED | SD_ST_ERROR | SD_ST_FSM_ERROR);
633  break;
634  case 0x83: // SD write fill mode set
635  sd_fill_mode = 1;
636  break;
637  case 0x84: // SD write fill mode clear
638  sd_fill_mode = 0;
639  break;
640  case 0xC0: // select internal SD-card bus
641  sd_status &= ~SD_ST_EXT_BUS;
642  break;
643  case 0xC1: // select external SD-card bus
644  sd_status |= SD_ST_EXT_BUS;
645  break;
646  default:
647  sd_status |= SD_ST_ERROR;
648  DEBUGPRINT("SDCARD: warning, unimplemented SD-card controller command $%02X" NL, cmd);
649  break;
650  }
651  if (XEMU_UNLIKELY(sd_status & (SD_ST_ERROR | SD_ST_FSM_ERROR))) {
652  sd_last_ok_cmd = 0xFF;
653  } else {
654  sd_last_ok_cmd = cmd;
655  }
656 }
