161 int d81access_attach_fsobj ( const char *fn, int mode )
162 {
163  if (!fn || !*fn) {
164  DEBUGPRINT("D81: attach file request with empty file name, not using FS based disk attachment." NL);
165  return -1;
166  }
167  if (mode & D81ACCESS_DIR) {
168  // if we passed D81ACCESS_DIR, we try to open the named object as a directory first.
169  // if it was OK, let's containue with that
170  // if not OK and error was ENOTDIR or ENOENT then simply assume to continue with other methods (not as directory).
171  // this is because, the "fn" parameter can be a relative path too, later can be used with relative-to-preferences directory or so.
172  // though directory opening is always absolute path what we're assuming.
173  DIR *dir = opendir(fn);
174  if (dir) {
175  // It seems we could open the "raw" object as directory
176  d81access_close_internal();
177  d81.dir = dir;
178  d81.mode = D81ACCESS_DIR | D81ACCESS_RO | D81ACCESS_AUTOCLOSE; // TODO? directory access is always read only currently ...
179  DEBUGPRINT("D81: file system object \"%s\" opened as a directory." NL, fn);
180  d81access_cb_chgmode(d81.mode);
181  return 0;
182  } else if (errno != ENOTDIR && errno != ENOENT) {
183  ERROR_WINDOW("D81: cannot open directory %s for virtual D81 mode: %s", fn, strerror(errno));
184  return 1;
185  }
186  }
187  // So, we can assume that the object should be a file ...
188  if (!(mode & (D81ACCESS_IMG | D81ACCESS_PRG))) {
189  if (mode & D81ACCESS_DIR)
190  DEBUGPRINT("D81: could not open file system object \"%s\" as a directory.", fn);
191  else
192  FATAL("d81access_attach_fsobj(): insane mode argument, no DIR,IMG,PRG given");
193  return 1; // but if no success with directory open, and no request for file based attach, then we give up here
194  }
195  // OK, so some file based open request can happen, let's continue
196  char fnbuf[PATH_MAX + 1];
197  int ro, fd = xemu_open_file(fn, IS_RO(mode) ? O_RDONLY : O_RDWR, &ro, fnbuf);
198  if (fd < 0) {
199  ERROR_WINDOW("D81: image/program file was specified (%s) but it cannot be opened: %s", fn, strerror(errno));
200  return 1;
201  }
202  ro = (IS_RO(mode) || ro) ? D81ACCESS_RO : 0;
203  off_t size = xemu_safe_file_size_by_fd(fd);
204  if (size == OFF_T_ERROR) {
205  ERROR_WINDOW("D81: Cannot query the size of external D81 image/program file %s ERROR: %s", fn, strerror(errno));
206  close(fd);
207  return 1;
208  }
209  // Check if it's a PRG-file mode, based on the "sane" size of such a file ...
210  if (size >= PRG_MIN_SIZE && size <= PRG_MAX_SIZE) {
211  if (mode & D81ACCESS_PRG) {
212  d81access_attach_fd_internal(fd, 0, D81ACCESS_PRG | D81ACCESS_AUTOCLOSE | D81ACCESS_RO);
213  d81.prg_size = size; // store real size of the object
214  d81.prg_blk_size = size / 254;
215  d81.prg_blk_last_size = size % 254;
216  if (d81.prg_blk_last_size)
217  d81.prg_blk_size++;
218  else
219  d81.prg_blk_last_size = 254;
220  return 0;
221  } else {
222  close(fd);
223  ERROR_WINDOW("Specified size for D81 seems to be a program file (too small for real D81), but PRG mode virtual disk feature was not requested");
224  return 1;
225  }
226  }
227  if (!(mode & D81ACCESS_IMG)) {
228  close(fd);
229  ERROR_WINDOW("D81 image mode was not requested ...");
230  return 1;
231  }
232  // Only the possibility left that it's a D81 image
233  if (size == D81_SIZE) {
234  // candidate for the "normal" D81 as being used
235  d81access_attach_fd_internal(fd, 0, D81ACCESS_IMG | D81ACCESS_AUTOCLOSE | ro);
236  return 0;
237  }
238  close(fd);
239  ERROR_WINDOW("Cannot guess the type of object (from its size) wanted to use for floppy emulation, sorry");
240  return 1;

277 static int file_io_op ( int is_write, int d81_offset, Uint8 *buffer, int size )
278 {
279  off_t offset = d81.start_at + (off_t)d81_offset;
280  if (lseek(d81.fd, offset, SEEK_SET) != offset)
281  FATAL("D81: SEEK: seek host-OS failure: %s", strerror(errno));
282  int ret = is_write ? xemu_safe_write(d81.fd, buffer, size) : xemu_safe_read(d81.fd, buffer, size);
283  if (ret >= 0)
284  return ret;
285  FATAL("D81: %s: host-OS error: %s", is_write ? "WRITE" : "READ", strerror(errno));
286  return -1;
287 }

340 static int read_prg ( Uint8 *buffer, int d81_offset, int number_of_logical_sectors )
341 {
342  // just pre-zero buffer, so we don't need to take care on this at various code points with possible partly filled output
343  memset(buffer, 0, 512);
344  // disk organization at CBM-DOS level is 256 byte sector based, though FDC F011 itself is 512 bytes sectored stuff
345  // so we always need to check to 256 bytes "DOS-evel" sectors even if F011 itself handled 512 bytes long sectors
346  for (; number_of_logical_sectors; number_of_logical_sectors--, d81_offset += 0x100, buffer += 0x100) {
347  DEBUGPRINT("D81VIRTUAL: reading sub-sector @ $%X" NL, d81_offset);
348  if (d81_offset == 0x61800) { // the header sector
349  memcpy(buffer, vdsk_head_sect, sizeof vdsk_head_sect);
350  } else if (d81_offset == 0x61900 || d81_offset == 0x61A00) { // BAM sectors (we don't handle BAM entries at all, so it will be a filled disk ...)
351  if (d81_offset == 0x61900) {
352  buffer[0] = 0x28;
353  buffer[1] = 0x02;
354  } else
355  buffer[1] = 0xFF; // chain, byte #0 is already 0
356  buffer[2] = 0x44;
357  buffer[3] = 0xBB;
358  buffer[4] = vdsk_head_sect[0x16];
359  buffer[5] = vdsk_head_sect[0x17];
360  buffer[6] = 0xC0;
361  } else if (d81_offset == 0x61B00) { // directory sector, the only one we want to handle here
362  buffer[2] = 0x82; // PRG
363  buffer[3] = 0x01; // starts on track-1
364  // starts on sector-0 of track-1, 0 is already set
365  memcpy(buffer + 5, vdsk_file_name, 16);
366  buffer[0x1E] = d81.prg_blk_size & 0xFF;
367  buffer[0x1F] = d81.prg_blk_size >> 8;
368  } else { // what we want to handle at all yet, is the file itself, which starts at the very beginning at our 'virtual' disk
369  int block = d81_offset >> 8; // calculate the block from offset
370  if (block < d81.prg_blk_size) { // so it seems, we need to do something here at last, disk area belongs to our file!
371  int reqsize, ret;
372  if (block == d81.prg_blk_size -1) { // last block of file
373  reqsize = d81.prg_blk_last_size;
374  buffer[1] = 0xFF; // offs 0 is already 0
375  } else { // not the last block, we must resolve the track/sector info of the next block
376  reqsize = 254;
377  buffer[0] = ((block + 1) / 40) + 1;
378  buffer[1] = (block + 1) % 40;
379  }
380  ret = file_io_op(0, block * 254, buffer + 2, reqsize);
381  DEBUGPRINT("D81VIRTUAL: ... data block, block number %d, next_track = $%02X next_sector = $%02X" NL, block, buffer[0], buffer[1]);
382 #if 0
383  if (host_seek_to(NULL, block * 254, "reading[PRG81VIRT@HOST]", d81_is_prg + 512, d81fd) < 0)
384  return -1;
385  block = xemu_safe_read(d81fd, buffer + 2, reqsize);
386 #endif
387  DEBUGPRINT("D81VIRTUAL: ... reading result: expected %d retval %d" NL, reqsize, ret);
388  if (ret != reqsize)
389  return -1;
390  } // if it's not our block of the file, not BAMs, header block or directory, the default zeroed area is returned, what we memset()'ed to zero
391  }
392  }
393  return 0;
394 }

406 int d81access_read_sect ( Uint8 *buffer, int d81_offset, int sector_size )
407 {
408  check_io_req_params(d81_offset, sector_size);
409  switch (d81.mode & 0xFF) {
410  case D81ACCESS_EMPTY:
411  return -1;
412  case D81ACCESS_IMG:
413  if (file_io_op(0, d81_offset, buffer, sector_size) == sector_size)
414  return 0;
415  else
416  return -1;
417  case D81ACCESS_PRG:
418  return read_prg(buffer, d81_offset, sector_size >> 8);
419  case D81ACCESS_DIR:
420  FATAL("DIR access method is not yet implemented in Xemu, sorry :-(");
421  case D81ACCESS_CALLBACKS:
422  return d81.read_cb(buffer, d81.start_at + d81_offset, sector_size);
423  default:
424  FATAL("d81access_read_sect(): invalid d81.mode & 0xFF");
425  }
426  return -1;
427 }
