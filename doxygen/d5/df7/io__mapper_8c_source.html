57 static void update_hw_multiplier ( void )
58 {
59  D7XX[BIGMULT_ADDR + 3] &= 0x01;
60  D7XX[BIGMULT_ADDR + 6] &= 0x03;
61  D7XX[BIGMULT_ADDR + 7] = 0x00;
62  register Uint64 result = (Uint64)(
63  ((Uint32) D7XX[BIGMULT_ADDR + 0] ) |
64  ((Uint32) D7XX[BIGMULT_ADDR + 1] << 8) |
65  ((Uint32) D7XX[BIGMULT_ADDR + 2] << 16) |
66  ((Uint32) D7XX[BIGMULT_ADDR + 3] << 24)
67  ) * (Uint64)(
68  ((Uint32) D7XX[BIGMULT_ADDR + 4] ) |
69  ((Uint32) D7XX[BIGMULT_ADDR + 5] << 8) |
70  ((Uint32) D7XX[BIGMULT_ADDR + 6] << 16) |
71  ((Uint32) D7XX[BIGMULT_ADDR + 7] << 24)
72  );
73  D7XX[BIGMULT_ADDR + 0x8] = (result ) & 0xFF;
74  D7XX[BIGMULT_ADDR + 0x9] = (result >> 8) & 0xFF;
75  D7XX[BIGMULT_ADDR + 0xA] = (result >> 16) & 0xFF;
76  D7XX[BIGMULT_ADDR + 0xB] = (result >> 24) & 0xFF;
77  D7XX[BIGMULT_ADDR + 0xC] = (result >> 32) & 0xFF;
78  D7XX[BIGMULT_ADDR + 0xD] = (result >> 40) & 0xFF;
79  D7XX[BIGMULT_ADDR + 0xE] = 0;
80  D7XX[BIGMULT_ADDR + 0xF] = 0;
81  bigmult_valid_result = 1;
82 }

86 /* Internal decoder for I/O reads. Address *must* be within the 0-$3FFF (!!) range. The low 12 bits is the actual address inside the I/O area,
87  while the most significant nibble shows the I/O mode the operation is meant, according to the following table:
88  0 = C64 (VIC-II) I/O mode
89  1 = C65 (VIC-III) I/O mode
90  2 = *INVALID* should not happen, unless some maps the $FF-megabyte M65 specific area, then it should do nothing or so ...
91  3 = M65 (VIC-IV) I/O mode.
92  Even writing the "classic" I/O area at $D000 will ends up here. These addressed are stricly based on the VIC mode
93  according the table above, no $D000 addresses here at all!
94  FIXME: it seems I/O mode "2" may be not invalid, but some tricky stuff, C64 with extensions, or such. I have not so much idea yet :(
95  NOTE: vic_read_reg() and vic_write_reg() is kinda special! It uses offset'ed register addresses for different VIC mode registers ... */
96 
97 Uint8 io_read ( unsigned int addr )
98 {
99  // DEBUG("IO: read $%03X IO_mode is %d @ PC=$%04X" NL, addr & 0xFFF, addr >> 12, cpu65.pc);
100  switch (addr >> 8) {
101  /* ---------------------------------------------------- */
102  /* $D000-$D3FF: VIC-II, VIC-III+FDC+REC, VIC-IV+FDC+REC */
103  /* ---------------------------------------------------- */
104  case 0x00: // $D000-$D0FF ~ C64 I/O mode
105  case 0x01: // $D100-$D1FF ~ C64 I/O mode
106  case 0x02: // $D200-$D2FF ~ C64 I/O mode
107  case 0x03: // $D300-$D3FF ~ C64 I/O mode
108  return vic_read_reg((addr & 0x3F) | 0x100); // VIC-II read register
109  case 0x10: // $D000-$D0FF ~ C65 I/O mode
110  addr &= 0xFF;
111  if (XEMU_LIKELY(addr < 0x80))
112  return vic_read_reg(addr | 0x80); // VIC-III read register
113  if (XEMU_LIKELY(addr < 0xA0))
114  return fdc_read_reg(addr & 0xF);
115  RETURN_ON_IO_READ_NOT_IMPLEMENTED("RAM expansion controller", 0xFF);
116  case 0x30: // $D000-$D0FF ~ M65 I/O mode
117  addr &= 0xFF;
118  if (XEMU_LIKELY(addr < 0x80))
119  return vic_read_reg(addr); // VIC-IV read register
120  if (XEMU_LIKELY(addr < 0xA0))
121  return fdc_read_reg(addr & 0xF);
122  RETURN_ON_IO_READ_NOT_IMPLEMENTED("RAM expansion controller", 0xFF);
123  case 0x11: // $D100-$D1FF ~ C65 I/O mode
124  case 0x12: // $D200-$D2FF ~ C65 I/O mode
125  case 0x13: // $D300-$D3FF ~ C65 I/O mode
126  case 0x31: // $D100-$D1FF ~ M65 I/O mode
127  case 0x32: // $D200-$D2FF ~ M65 I/O mode
128  case 0x33: // $D300-$D3FF ~ M65 I/O mode
129  return 0xFF; // FIXME: check, if applies: palette register entries are write-only, in both of C65 and M65 modes??
130  /* ------------------------------------------------ */
131  /* $D400-$D7FF: SID, SID+UART+DMA, SID+UART+DMA+M65 */
132  /* ------------------------------------------------ */
133  case 0x04: // $D400-$D4FF ~ C64 I/O mode
134  case 0x05: // $D500-$D5FF ~ C64 I/O mode
135  case 0x06: // $D600-$D6FF ~ C64 I/O mode
136  case 0x07: // $D700-$D7FF ~ C64 I/O mode
137  case 0x14: // $D400-$D4FF ~ C65 I/O mode
138  case 0x15: // $D500-$D5FF ~ C65 I/O mode
139  case 0x34: // $D400-$D4FF ~ M65 I/O mode
140  case 0x35: // $D500-$D5FF ~ M65 I/O mode
141  if (is_mouse_grab()) { // Rudimentary C1351 mouse emulation (on SID#1) ...
142  switch (addr & 0x5F) {
143  case 0x19:
144  mouse_x = (mouse_x + hid_read_mouse_rel_x(-31, 31)) & 63;
145  return mouse_x << 1;
146  case 0x1A:
147  mouse_y = (mouse_y - hid_read_mouse_rel_y(-31, 31)) & 63;
148  return mouse_y << 1;
149  }
150  }
151  return 0xFF;
152  case 0x16: // $D600-$D6FF ~ C65 I/O mode
153  RETURN_ON_IO_READ_NOT_IMPLEMENTED("UART", 0xFF); // FIXME: UART is not yet supported!
154  case 0x36: // $D600-$D6FF ~ M65 I/O mode
155  addr &= 0xFF;
156  if (addr < 9)
157  RETURN_ON_IO_READ_NOT_IMPLEMENTED("UART", 0xFF); // FIXME: UART is not yet supported!
158  if (addr >= 0x80 && addr <= 0x93) // SDcard controller etc of Mega65
159  return sdcard_read_register(addr - 0x80);
160  if ((addr & 0xF0) == 0xE0)
161  return eth65_read_reg(addr);
162  switch (addr) {
163  case 0x7C:
164  return 0; // emulate the "UART is ready" situation (used by newer kickstarts around from v0.11 or so)
165  case 0x7E: // upgraded hypervisor signal
166  if (D6XX_registers[0x7E] == 0x80) // 0x80 means for Xemu (not for a real M65!): ask the user!
167  D6XX_registers[0x7E] = QUESTION_WINDOW(
168  "Not upgraded yet, it can do it|Already upgraded, I test kicked state",
169  "Kickstart asks hypervisor upgrade state. What do you want Xemu to answer?\n"
170  "(don't worry, it won't be asked again without RESET)"
171  ) ? 0xFF : 0;
172  return D6XX_registers[0x7E];
173  case 0x7F:
174  return in_hypervisor ? 'H' : 'U'; // FIXME: I am not sure about 'U' here (U for userspace, H for hypervisor mode)
175  case 0xF0:
176  return fpga_switches & 0xFF;
177  case 0xF1:
178  return (fpga_switches >> 8) & 0xFF;
179  case 0x10: // last keypress ASCII value
180  return hwa_kbd_get_last();
181  case 0x11: // modifier keys on kbd being used
182  return hwa_kbd_get_modifiers();
183  default:
184  DEBUG("MEGA65: reading Mega65 specific I/O @ $D6%02X result is $%02X" NL, addr, D6XX_registers[addr]);
185  return D6XX_registers[addr];
186  }
187  case 0x17: // $D700-$D7FF ~ C65 I/O mode
188  // FIXME: really a partial deconding like this? really on every 16 bytes?!
189  return dma_read_reg(addr & 0xF);
190  case 0x37: // $D700-$D7FF ~ M65 I/O mode
191  // FIXME: this is probably very bad! I guess DMA does not decode for every 16 addresses ... Proposed fix is here:
192  addr &= 0xFF;
193  if (addr < 16)
194  return dma_read_reg(addr & 0xF);
195  if (XEMU_UNLIKELY(!bigmult_valid_result))
196  update_hw_multiplier();
197  return D7XX[addr];
198  /* ----------------------- */
199  /* $D800-$DBFF: COLOUR RAM */
200  /* ----------------------- */
201  case 0x08: // $D800-$D8FF ~ C64 I/O mode
202  case 0x09: // $D900-$D9FF ~ C64 I/O mode
203  case 0x0A: // $DA00-$DAFF ~ C64 I/O mode
204  case 0x0B: // $DB00-$DBFF ~ C64 I/O mode
205  return colour_ram[addr - 0x0800] | 0xF0; // FIXME: is this true, that high nibble if faked to be '1' in C64 I/O mode, always?
206  case 0x18: // $D800-$D8FF ~ C65 I/O mode
207  case 0x19: // $D900-$D9FF ~ C65 I/O mode
208  case 0x1A: // $DA00-$DAFF ~ C65 I/O mode
209  case 0x1B: // $DB00-$DBFF ~ C65 I/O mode
210  return colour_ram[addr - 0x1800];
211  case 0x38: // $D800-$D8FF ~ M65 I/O mode
212  case 0x39: // $D900-$D9FF ~ M65 I/O mode
213  case 0x3A: // $DA00-$DAFF ~ M65 I/O mode
214  case 0x3B: // $DB00-$DBFF ~ M65 I/O mode
215  return colour_ram[addr - 0x3800];
216  /* --------------------------------------- */
217  /* $DC00-$DCFF: CIA#1, EXTENDED COLOUR RAM */
218  /* --------------------------------------- */
219  case 0x0C: // $DC00-$DCFF ~ C64 I/O mode
220  return (vic_registers[0x30] & 1) ? colour_ram[addr - 0x0800] : cia_read(&cia1, addr & 0xF);
221  case 0x1C: // $DC00-$DCFF ~ C65 I/O mode
222  return (vic_registers[0x30] & 1) ? colour_ram[addr - 0x1800] : cia_read(&cia1, addr & 0xF);
223  case 0x3C: // $DC00-$DCFF ~ M65 I/O mode
224  return (vic_registers[0x30] & 1) ? colour_ram[addr - 0x3800] : cia_read(&cia1, addr & 0xF);
225  /* --------------------------------------- */
226  /* $DD00-$DDFF: CIA#2, EXTENDED COLOUR RAM */
227  /* --------------------------------------- */
228  case 0x0D: // $DD00-$DDFF ~ C64 I/O mode
229  return (vic_registers[0x30] & 1) ? colour_ram[addr - 0x0800] : cia_read(&cia2, addr & 0xF);
230  case 0x1D: // $DD00-$DDFF ~ C65 I/O mode
231  return (vic_registers[0x30] & 1) ? colour_ram[addr - 0x1800] : cia_read(&cia2, addr & 0xF);
232  case 0x3D: // $DD00-$DDFF ~ M65 I/O mode
233  return (vic_registers[0x30] & 1) ? colour_ram[addr - 0x3800] : cia_read(&cia2, addr & 0xF);
234  /* --------------------------------------------------- */
235  /* $DE00-$DFFF: IO exp, EXTENDED COLOUR RAM, SD buffer */
236  /* --------------------------------------------------- */
237  case 0x0E: // $DE00-$DEFF ~ C64 I/O mode
238  case 0x0F: // $DF00-$DFFF ~ C64 I/O mode
239  if (vic_registers[0x30] & 1)
240  return colour_ram[addr - 0x0800];
241  if (XEMU_LIKELY(sd_status & SD_ST_MAPPED))
242  return sd_buffer[addr - 0x0E00];
243  return 0xFF; // I/O exp is not supported
244  case 0x1E: // $DE00-$DEFF ~ C65 I/O mode
245  case 0x1F: // $DF00-$DFFF ~ C65 I/O mode
246  if (vic_registers[0x30] & 1)
247  return colour_ram[addr - 0x1800];
248  if (XEMU_LIKELY(sd_status & SD_ST_MAPPED))
249  return sd_buffer[addr - 0x1E00];
250  return 0xFF; // I/O exp is not supported
251  case 0x3E: // $DE00-$DEFF ~ M65 I/O mode
252  case 0x3F: // $DF00-$DFFF ~ M65 I/O mode
253  if (vic_registers[0x30] & 1)
254  return colour_ram[addr - 0x3800];
255  if (XEMU_LIKELY(sd_status & SD_ST_MAPPED))
256  return sd_buffer[addr - 0x3E00];
257  return 0xFF; // I/O exp is not supported
258  /* --------------------------------------------------------------- */
259  /* $2xxx I/O area is not supported: FIXME: what is that for real?! */
260  /* --------------------------------------------------------------- */
261  case 0x20: case 0x21: case 0x22: case 0x23: case 0x24: case 0x25: case 0x26: case 0x27:
262  case 0x28: case 0x29: case 0x2A: case 0x2B: case 0x2C: case 0x2D: case 0x2E: case 0x2F:
263  return 0xFF;
264  default:
265  FATAL("Xemu internal error: undecoded I/O area reading for address $(%X)%03X", addr >> 8, addr & 0xFFF);
266  }
267 }

271 /* Please read comments at io_read() above, those apply here too.
272  In nutshell: this function *NEEDS* addresses 0-$3FFF based on the given I/O (VIC) mode! */
273 void io_write ( unsigned int addr, Uint8 data )
274 {
275  // DEBUG("IO: write $%03X with data $%02X IO_mode is %d @ PC=$%04X" NL, addr & 0xFFF, data, addr >> 12, cpu65.pc);
276  if (XEMU_UNLIKELY(cpu_rmw_old_data >= 0)) {
277  // RMW handling! FIXME: do this only in the needed I/O ports only, not here, globally!
278  // however, for that, we must check this at every devices where it can make any difference ...
279  Uint8 old_data = cpu_rmw_old_data;
280  cpu_rmw_old_data = -1; // set this back to minus _before_ doing self-call, or it would cause an infinite recursion-like madness ...
281  DEBUG("RMW: I/O internal addr %04X old data %02X new data %02X" NL, addr, old_data, data);
282  io_write(addr, old_data);
283  }
284  switch (addr >> 8) {
285  /* ---------------------------------------------------- */
286  /* $D000-$D3FF: VIC-II, VIC-III+FDC+REC, VIC-IV+FDC+REC */
287  /* ---------------------------------------------------- */
288  case 0x00: // $D000-$D0FF ~ C64 I/O mode
289  case 0x01: // $D100-$D1FF ~ C64 I/O mode
290  case 0x02: // $D200-$D2FF ~ C64 I/O mode
291  case 0x03: // $D300-$D3FF ~ C64 I/O mode
292  vic_write_reg((addr & 0x3F) | 0x100, data); // VIC-II write register
293  return;
294  case 0x10: // $D000-$D0FF ~ C65 I/O mode
295  addr &= 0xFF;
296  if (XEMU_LIKELY(addr < 0x80)) {
297  vic_write_reg(addr | 0x80, data); // VIC-III write register
298  return;
299  }
300  if (XEMU_LIKELY(addr < 0xA0)) {
301  fdc_write_reg(addr & 0xF, data);
302  return;
303  }
304  RETURN_ON_IO_WRITE_NOT_IMPLEMENTED("RAM expansion controller");
305  case 0x30: // $D000-$D0FF ~ M65 I/O mode
306  addr &= 0xFF;
307  if (XEMU_LIKELY(addr < 0x80)) {
308  vic_write_reg(addr, data); // VIC-IV write register
309  return;
310  }
311  if (XEMU_LIKELY(addr < 0xA0)) {
312  fdc_write_reg(addr & 0xF, data);
313  return;
314  }
315  RETURN_ON_IO_WRITE_NOT_IMPLEMENTED("RAM expansion controller");
316  case 0x11: // $D100-$D1FF ~ C65 I/O mode
317  case 0x12: // $D200-$D2FF ~ C65 I/O mode
318  case 0x13: // $D300-$D3FF ~ C65 I/O mode
319  vic3_write_palette_reg(addr - 0x1100, data);
320  return;
321  case 0x31: // $D100-$D1FF ~ M65 I/O mode
322  case 0x32: // $D200-$D2FF ~ M65 I/O mode
323  case 0x33: // $D300-$D3FF ~ M65 I/O mode
324  vic4_write_palette_reg(addr - 0x3100, data);
325  return;
326  /* ------------------------------------------------ */
327  /* $D400-$D7FF: SID, SID+UART+DMA, SID+UART+DMA+M65 */
328  /* ------------------------------------------------ */
329  case 0x04: // $D400-$D4FF ~ C64 I/O mode
330  case 0x05: // $D500-$D5FF ~ C64 I/O mode
331  case 0x06: // $D600-$D6FF ~ C64 I/O mode
332  case 0x07: // $D700-$D7FF ~ C64 I/O mode
333  case 0x14: // $D400-$D4FF ~ C65 I/O mode
334  case 0x15: // $D500-$D5FF ~ C65 I/O mode
335  case 0x34: // $D400-$D4FF ~ M65 I/O mode
336  case 0x35: // $D500-$D5FF ~ M65 I/O mode
337  sid_write_reg(addr & 0x40 ? &sid2 : &sid1, addr & 31, data);
338  return;
339  case 0x16: // $D600-$D6FF ~ C65 I/O mode
340  if ((addr & 0xFF) == 0x07) {
341  port_d607 = data;
342  return;
343  } else
344  RETURN_ON_IO_WRITE_NOT_IMPLEMENTED("UART"); // FIXME: UART is not yet supported!
345  case 0x36: // $D600-$D6FF ~ M65 I/O mode
346  addr &= 0xFF;
347  if (!in_hypervisor && addr >= 0x40 && addr <= 0x7F) {
348  // In user mode, writing to $D640-$D67F (in VIC4 iomode) causes to enter hypervisor mode with
349  // the trap number given by the offset in this range
350  hypervisor_enter(addr & 0x3F);
351  return;
352  }
353  D6XX_registers[addr] = data; // I guess, the actual write won't happens if it was trapped, so I moved this to here after the previous "if"
354  if (addr < 9) {
355  if (addr == 7) {
356  port_d607 = data;
357  return;
358  } else
359  RETURN_ON_IO_WRITE_NOT_IMPLEMENTED("UART"); // FIXME: UART is not yet supported!
360  }
361  if (addr >= 0x80 && addr <= 0x93) { // SDcard controller etc of Mega65
362  sdcard_write_register(addr - 0x80, data);
363  return;
364  }
365  if ((addr & 0xF0) == 0xE0) {
366  eth65_write_reg(addr, data);
367  return;
368  }
369  switch (addr) {
370  case 0x10: // ASCII kbd last press value to zero whatever the written data would be
371  hwa_kbd_move_next();
372  return;
373  case 0x7C: // hypervisor serial monitor port
374  hypervisor_serial_monitor_push_char(data);
375  return;
376  case 0x7D:
377  DEBUG("MEGA65: features set as $%02X" NL, data);
378  if ((data & 2) != cpu_linear_memory_addressing_is_enabled) {
379  DEBUG("MEGA65: 32-bit linear addressing opcodes have been turned %s." NL, data & 2 ? "ON" : "OFF");
380  cpu_linear_memory_addressing_is_enabled = data & 2;
381  }
382  if ((data & 4) != rom_protect) {
383  DEBUG("MEGA65: ROM protection has been turned %s." NL, data & 4 ? "ON" : "OFF");
384  rom_protect = data & 4;
385  }
386  force_fast = data & 16;
387  return;
388  case 0x7E:
389  D6XX_registers[0x7E] = 0xFF; // iomap.txt: "Hypervisor already-upgraded bit (sets permanently)"
390  DEBUG("MEGA65: Writing already-kicked register $%04X!" NL, addr);
391  hypervisor_debug_invalidate("$D67E was written, maybe new kickstart will boot!");
392  return;
393  case 0x7F: // hypervisor leave
394  hypervisor_leave(); // 0x67F is also handled on enter's state, so it will be executed only in_hypervisor mode, which is what I want
395  return;
396  default:
397  DEBUG("MEGA65: this I/O port is not emulated in Xemu yet: $D6%02X (tried to be written with $%02X)" NL, addr, data);
398  return;
399  }
400  return;
401  case 0x17: // $D700-$D7FF ~ C65 I/O mode
402  // FIXME: really a partial deconding like this? really on every 16 bytes?!
403  dma_write_reg(addr & 0xF, data);
404  return;
405  case 0x37: // $D700-$D7FF ~ M65 I/O mode
406  // FIXME: this is probably very bad! I guess DMA does not decode for every 16 addresses ... Proposed fix is here:
407  addr &= 0xFF;
408  if (addr < 16)
409  dma_write_reg(addr & 0xF, data);
410  else if (XEMU_UNLIKELY((addr & 0xF0) == BIGMULT_ADDR))
411  bigmult_valid_result = 0;
412  D7XX[addr] = data;
413  return;
414  /* ----------------------- */
415  /* $D800-$DBFF: COLOUR RAM */
416  /* ----------------------- */
417  case 0x08: // $D800-$D8FF ~ C64 I/O mode
418  case 0x09: // $D900-$D9FF ~ C64 I/O mode
419  case 0x0A: // $DA00-$DAFF ~ C64 I/O mode
420  case 0x0B: // $DB00-$DBFF ~ C64 I/O mode
421  colour_ram[addr - 0x0800] = data & 0xF; // FIXME: is this true, that high nibble is masked, so switching to C65/M65 mode high nibble will be zero??
422  return;
423  case 0x18: // $D800-$D8FF ~ C65 I/O mode
424  case 0x19: // $D900-$D9FF ~ C65 I/O mode
425  case 0x1A: // $DA00-$DAFF ~ C65 I/O mode
