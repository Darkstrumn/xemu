88 int xemugui_init ( const char *name )
89 {
90  char avail[256]; // "should be enough" (yeah, I know, the 640K ...) for some names ...
91  avail[0] = 0;
92  is_xemugui_ok = 0;
93  for (int a = 0 ;; a++) {
94  strcat(avail, " ");
95  strcat(avail, xemugui_descriptor_list[a]->name);
96  if (name && !strcasecmp(xemugui_descriptor_list[a]->name, name)) {
97  current_gui = xemugui_descriptor_list[a];
98  break;
99  }
100  if (xemugui_descriptor_list[a] == &xemunullgui_descriptor) {
101  current_gui = xemugui_descriptor_list[0];
102  if (name)
103  ERROR_WINDOW(
104  "Requested GUI (\"%s\") cannot be found, using \"%s\" instead.\nAvailable GUI implementations:%s",
105  name, current_gui->name, avail
106  );
107  else
108  DEBUGPRINT("GUI: no GUI was specified, using the first available one from this list:%s" NL, avail);
109  break;
110  }
111  }
112  DEBUGPRINT("GUI: using \"%s\" (%s)" NL, current_gui->name, current_gui->description);
113  return current_gui->init ? current_gui->init() : 1;
114 }
