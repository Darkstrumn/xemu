150 Z80EX_BYTE z80ex_pread_cb ( Z80EX_WORD port16 )
151 {
152  port16 &= 0xFF;
153  DEBUG("IO: reading I/O port %02Xh" NL, port16);
154  switch (port16) {
155  case 0x58: case 0x5C:
156  DEBUG("Reading keyboard (row=%d, result=$%02X)!" NL, keyboard_row, kbd_matrix[keyboard_row]);
157  return keyboard_row < 10 ? kbd_matrix[keyboard_row] : 0xFF;
158  case 0x59: case 0x5D:
159  return interrupt_active ? 0xEF: 0xFF;
160  case 0x70:
161  return crtc.regsel;
162  case 0x71:
163  return crtc_read_register(crtc.regsel);
164  }
165  return 0xFF;
166 }

170 void z80ex_pwrite_cb ( Z80EX_WORD port16, Z80EX_BYTE value )
171 {
172  port16 &= 0xFF;
173  io_port_values[port16] = value;
174  if (port16 != 2) // to avoid flooding ...
175  DEBUG("IO: writing I/O port %02Xh with data %02Xh" NL, port16, value);
176  switch (port16) {
177  case 0x00:
178  border_colour = TVC_COLOUR_BYTE_TO_SDL(value >> 1);
179  break;
180  case 0x02:
181  // page 3
182  mem.rd_selector[3] = memory_rd_selectors_for_page3[value >> 6];
183  mem.wr_selector[3] = memory_wr_selectors_for_page3[value >> 6];
184  // page 2
185  if (value & 32) {
186  mem.rd_selector[2] = memcbrd_ram_u2;
187  mem.wr_selector[2] = memcbwr_ram_u2;
188  } else {
189  mem.rd_selector[2] = memcbrd_ram_video;
190  mem.wr_selector[2] = memcbwr_ram_video;
191  }
192  // page 0
193  mem.rd_selector[0] = memory_rd_selectors_for_page0[(value >> 3) & 3];
194  mem.wr_selector[0] = memory_wr_selectors_for_page0[(value >> 3) & 3];
195  // page 1
196  if (!(value & 4)) {
197  mem.rd_selector[1] = memcbrd_ram_u1;
198  mem.wr_selector[1] = memcbwr_ram_u1;
199  } else {
200  mem.rd_selector[1] = memcbrd_ram_video;
201  mem.wr_selector[1] = memcbwr_ram_video;
202  }
203  break;
204  case 0x03:
205  // z80ex_pwrite_cb(2, io_port_values[2]); // TODO: we need this later with expansion mem paging!
206  keyboard_row = value & 0xF; // however the lower 4 bits are for selecting row
207  break;
208  case 0x05:
209  DEBUG("Enabled_SoundIT=%d Enabled_CursorIT=%d" NL, value & 32 ? 1 : 0, value & 16 ? 1 : 0);
210  break;
211  case 0x06:
212  colour_mode = value & 3;
213  if (colour_mode == 3)
214  colour_mode = 2;
215  DEBUG("VIDEO: colour mode is %d" NL, colour_mode);
216  break;
217  case 0x07:
218  // clear cursor/sound IT. FIXME: any write would do it?!
219  interrupt_active = 0;
220  break;
221  case 0x0C: case 0x0D: case 0x0E: case 0x0F:
222  crtc.vmem = mem.video_ram + ((value & 0x30) << 10);
223  mem.vmem = mem.video_ram + ((value & 0x03) << 14);
224  break;
225  case 0x60: case 0x61: case 0x62: case 0x63: case 0x64: case 0x65: case 0x66: case 0x67:
226  case 0x68: case 0x69: case 0x6A: case 0x6B: case 0x6C: case 0x6D: case 0x6E: case 0x6F:
227  palette[port16 & 3] = TVC_COLOUR_BYTE_TO_SDL(value);
228  break;
229  case 0x70: case 0x72: case 0x74: case 0x76: case 0x78: case 0x7A: case 0x7C: case 0x7E:
230  crtc.regsel = value & 31;
231  break;
232  case 0x71: case 0x73: case 0x75: case 0x77: case 0x79: case 0x7B: case 0x7D: case 0x7F:
233  crtc_write_register(crtc.regsel, value);
234  break;
235  }
236 }

257 // Well :) It's just an ugly hack to render one frame at once.
258 // There is "some CRTC emulation", though far from being all
259 // features are expected to work!
260 // Note: though the selected 16K (TVC64+ is emulated!) is used ...
261 static inline void render_tvc_screen ( void )
262 {
263  int tail, y;
264  Uint32 *pix = xemu_start_pixel_buffer_access(&tail);
265  int ma = (crtc.registers[12] << 8) | crtc.registers[13]; // CRTC MA signals, 14 bit
266  int ra = 0; // CRTC RA signals
267  int start_line, limit_line, start_cpos, limit_cpos;
268  //DEBUG("CRTCINFO: start addr = $%04X" NL, ma);
269  //for (y = 0; y < sizeof crtc.registers; y++)
270  // DEBUG("CRTCINFO: R%02d=%d" NL, y, crtc.registers[y]);
271  start_line = (SCREEN_HEIGHT - (crtc.registers[6] * (crtc.registers[9] + 1))) >> 1;
272  limit_line = SCREEN_HEIGHT - start_line;
273  start_cpos = ((SCREEN_WIDTH >> 3) - crtc.registers[1]) >> 1;
274  limit_cpos = (SCREEN_WIDTH >> 3) - start_cpos;
275  for (y = 0; y < SCREEN_HEIGHT; y++) {
276  int x;
277  if (y >= start_line && y < limit_line) { // active content!
278  int addr = (ma & 63) | (ra << 6) | ((ma & 0xFC0) << 2);
279  for (x = 0; x < (SCREEN_WIDTH >> 3); x++) {
280  if (x >= start_cpos && x < limit_cpos) {
281  Uint8 b = crtc.vmem[(addr++) & 0x3FFF];
282  switch (colour_mode) {
283  case 0: // 2-colours mode
284  pix[0] = palette[(b >> 7) & 1];
285  pix[1] = palette[(b >> 6) & 1];
286  pix[2] = palette[(b >> 5) & 1];
287  pix[3] = palette[(b >> 4) & 1];
288  pix[4] = palette[(b >> 3) & 1];
289  pix[5] = palette[(b >> 2) & 1];
290  pix[6] = palette[(b >> 1) & 1];
291  pix[7] = palette[ b & 1];
292  break;
293  case 1: // 4-colours mode
294  pix[0] = pix[1] = palette[((b & 0x80) >> 7) | ((b & 0x08) >> 2)];
295  pix[2] = pix[3] = palette[((b & 0x40) >> 6) | ((b & 0x04) >> 1)];
296  pix[4] = pix[5] = palette[((b & 0x20) >> 5) | (b & 0x02) ];
297  pix[6] = pix[7] = palette[((b & 0x10) >> 4) | ((b & 0x01) << 1)];
298  break;
299  case 2: // 16-colours mode
300  pix[0] = pix[1] = pix[2] = pix[3] = palette_col16_pixel1[b];
301  pix[4] = pix[5] = pix[6] = pix[7] = palette_col16_pixel2[b];
302  break;
303  }
304  } else // sider border
305  pix[0] = pix[1] = pix[2] = pix[3] = pix[4] = pix[5] = pix[6] = pix[7] = border_colour;
306  pix += 8;
307  }
308  if (ra == crtc.registers[9]) {
309  ra = 0;
310  ma += crtc.registers[1];
311  } else
312  ra++;
313  } else // top or bottom border ...
314  for (x = 0; x < SCREEN_WIDTH; x++)
315  *pix++ = border_colour;
316  pix += tail;
317  }
318  xemu_update_screen();
319 }

331 static void update_emulator ( void )
332 {
333  // FIXME: Ugly trick, we generate IRQ here, not by CRTC cursor stuff, what would do it for real on TVC!
334  // This means, that TVC uses the cursor info of CRTC "abused" to be used an a periodic interrupt source. Tricky!
335  // However, this implementation of mine if very wrong, and the exact time for the IRQ should depend on CRTC reg settings
336  // I THINK ... :-D
337  if (io_port_values[5] & 16) { // if enabled at all!
338  interrupt_active = 1;
339  DEBUG("Cursor interrupt!" NL);
340  }
341  // Rest of the update stuff, but only at 25Hz rate ...
342  if (!frameskip) {
343  render_tvc_screen();
344  hid_handle_all_sdl_events();
345  xemu_timekeeping_delay(40000); // number: the time needed (real-time) for a "full frame"
346  }
347 }

350 static void init_tvc ( void )
351 {
352  int a;
353  // Initialize colours (TODO: B&W mode is not used currently!)
354  for (a = 0; a < 16; a++) {
355  int red = (a & 2) ? ((a & 8) ? 0xFF : 0x80) : 0;
356  int green = (a & 4) ? ((a & 8) ? 0xFF : 0x80) : 0;
357  int blue = (a & 1) ? ((a & 8) ? 0xFF : 0x80) : 0;
358  int y = 0.299 * red + 0.587 * green + 0.114 * blue;
359  tvc_palette_rgb[a] = SDL_MapRGBA(sdl_pix_fmt, red, green, blue, 0xFF);
360  tvc_palette_bw [a] = SDL_MapRGBA(sdl_pix_fmt, y, y, y, 0xFF);
361  }
362  // Initialize helper tables for 16 colours mode
363  // 16 colour mode does not use the 4 element palette register, but layout is not "TVC colour byte" ...
364  // TVC "colour byte is": -I-G-R-B
365  // Format in 16 col mode:
366  // * first pixel: I-G-R-B-
367  // * second pixel: -I-G-R-B (this matches the colour byte layout, btw!)
368  for (a = 0; a < 256; a++) {
369  palette_col16_pixel1[a] = TVC_COLOUR_BYTE_TO_SDL(a >> 1);
370  palette_col16_pixel2[a] = TVC_COLOUR_BYTE_TO_SDL(a);
371  }
372  // I/O, ROM and RAM intialization ...
373  memset(&mem, 0xFF, sizeof mem);
374  memset(&io_port_values, 0x00, 0x100); // some I/O handlers re-calls itself with other values, so we zero the stuff first
375  memset(crtc.registers, 0x00, sizeof crtc.registers);
376  for (a = 0; a < 0x100; a++) // ... and now use the callback which sets our variables for some initial value ...
377  z80ex_pwrite_cb(a, 0);
378  for (a = 0; a < sizeof crtc.registers; a++)
379  crtc_write_register(a, 0);
380  if (xemu_load_file("#tvc22_d6_64k.rom", mem.sys_rom + 0x0000, 0x2000, 0x2000, NULL) < 0 ||
381  xemu_load_file("#tvc22_d4_64k.rom", mem.sys_rom + 0x2000, 0x2000, 0x2000, NULL) < 0 ||
382  xemu_load_file("#tvc22_d7_64k.rom", mem.ext_rom + 0x2000, 0x2000, 0x2000, NULL) < 0
383  )
384  FATAL("Cannot load ROM(s).");
385 #ifdef CONFIG_SDEXT_SUPPORT
386  if (xemucfg_get_bool("sdext"))
387  sdext_init(xemucfg_get_str("sdimg"), xemucfg_get_str("sdrom"));
388 #endif
389 }

395 static void emulation_loop ( void )
396 {
397  for (;;) { // our emulation loop ...
398  if (interrupt_active) {
399  int a = z80ex_int();
400  if (a)
401  cycles += a;
402  else
403  cycles += z80ex_step();
404  } else
405  cycles += z80ex_step();
406  if (cycles >= CLOCKS_PER_FRAME) {
407  update_emulator();
408  frameskip = !frameskip;
409  cycles -= CLOCKS_PER_FRAME;
410  return;
411  }
412  }

437  NULL, // -- "" --

441  NULL // no emulator specific shutdown function

447  SDL_DISABLE // no joystick HID events

458  XEMU_MAIN_LOOP(emulation_loop, 25, 1);
