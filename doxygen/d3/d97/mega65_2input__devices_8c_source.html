106 /* basically the opposite as kbd_get_last() but this one used internally only */
107 static void hwa_kbd_convert_and_push ( int pos )
108 {
109  // Xemu has a design to have key positions stored in row/col as low/high nible of a byte
110  // normalize this here, to have a linear index
111  int i = ((pos & 0xF0) >> 1) | (pos & 7);
112  if (i < MAT2ASC_TAB_SIZE) {
113  int ascii = matrix_to_ascii_table_selector[hwa_kbd.modifiers & 0xF][i];
114  if (ascii) {
115  if (!hwa_kbd.next) {
116  DEBUGKBDHWA("KBD: HWA: storing key $%02X '%c' from kbd pos $%02X and table index $%02X at PC=$%04X" NL, ascii, CHR_EQU(ascii), pos, i, cpu65.pc);
117  hwa_kbd.next = ascii;
118  } else
119  DEBUGKBDHWA("KBD: HWA: NOT storing key (already waiting) $%02X '%c' from kbd pos $%02X and table index $%02X at PC=$%04X" NL, ascii, CHR_EQU(ascii), pos, i, cpu65.pc);
120  } else
121  DEBUGKBDHWA("KBD: HWA: NOT storing key (zero in translation table) from kbd pos $%02X and table index $%02X at PC=$%04X" NL, pos, i, cpu65.pc);
122  } else
123  DEBUGKBDHWA("KBD: HWA: NOT storing key (outside of translation table) from kbd pos $%02X and table index $%02X at PC=$%04X" NL, pos, i, cpu65.pc);
124 }

163 void kbd_trigger_restore_trap ( void )
164 {
165  if (XEMU_UNLIKELY(restore_is_held)) {
166  restore_is_held++;
167  if (restore_is_held >= 20) {
168  restore_is_held = 0;
169  if (!in_hypervisor) {

171  KBD_RELEASE_KEY(RESTORE_KEY_POS);
172  hypervisor_enter(TRAP_RESTORE);
173  } else
174  DEBUGPRINT("KBD: *IGNORING* RESTORE trap trigger, already in hypervisor mode!" NL);
175  }

193 // Called by emutools_hid!!! to handle special private keys assigned to this emulator
194 int emu_callback_key ( int pos, SDL_Scancode key, int pressed, int handled )
195 {
196  // Update status of modifier keys
197  hwa_kbd.modifiers =
198  (IS_KEY_PRESSED(LSHIFT_KEY_POS) ? MODKEY_LSHIFT : 0)
199  | (IS_KEY_PRESSED(RSHIFT_KEY_POS) ? MODKEY_RSHIFT : 0)
200  | (IS_KEY_PRESSED(CTRL_KEY_POS) ? MODKEY_CTRL : 0)
201  | (IS_KEY_PRESSED(CBM_KEY_POS) ? MODKEY_CBM : 0)
202 #ifdef ALT_KEY_POS
203  | (IS_KEY_PRESSED(ALT_KEY_POS) ? MODKEY_ALT : 0)
204 #endif
205 #ifdef SCRL_KEY_POS
206  | (IS_KEY_PRESSED(SCRL_KEY_POS) ? MODKEY_SCRL : 0)
207 #endif
208  ;
209  DEBUGKBD("KBD: HWA: pos = %d sdl_key = %d, pressed = %d, handled = %d" NL, pos, key, pressed, handled);
210  if (pressed) {
211  // check if we have the ALT-TAB trap triggered (TAB is pressed now, and ALT is hold)
212  if (pos == TAB_KEY_POS && (hwa_kbd.modifiers & MODKEY_ALT)) {
213  kbd_trigger_alttab_trap();
214  return 0;
215  }
216  // RESTORE triggered trap is different as it depends on timing (how long it's pressed)
217  // So we just flag this, and the main emulation loop need to increment the value to see if the long press event comes, and trigger the trap.
218  // This is done by main emulation loop calling kbd_trigger_restore_trap() function, see above in this very source.
219  // Please note about the pair of this condition below with the "else" branch of the "pressed" condition.
220  if (pos == RESTORE_KEY_POS)
221  restore_is_held = 1;
222  // Check to be sure, some special Xemu internal stuffs uses kbd matrix positions does not exist for real
223  if (pos >= 0 && pos < 0x100)
224  hwa_kbd_convert_and_push(pos);
225  // Also check for special, emulator-related hot-keys
226  if (key == SDL_SCANCODE_F10) {
227  reset_mega65_asked();
228  } else if (key == SDL_SCANCODE_KP_ENTER) {
229  c64_toggle_joy_emu();
230  OSD(-1, -1, "Joystick emulation on port #%d", joystick_emu);
231  } else if (((hwa_kbd.modifiers & (MODKEY_LSHIFT | MODKEY_RSHIFT)) == (MODKEY_LSHIFT | MODKEY_RSHIFT)) && set_mouse_grab(SDL_FALSE)) {
232  DEBUGPRINT("UI: mouse grab cancelled" NL);
233  }
234  } else {
235  if (pos == RESTORE_KEY_POS)
236  restore_is_held = 0;
237  if (pos == -2 && key == 0) { // special case pos = -2, key = 0, handled = mouse button (which?) and release event!
238  if ((handled == SDL_BUTTON_LEFT) && set_mouse_grab(SDL_TRUE)) {
239  OSD(-1, -1, "Mouse grab activated. Press\nboth SHIFTs together to cancel.");
240  DEBUGPRINT("UI: mouse grab activated" NL);
241  }
242  if (handled == SDL_BUTTON_RIGHT) {
243  ui_enter();
244  }
245  }
246  }
247  return 0;
248 }
