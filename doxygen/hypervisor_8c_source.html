55 int hypervisor_debug_init ( const char *fn, int hypervisor_debug, int use_hypervisor_serial_out_asciizer )
56 {
57  char buffer[1024];
58  FILE *fp;
59  int fd;
60  hypervisor_serial_out_asciizer = use_hypervisor_serial_out_asciizer;
61  if (!fn || !*fn) {
62  DEBUG("MEGADEBUG: feature is not enabled, null file name for list file" NL);
63  return 1;
64  }
65  for (fd = 0; fd < 0x4000; fd++) {
66  debug_lines[fd][0][0] = 0;
67  }
68  fd = xemu_open_file(fn, O_RDONLY, NULL, NULL);
69  if (fd < 0) {
70  INFO_WINDOW("Cannot open %s, no resolved symbols will be used.", fn);
71  return 1;
72  }
73  fp = fdopen(fd, "rb");
74  while (fgets(buffer, sizeof buffer, fp)) {
75  int addr = 0xFFFF;
76  char *p, *p1, *p2;
77  if (sscanf(buffer, "%04X", &addr) != 1) continue;
78  if (addr < 0x8000 || addr >= 0xC000) continue;
79  p2 = strchr(buffer, '|');
80  if (!p2) continue; // no '|' pipe in the line, skip
81  if (strchr(p2 + 1, '|')) continue; // more than one '|' pipes, it's a hex dump, skip
82  *(p2++) = 0;
83  p = strrchr(p2, '/');
84  if (p)
85  p2 = p + 1;
86  else {
87  p = strrchr(p2, '\\');
88  if (p)
89  p2 = p + 1;
90  }
91  while (*p2 <= 32 && *p2)
92  p2++;
93  p = p2 + strlen(p2) - 1;
94  while (p > p2 && *p <= 32)
95  *(p--) = 0;
96  if (strlen(p2) >= 32)
97  FATAL("Bad list file, too long file reference part");
98  // that was awful. Now the first part
99  p1 = buffer + 5;
100  while (*p1 && *p1 <= 32)
101  p1++;
102  p = p1 + strlen(p1) - 1;
103  while (p > p1 && *p <= 32)
104  *(p--) = 0;
105  if (strlen(p2) >= 32)
106  FATAL("Bad list file, too long assembly part");
107  // And finally, copy line to its well deserved place(s)
108  snprintf(debug_lines[addr - 0x8000][0], INFO_MAX_SIZE, "%s", p1);
109  snprintf(debug_lines[addr - 0x8000][1], INFO_MAX_SIZE, "%s", p2);
110  }
111  fclose(fp);
112  close(fd);
113  resolver_ok = 1;
114  debug_on = hypervisor_debug;
115  return 0;
116 }

137 void hypervisor_enter ( int trapno )
138 {
139  // Sanity checks
140  if (XEMU_UNLIKELY(trapno > 0x7F || trapno < 0))
141  FATAL("FATAL: got invalid trap number %d", trapno);
142  if (XEMU_UNLIKELY(in_hypervisor))
143  FATAL("FATAL: already in hypervisor mode while calling hypervisor_enter()");
144  // First, save machine status into hypervisor registers, TODO: incomplete, can be buggy!
145  D6XX_registers[0x40] = cpu65.a;
146  D6XX_registers[0x41] = cpu65.x;
147  D6XX_registers[0x42] = cpu65.y;
148  D6XX_registers[0x43] = cpu65.z;
149  D6XX_registers[0x44] = cpu65.bphi >> 8; // "B" register
150  D6XX_registers[0x45] = cpu65.s;
151  D6XX_registers[0x46] = cpu65.sphi >> 8; // stack page register
152  D6XX_registers[0x47] = cpu65_get_pf();
153  D6XX_registers[0x48] = cpu65.pc & 0xFF;
154  D6XX_registers[0x49] = cpu65.pc >> 8;
155  D6XX_registers[0x4A] = ((map_offset_low >> 16) & 0x0F) | ((map_mask & 0x0F) << 4);
156  D6XX_registers[0x4B] = ( map_offset_low >> 8) & 0xFF ;
157  D6XX_registers[0x4C] = ((map_offset_high >> 16) & 0x0F) | ( map_mask & 0xF0);
158  D6XX_registers[0x4D] = ( map_offset_high >> 8) & 0xFF ;
159  D6XX_registers[0x4E] = map_megabyte_low >> 20;
160  D6XX_registers[0x4F] = map_megabyte_high >> 20;
161  D6XX_registers[0x50] = memory_get_cpu_io_port(0);
162  D6XX_registers[0x51] = memory_get_cpu_io_port(1);
163  D6XX_registers[0x52] = vic_iomode;
164  D6XX_registers[0x53] = dma_registers[5]; // GS $D653 - Hypervisor DMAgic source MB
165  D6XX_registers[0x54] = dma_registers[6]; // GS $D654 - Hypervisor DMAgic destination MB
166  D6XX_registers[0x55] = dma_registers[0]; // GS $D655 - Hypervisor DMAGic list address bits 0-7
167  D6XX_registers[0x56] = dma_registers[1]; // GS $D656 - Hypervisor DMAGic list address bits 15-8
168  D6XX_registers[0x57] = (dma_registers[2] & 15) | ((dma_registers[4] & 15) << 4); // GS $D657 - Hypervisor DMAGic list address bits 23-16
169  D6XX_registers[0x58] = dma_registers[4] >> 4; // GS $D658 - Hypervisor DMAGic list address bits 27-24
170  // Now entering into hypervisor mode
171  in_hypervisor = 1; // this will cause apply_memory_config to map hypervisor RAM, also for checks later to out-of-bound execution of hypervisor RAM, etc ...
172  vic_iomode = VIC4_IOMODE;
173  memory_set_cpu_io_port_ddr_and_data(0x3F, 0x35); // sets all-RAM + I/O config up!
174  cpu65.pf_d = 0; // clear decimal mode ... according to Paul, punnishment will be done, if it's removed :-)
175  cpu65.pf_i = 1; // disable IRQ in hypervisor mode
176  cpu65.pf_e = 1; // 8 bit stack in hypervisor mode
177  cpu65.sphi = 0xBE00; // set a nice shiny stack page
178  cpu65.bphi = 0xBF00; // ... and base page (aka zeropage)
179  cpu65.s = 0xFF;
180  // Set mapping for the hypervisor
181  map_mask = (map_mask & 0xF) | 0x30; // mapping: 0011XXXX (it seems low region map mask is not changed by hypervisor entry)
182  map_megabyte_high = 0xFF << 20;
183  map_offset_high = 0xF0000;
184  memory_set_vic3_rom_mapping(0); // for VIC-III rom mapping disable in hypervisor mode
185  memory_set_do_map(); // now the memory mapping is changed
186  machine_set_speed(0); // set machine speed (hypervisor always runs at M65 fast ... ??) FIXME: check this!
187  cpu65.pc = 0x8000 | (trapno << 2); // load PC with the address assigned for the given trap number
188  DEBUG("HYPERVISOR: entering into hypervisor mode, trap=$%02X @ $%04X -> $%04X" NL, trapno, D6XX_registers[0x48] | (D6XX_registers[0x49] << 8), cpu65.pc);
189 }

202 void hypervisor_leave ( void )
203 {
204  // Sanity check
205  if (XEMU_UNLIKELY(!in_hypervisor))
206  FATAL("FATAL: not in hypervisor mode while calling hypervisor_leave()");
207  // First, restore machine status from hypervisor registers
208  DEBUG("HYPERVISOR: leaving hypervisor mode @ $%04X -> $%04X" NL, cpu65.pc, D6XX_registers[0x48] | (D6XX_registers[0x49] << 8));
209  cpu65.a = D6XX_registers[0x40];
210  cpu65.x = D6XX_registers[0x41];
211  cpu65.y = D6XX_registers[0x42];
212  cpu65.z = D6XX_registers[0x43];
213  cpu65.bphi = D6XX_registers[0x44] << 8; // "B" register
214  cpu65.s = D6XX_registers[0x45];
215  cpu65.sphi = D6XX_registers[0x46] << 8; // stack page register
216  cpu65_set_pf(D6XX_registers[0x47]);
217  cpu65.pf_e = D6XX_registers[0x47] & CPU65_PF_E; // cpu65_set_pf() does NOT set 'E' bit by design, so we do at our own
218  cpu65.pc = D6XX_registers[0x48] | (D6XX_registers[0x49] << 8);
219  map_offset_low = ((D6XX_registers[0x4A] & 0xF) << 16) | (D6XX_registers[0x4B] << 8);
220  map_offset_high = ((D6XX_registers[0x4C] & 0xF) << 16) | (D6XX_registers[0x4D] << 8);
221  map_mask = (D6XX_registers[0x4A] >> 4) | (D6XX_registers[0x4C] & 0xF0);
222  map_megabyte_low = D6XX_registers[0x4E] << 20;
223  map_megabyte_high = D6XX_registers[0x4F] << 20;
224  memory_set_cpu_io_port_ddr_and_data(D6XX_registers[0x50], D6XX_registers[0x51]);
225  vic_iomode = D6XX_registers[0x52] & 3;
226  if (vic_iomode == VIC_BAD_IOMODE)
227  vic_iomode = VIC3_IOMODE; // I/O mode "2" (binary: 10) is not used, I guess
228  dma_registers[5] = D6XX_registers[0x53]; // GS $D653 - Hypervisor DMAgic source MB
229  dma_registers[6] = D6XX_registers[0x54]; // GS $D654 - Hypervisor DMAgic destination MB
230  dma_registers[0] = D6XX_registers[0x55]; // GS $D655 - Hypervisor DMAGic list address bits 0-7
231  dma_registers[1] = D6XX_registers[0x56]; // GS $D656 - Hypervisor DMAGic list address bits 15-8
232  dma_registers[2] = D6XX_registers[0x57] & 15; //
233  dma_registers[4] = (D6XX_registers[0x57] >> 4) | (D6XX_registers[0x58] << 4);
234  // Now leaving hypervisor mode ...
235  in_hypervisor = 0;
236  machine_set_speed(0); // restore speed ...
237  memory_set_vic3_rom_mapping(vic_registers[0x30]); // restore possible active VIC-III mapping
238  memory_set_do_map(); // restore mapping ...
239  if (XEMU_UNLIKELY(first_hypervisor_leave)) {
240  first_hypervisor_leave = 0;
241  int new_pc = refill_c65_rom_from_preinit_cache(); // this function should decide then, if it's really a (forced) thing to do ...
242  if (new_pc >= 0) {
243  // positive return value from the re-fill routine: we DID re-fill, we should re-initialize "user space" PC from the return value
244  DEBUGPRINT("MEM: force ROM re-apply policy, PC change: $%04X -> $%04X" NL,
245  cpu65.pc, new_pc
246  );
247  cpu65.pc = new_pc;
248  } else
249  DEBUGPRINT("MEM: no force ROM re-apply policy was requested" NL);
250  dma_init_set_rev(xemucfg_get_num("dmarev"), main_ram + 0x20000 + 0x16);
251  }
252  if (XEMU_UNLIKELY(hypervisor_queued_trap >= 0)) {
253  // Not so much used currently ...
254  DEBUG("HYPERVISOR: processing queued trap on leaving hypervisor: trap #$%02X" NL, hypervisor_queued_trap);
255  hypervisor_enter(hypervisor_queued_trap);
256  hypervisor_queued_trap = -1;
257  }
258 }

262 void hypervisor_serial_monitor_push_char ( Uint8 chr )
263 {
264  if (hypervisor_monout_p >= hypervisor_monout - 1 + sizeof hypervisor_monout)
265  return;
266  int flush = (chr == 0x0A || chr == 0x0D || chr == 0x8A || chr == 0x8D);
267  if (hypervisor_monout_p == hypervisor_monout && flush)
268  return;
269  if (flush) {
270  *hypervisor_monout_p = 0;
271  if (hypervisor_serial_out_asciizer) {
272  unsigned char *p = (unsigned char*)hypervisor_monout;
273  while (*p) {
274  if (*p >= 0x61 && *p <= 0x7A)
275  *p -= 0x20;
276  else if (*p >= 0xC1 && *p <= 0xDA)
277  *p -= 0x80;
278  else if (*p < 0x20 || *p >= 0x80)
279  *p = '?';
280  p++;
281  }
282  }
283  fprintf(stderr, "Hypervisor serial output: \"%s\"." NL, hypervisor_monout);
284  DEBUG("MEGA65: Hypervisor serial output: \"%s\"." NL, hypervisor_monout);
285  hypervisor_monout_p = hypervisor_monout;
286  return;
287  }
288  *(hypervisor_monout_p++) = (char)chr;
289 }

303 void hypervisor_debug ( void )
304 {
305  if (!in_hypervisor)
306  return;
307  // TODO: better hypervisor upgrade check, maybe with checking the exact range kickstart uses for upgrade outside of the "normal" hypervisor mem range
308  if (XEMU_UNLIKELY((cpu65.pc & 0xFF00) == 0x3000)) { // this area is used by kickstart upgrade
309  DEBUG("HYPERVISOR-DEBUG: allowed to run outside of hypervisor memory, no debug info, PC = $%04X" NL, cpu65.pc);
310  return;
311  }
312  if (XEMU_UNLIKELY((cpu65.pc & 0xC000) != 0x8000)) {
313  DEBUG("HYPERVISOR-DEBUG: execution outside of the hypervisor memory, PC = $%04X" NL, cpu65.pc);
314  ERROR_WINDOW("Hypervisor fatal error: execution outside of the hypervisor memory, PC=$%04X SP=$%04X", cpu65.pc, cpu65.sphi | cpu65.s);
315  if (QUESTION_WINDOW("Reset|Exit Xemu", "What to do now?"))
316  XEMUEXIT(1);
317  else
318  hypervisor_start_machine();
319  return;
320  }
321  if (!resolver_ok) {
322  return; // no debug info loaded from kickstart.list ...
323  }
324  if (XEMU_UNLIKELY(!debug_lines[cpu65.pc - 0x8000][0][0])) {
325  DEBUG("HYPERVISOR-DEBUG: execution address not found in list file (out-of-bound code?), PC = $%04X" NL, cpu65.pc);
326  FATAL("Hypervisor fatal error: execution address not found in list file (out-of-bound code?), PC = $%04X", cpu65.pc);
327  return;
328  }
329  // WARNING: as it turned out, using stdio I/O to log every opcodes even "only" at ~3.5MHz rate makes emulation _VERY_ slow ...
330  if (XEMU_UNLIKELY(debug_on)) {
331  if (debug_fp) {
332  Uint8 pf = cpu65_get_pf();
333  fprintf(
334  debug_fp,
335  "HYPERVISOR-DEBUG: %-32s PC=%04X SP=%04X B=%02X A=%02X X=%02X Y=%02X Z=%02X P=%c%c%c%c%c%c%c%c IO=%d OPC=%02X @ %s" NL,
336  debug_lines[cpu65.pc - 0x8000][0],
337  cpu65.pc, cpu65.sphi | cpu65.s, cpu65.bphi >> 8, cpu65.a, cpu65.x, cpu65.y, cpu65.z,
338  (pf & CPU65_PF_N) ? 'N' : 'n',
339  (pf & CPU65_PF_V) ? 'V' : 'v',
340  (pf & CPU65_PF_E) ? 'E' : 'e',
341  '-',
342  (pf & CPU65_PF_D) ? 'D' : 'd',
343  (pf & CPU65_PF_I) ? 'I' : 'i',
344  (pf & CPU65_PF_Z) ? 'Z' : 'z',
345  (pf & CPU65_PF_C) ? 'C' : 'c',
346  vic_iomode,
347  cpu65_read_callback(cpu65.pc),
348  debug_lines[cpu65.pc - 0x8000][1]
349  );
350  }
351  }
