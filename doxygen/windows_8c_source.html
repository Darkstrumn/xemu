54 static int xemuwingui_file_selector ( int dialog_mode, const char *dialog_title, char *default_dir, char *selected, int path_max_size )
55 {
56  int res;
57  OPENFILENAME ofn; // common dialog box structure
58  ZeroMemory(&ofn, sizeof ofn);
59  ofn.lStructSize = sizeof ofn;
60  ofn.hwndOwner = xemuwinmenu.win_hwnd; // FIXME: it should be this way, though it seems sometimes works better with the value 0 ...
61  ofn.lpstrFile = selected;
62  *selected = '\0'; // sets to zero, since it seems windows dialog handler also used this as input?
63  ofn.nMaxFile = path_max_size;
64  ofn.lpstrFilter = NULL; // "All\0*.*\0Text\0*.TXT\0";
65  ofn.nFilterIndex = 0; // 1
66  ofn.lpstrFileTitle = NULL;
67  ofn.nMaxFileTitle = 0;
68  ofn.lpstrInitialDir = default_dir ? default_dir : NULL;
69  ofn.lpstrTitle = dialog_title;
70  ofn.Flags = OFN_PATHMUSTEXIST | OFN_FILEMUSTEXIST | OFN_EXPLORER | OFN_HIDEREADONLY | OFN_NOCHANGEDIR;
71  res = !GetOpenFileName(&ofn);
72  if (res) {
73  int err = CommDlgExtendedError();
74  *selected = '\0';
75  DEBUGGUI("GUI: file selector (Windows) error code: %04Xh for HWND owner %p" NL, err, ofn.hwndOwner);
76  if (err)
77  ERROR_WINDOW("Windows CommDlgExtendedError: %04Xh for HWND owner %p", err, ofn.hwndOwner);
78  } else
79  store_dir_from_file_selection(default_dir, selected, dialog_mode);
80  //xemunativegui_iteration();
81  xemu_drop_events();
82  return res;
83 }

88 static HMENU _wingui_recursive_menu_builder ( const struct menu_st desc[] )
89 {
90  HMENU menu = CreatePopupMenu();
91  if (!menu) {
92  ERROR_WINDOW("CreatePopupMenu() failed in menu builder!");
93  goto PROBLEM;
94  }
95  if (xemuwinmenu.num_of_hmenus >= XEMUGUI_MAX_SUBMENUS)
96  FATAL("GUI: too many submenus!");
97  xemuwinmenu.hmenus[xemuwinmenu.num_of_hmenus++] = menu;
98  int radio_begin = xemuwinmenu.num_of_items;
99  int radio_active = xemuwinmenu.num_of_items; // radio active is a kinda odd name, but never mind ...
100  for (int a = 0; desc[a].name; a++) {
101  if (!desc[a].handler || !desc[a].name) {
102  DEBUGPRINT("GUI: invalid meny entry found, skipping it" NL);
103  continue;
104  }
105  if (xemuwinmenu.num_of_items >= XEMUGUI_MAX_ITEMS)
106  FATAL("GUI: too many items in menu builder!");
107  int ret = 1, type = desc[a].type;
108  switch (type & 0xFF) {
109  case XEMUGUI_MENUID_SUBMENU: {
110  HMENU submenu = _wingui_recursive_menu_builder(desc[a].handler); // that's a prime example for using recursion :)
111  if (!submenu)
112  goto PROBLEM;
113  ret = AppendMenu(menu, MF_POPUP, (UINT_PTR)submenu, desc[a].name);
114  }
115  break;
116  case XEMUGUI_MENUID_CALLABLE:
117  if ((type & XEMUGUI_MENUFLAG_QUERYBACK)) {
118  DEBUGGUI("GUI: query-back for \"%s\"" NL, desc[a].name);
119  ((xemugui_callback_t)(desc[a].handler))(&desc[a], &type);
120  }
121  xemuwinmenu.items[xemuwinmenu.num_of_items] = &desc[a];
122  // Note the +1 for ID. That is because some stange Windows sting:
123  // TrackPopupMenu() with TPM_RETURNCMD returns zero as error/no-selection ...
124  // So we want to being with '1' that's why the PRE-incrementation instead of the POST
125  ret = AppendMenu(menu, MF_STRING, ++xemuwinmenu.num_of_items, desc[a].name);
126  break;
127  default:
128  break;
129  }
130  if (!ret) {
131  ERROR_WINDOW("AppendMenu() failed in menu builder!");
132  goto PROBLEM;
133  }
134  if ((type & XEMUGUI_MENUFLAG_CHECKED))
135  CheckMenuItem(menu, xemuwinmenu.num_of_items, MF_CHECKED);
136  if ((type & XEMUGUI_MENUFLAG_SEPARATOR))
137  AppendMenu(menu, MF_SEPARATOR, 0, NULL);
138  if ((type & XEMUGUI_MENUFLAG_BEGIN_RADIO)) {
139  radio_begin = xemuwinmenu.num_of_items;
140  radio_active = xemuwinmenu.num_of_items;
141  }
142  if ((type & XEMUGUI_MENUFLAG_ACTIVE_RADIO))
143  radio_active = xemuwinmenu.num_of_items;
144  if ((type & XEMUGUI_MENUFLAG_END_RADIO)) {
145  CheckMenuRadioItem(menu, radio_begin, xemuwinmenu.num_of_items, radio_active, MF_BYCOMMAND);
146  radio_begin = xemuwinmenu.num_of_items;
147  radio_active = xemuwinmenu.num_of_items;
148  }
149  }
150  return menu;
151 PROBLEM:
152  xemuwinmenu.problem = 1;
153  return NULL;
154 }

157 static void _wingui_destroy_menu ( void )
158 {
159  while (xemuwinmenu.num_of_hmenus > 0) {

164  xemuwinmenu.num_of_items = 0;

168 static HMENU _wingui_create_popup_menu ( const struct menu_st desc[] )
169 {
170  _wingui_destroy_menu();
171  xemuwinmenu.problem = 0;
172  HMENU menu = _wingui_recursive_menu_builder(desc);
173  if (!menu || xemuwinmenu.problem) {
174  _wingui_destroy_menu();
175  return NULL;
176  } else
177  return menu;

198 // Compared to the GTK version, it seems Windows is unable to do async GUI so it's a blocking implementation :(
199 static int xemuwingui_popup ( const struct menu_st desc[] )
200 {
201  DEBUGGUI("GUI: WIN: popup!" NL);
202  HMENU menu = _wingui_create_popup_menu(desc);
203  if (!menu)
204  return 1;
205  int num_of_items = xemuwinmenu.num_of_items;
206  POINT point;
207  _wingui_getmousecoords(&point);
208  if (!ClientToScreen(xemuwinmenu.win_hwnd, &point)) {
209  // ClientToScreen returns with non-zero if it's OK!!!!
210  _wingui_destroy_menu();
211  ERROR_WINDOW("ClientToScreen returned with zero");
212  return 1;
213  }
214  int n = TrackPopupMenu(menu, TPM_RIGHTBUTTON | TPM_RETURNCMD | TPM_NONOTIFY, point.x, point.y, 0, xemuwinmenu.win_hwnd, NULL); // TPM_RETURNCMD causes to return with the selected ID (ie the 3rd param of AppendMenu)
215  xemu_drop_events();
216  DEBUGGUI("Returned: items=%d RETVAL=%d" NL, num_of_items, n);
217  // again, do not forget that IDs are from one in windows, since zero means non-selection or error!
218  if (n > 0 && n <= num_of_items) {
219  const struct menu_st *item = xemuwinmenu.items[n - 1];
220  _wingui_destroy_menu();
221  _wingui_callback(item);
222  } else
223  _wingui_destroy_menu();
224  xemu_drop_events();
225  return 0;
226 }
