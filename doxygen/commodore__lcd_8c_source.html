172 Uint8 cpu65_read_callback ( Uint16 addr ) {
173  if (addr < 0x1000) return GET_MEMORY(addr);
174  if (addr < 0xF800) return GET_MEMORY((mmu_current[addr >> 14] + addr) & 0x3FFFF);
175  if (addr >= 0xFA00) return GET_MEMORY(addr | 0x30000);
176  if (addr >= 0xF980) return 0; // ACIA
177  if (addr >= 0xF900) return 0xFF; // I/O exp
178  if (addr >= 0xF880) return via_read(&via2, addr & 15);
179  return via_read(&via1, addr & 15);
180 }

182 void cpu65_write_callback ( Uint16 addr, Uint8 data ) {
183  int maddr;
184  if (addr < 0x1000) {
185  memory[addr] = data;
186  return;
187  }
188  if (addr >= 0xF800) {
189  switch ((addr - 0xF800) >> 7) {
190  case 0: via_write(&via1, addr & 15, data); return;
191  case 1: via_write(&via2, addr & 15, data); return;
192  case 2: return; // I/O exp area is not handled
193  case 3: return; // no ACIA yet
194  case 4: mmu_current = mmu[2]; return;
195  case 5: mmu_current = mmu[1]; return;
196  case 6: mmu_current = mmu[0]; return;
197  case 7: mmu_current = mmu_saved; return;
198  case 8: mmu_saved = mmu_current; return;
199  case 9: FATAL("MMU test mode is set, it would not work"); break;
200  case 10: mmu[1][0] = data << 10; return;
201  case 11: mmu[1][1] = data << 10; return;
202  case 12: mmu[1][2] = data << 10; return;
203  case 13: mmu[1][3] = data << 10; return;
204  case 14: mmu[2][1] = data << 10; return;
205  case 15: lcd_ctrl[addr & 3] = data; return;
206  }
207  DEBUG("ERROR: should be not here!" NL);
208  return;
209  }
210  maddr = (mmu_current[addr >> 14] + addr) & 0x3FFFF;
211  if (maddr < ram_size) {
212  memory[maddr] = data;
213  return;
214  }
215  DEBUG("MEM: out-of-RAM write addr=$%04X maddr=$%05X" NL, addr, maddr);
216 }

290 static void render_screen ( void )
291 {
292  int ps = lcd_ctrl[1] << 7;
293  int x, y, ch;
294  int tail;
295  Uint32 *pix = xemu_start_pixel_buffer_access(&tail);
296  if (lcd_ctrl[2] & 2) { // graphic mode
297  for (y = 0; y < 128; y++) {
298  for (x = 0; x < 60; x++) {
299  ch = memory[ps++];
300  *(pix++) = (ch & 0x80) ? FG : BG;
301  *(pix++) = (ch & 0x40) ? FG : BG;
302  *(pix++) = (ch & 0x20) ? FG : BG;
303  *(pix++) = (ch & 0x10) ? FG : BG;
304  *(pix++) = (ch & 0x08) ? FG : BG;
305  *(pix++) = (ch & 0x04) ? FG : BG;
306  *(pix++) = (ch & 0x02) ? FG : BG;
307  *(pix++) = (ch & 0x01) ? FG : BG;
308  }
309  ps = (ps + 4) & 0x7FFF;
310  pix += tail;
311  }
312  } else { // text mode
313  int cof = (lcd_ctrl[2] & 1) << 10;
314  int maxx = (lcd_ctrl[3] & 4) ? 60 : 80;
315  ps += lcd_ctrl[0] & 0x7F; // X-Scroll register, only the lower 7 bits are used
316  for (y = 0; y < 128; y++) {
317  for (x = 0; x < maxx; x++) {
318  ps &= 0x7FFF;
319  ch = memory[ps++];
320  ch = charrom[cof + ((ch & 0x7F) << 3) + (y & 7)] ^ ((ch & 0x80) ? 0xFF : 0x00);
321  pix[0] = (ch & 0x80) ? FG : BG;
322  pix[1] = (ch & 0x40) ? FG : BG;
323  pix[2] = (ch & 0x20) ? FG : BG;
324  pix[3] = (ch & 0x10) ? FG : BG;
325  pix[4] = (ch & 0x08) ? FG : BG;
326  pix[5] = (ch & 0x04) ? FG : BG;
327  if (lcd_ctrl[3] & 4) {
328  pix[6] = (ch & 0x02) ? FG : BG;
329  pix[7] = (ch & 0x01) ? FG : BG;
330  pix += 8;
331  } else
332  pix += 6;
333  }
334  if ((y & 7) == 7)
335  ps += 128 - maxx;
336  else
337  ps -= maxx;
338  pix += tail;
339  }
340  }
341  xemu_update_screen();
342 }

354 static void update_rtc ( void )
355 {
356  struct tm *t = xemu_get_localtime();
357  rtc_regs[ 0] = t->tm_sec % 10;
358  rtc_regs[ 1] = t->tm_sec / 10;
359  rtc_regs[ 2] = t->tm_min % 10;
360  rtc_regs[ 3] = t->tm_min / 10;
361  rtc_regs[ 4] = t->tm_hour % 10;
362  rtc_regs[ 5] = (t->tm_hour / 10) | 8; // TODO: AM/PM, 24h/12h time format
363  rtc_regs[ 6] = t->tm_wday; // day of week
364  rtc_regs[ 9] = t->tm_mday % 10;
365  rtc_regs[10] = t->tm_mday / 10;
366  rtc_regs[ 7] = (t->tm_mon + 1) % 10;
367  rtc_regs[ 8] = (t->tm_mon + 1) / 10;
368  rtc_regs[11] = (t->tm_year - 84) % 10; // beware of 2084, Commodore LCDs will have "Y2K-like" problem ... :)
369  rtc_regs[12] = (t->tm_year - 84) / 10;
370 }

374 static void update_emulator ( void )
375 {
376  if (XEMU_UNLIKELY(prg_inject.phase == 1)) {
377  static const Uint8 screen_sample1[] = { 0x20, 0x03, 0x0f, 0x0d, 0x0d, 0x0f, 0x04, 0x0f, 0x12, 0x05, 0x20, 0x0c, 0x03, 0x04, 0x20 };
378  static const Uint8 screen_sample2[] = { 0x12, 0x05, 0x01, 0x04, 0x19, 0x2e };
379  if (
380  !memcmp(memory + 0x880, screen_sample1, sizeof screen_sample1) &&
381  !memcmp(memory + 0x980, screen_sample2, sizeof screen_sample2)
382  ) {
383  prg_inject.phase = 2;
384  DEBUGPRINT("BASIC: startup screen detected, injecting loaded basic program!" NL);
385  memory[0xA01] = 'R' - 'A' + 1;
386  memory[0xA02] = 'U' - 'A' + 1;
387  memory[0xA03] = 'N' - 'A' + 1;
388  memory[0x1000] = 0;
389  memset(memory + 0x1000, 0, 0x8000);
390  memcpy(memory + 0x1001, prg_inject.data + 2, prg_inject.size - 2);
391  //memset(memory + 0x1001 + prg_inject.size - 2, 0, 4);
392  int addr = 0x1001;
393  memory[0x65] = addr & 0xFF;
394  memory[0x66] = addr >> 8;
395  addr += prg_inject.size - 2;
396  memory[0x67] = addr & 0xFF;
397  memory[0x68] = addr >> 8;
398  }
399  }
400  render_screen();
401  hid_handle_all_sdl_events();
402  xemu_timekeeping_delay(40000); // 40000 microseconds would be the real time for a full TV frame (see main() for more info: CLCD is not TV based for real ...)
403  if (seconds_timer_trigger)
404  update_rtc();
405 }

452 static void load_program_for_inject ( const char *file_name, int new_address )
453 {
454  prg_inject.phase = 0;
455  if (!file_name || !*file_name)
456  return;
457  memset(prg_inject.data, 0, sizeof prg_inject.data);
458  prg_inject.size = xemu_load_file(file_name, prg_inject.data, 8, sizeof(prg_inject.data) - 4, "Cannot load program");
459  if (prg_inject.size <= 0)
460  return;
461  int old_address = prg_inject.data[0] | (prg_inject.data[1] << 8);
462  DEBUGPRINT("PRG: program \"%s\" load_addr=$%04X, new_load_addr=$%04X" NL, file_name, old_address, new_address);
463  if (old_address != new_address) {
464  int i = 2;
465  for (;;) {
466  if (prg_inject.data[i] == 0 && prg_inject.data[i + 1] == 0)
467  break;
468  int o = i; // offset of the next addr to patch
469  i += 4; // skip next line offset + line number
470  while (prg_inject.data[i])
471  i++;
472  i++;
473  new_address += i - o;
474  DEBUGPRINT("BASIC: re-linking line (%d) $%04X -> $%04X" NL,
475  prg_inject.data[o + 2] | (prg_inject.data[o + 3] << 8),
476  prg_inject.data[o] | (prg_inject.data[o + 1] << 8),
477  new_address
478  );
479  prg_inject.data[o] = new_address & 0xFF;
480  prg_inject.data[o + 1] = new_address >> 8;
481  }
482 
483  }
484  prg_inject.phase = 1;
485 }

506 static void emulation_loop ( void )
507 {
508  for (;;) {
509  int opcyc = cpu65_step(); // execute one opcode (or accept IRQ, etc), return value is the used clock cycles
510  viacyc += opcyc;
511  cycles += opcyc;
512  if (viacyc >= cpu_mhz) {
513  int steps = viacyc / cpu_mhz;
514  viacyc = viacyc % cpu_mhz;
515  via_tick(&via1, steps); // run VIA-1 tasks for the same amount of cycles as the CPU would do @ 1MHz
516  via_tick(&via2, steps); // -- "" -- the same for VIA-2
517  }
518  /* Note, Commodore LCD is not TV standard based ... Since I have no idea about the update etc, I still assume some kind of TV-related stuff, who cares :) */
519  if (XEMU_UNLIKELY(cycles >= cpu_cycles_per_tv_frame)) { // if enough cycles elapsed (what would be the amount of CPU cycles for a TV frame), let's call the update function.
520  update_emulator(); // this is the heart of screen update, also to handle SDL events (like key presses ...)
521  cycles -= cpu_cycles_per_tv_frame; // not just cycle = 0, to avoid rounding errors, but it would not matter too much anyway ...
522  return;
523  }
524  }
525 }
