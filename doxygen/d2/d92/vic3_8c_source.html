112 static void vic3_interrupt_checker ( void )
113 {
114  int vic_irq_old = cpu65.irqLevel & 2;
115  int vic_irq_new;
116  if ((interrupt_status & vic3_registers[0x1A])) {
117  interrupt_status |= 128;
118  vic_irq_new = 2;
119  } else {
120  interrupt_status &= 127;
121  vic_irq_new = 0;
122  }
123  if (vic_irq_old != vic_irq_new) {
124  DEBUG("VIC3: interrupt change %s -> %s" NL, vic_irq_old ? "active" : "inactive", vic_irq_new ? "active" : "inactive");
125  if (vic_irq_new)
126  cpu65.irqLevel |= 2;
127  else
128  cpu65.irqLevel &= ~2;
129  }
130 }

164 // FIXME: hardware attributes are supported now for only 40/80-col text mode!
165 // FIXME: for real, it's supported for everything, _even_ for bitplane mode .......... (surprisingly ...)
166 #define VIC3_ADJUST_BY_HARDWARE_ATTRIBUTES( has_attrib_condition, colour_var, vdata_var ) do { \
167  if (has_attrib_condition) { \
168  if ((colour_var & 0xF0) == 0x10) { /* only the blink bit for the character is set */ \
169  if (blink_phase) \
170  vdata_var = 0; /* blinking character, in one phase, the character "disappears", ie blinking */ \
171  colour_var &= 15; \
172  } else if ((!(colour_var & 0x10)) || blink_phase) { \
173  if ((colour_var & 0x80) && (row_counter == 7)) /* underline (must be before reverse, as underline can be reversed as well!) */ \
174  vdata_var = 0xFF; /* the underline */ \
175  if (colour_var & 0x20) /* reverse bit for char */ \
176  vdata_var ^= 0xFF; \
177  if (colour_var & 0x40) /* highlight, this must be the LAST, since it sets the low nibble of coldata ... */ \
178  colour_var = 0x10 | (colour_var & 15); \
179  else \
180  colour_var &= 15; \
181  } else \
182  colour_var &= 15; \
183  } else \
184  colour_var &= 15; \
185 } while (0)

189 static void renderer_text_40 ( void )
190 {
191  Uint8 *vp = vicptr_video_40 + video_counter;
192  Uint8 *cp = COLMEMPTR + video_counter;
193  Uint8 *chargen = vicptr_chargen + row_counter;
194  Uint32 bg_colour = VIC_REG_COLOUR(0x21);
195  int a;
196  STATIC_COLOUR_RENDERER(VIC_REG_COLOUR(0x20), LEFT_BORDER_SIZE);
197  for (a = 0; a < 40; a++) {
198  Uint8 vdata = chargen[(*(vp++)) << 3];
199  Uint32 colour = *(cp++);
200  Uint32 fg_colour;
201  VIC3_ADJUST_BY_HARDWARE_ATTRIBUTES(XEMU_UNLIKELY(colour & attributes), colour, vdata);
202  fg_colour = palette[colour];
203  pixel[ 0] = pixel[ 1] = vdata & 0x80 ? fg_colour : bg_colour;
204  pixel[ 2] = pixel[ 3] = vdata & 0x40 ? fg_colour : bg_colour;
205  pixel[ 4] = pixel[ 5] = vdata & 0x20 ? fg_colour : bg_colour;
206  pixel[ 6] = pixel[ 7] = vdata & 0x10 ? fg_colour : bg_colour;
207  pixel[ 8] = pixel[ 9] = vdata & 0x08 ? fg_colour : bg_colour;
208  pixel[10] = pixel[11] = vdata & 0x04 ? fg_colour : bg_colour;
209  pixel[12] = pixel[13] = vdata & 0x02 ? fg_colour : bg_colour;
210  pixel[14] = pixel[15] = vdata & 0x01 ? fg_colour : bg_colour;
211  pixel += 16;
212  }
213  STATIC_COLOUR_RENDERER(VIC_REG_COLOUR(0x20), RIGHT_BORDER_SIZE);
214 }

308 static void renderer_mcmtext_40 ( void )
309 {
310  Uint8 *vp = vicptr_video_40 + video_counter;
311  Uint8 *cp = COLMEMPTR + video_counter;
312  Uint8 *chargen = vicptr_chargen + row_counter;

314  VIC_REG_COLOUR(0x21),

613 // write register functions should call this on any register change which may affect display mode with settings already stored in vic3_registers[]!
614 static void select_renderer_func ( void )
615 {
616  // Ugly, but I really don't have idea what happens in VIC-III internally when you
617  // switch H640 bit within a frame ... Currently I just limit video counter to be
618  // within 1K if H640 is not set. That's certainly incorrect, but it should be tested
619  // on a real C65 what happens (eg in text mode) if H640 bit is flipped at the middle
620  // of a screen.
621  // Also FIXME: where are the sprites in bitplane mode?!
622  if (!IS_H640) {
623  video_counter &= 1023;
624  video_counter_inc = 40;
625  sprite_pointers = vicptr_video_40 + 0x3F8;
626  } else {
627  video_counter_inc = 80;
628  sprite_pointers = vicptr_video_80 + 0x7F8;
629  }
630  // if DEN (Display Enable) bit is zero, then you won't see anything (border colour is rendered!)
631  if (!(vic3_registers[0x11] & 0x10)) {
632  renderer_func = renderer_disabled_screen;
633  return;
634  }
635  // if BPM bit is set, it's bitplane mode (VIC-III)
636  if (vic3_registers[0x31] & 16) {
637  renderer_func = IS_H640 ? renderer_bitplane_640 : renderer_bitplane_320;
638  return;
639  }
640  // Otherwise, classic VIC-II modes, select one, from the possible 8 modes (note: some of them are INVALID!)
641  // the pattern: ECM - BMM - MCM
642  switch (((vic3_registers[0x11] & 0x60) | (vic3_registers[0x16] & 0x10)) >> 4) {
643  case 0: // Standard text mode (ECM/BMM/MCM=0/0/0)
644  renderer_func = IS_H640 ? renderer_text_80 : renderer_text_40;
645  break;
646  case 1: // Multicolor text mode (ECM/BMM/MCM=0/0/1)
647  renderer_func = IS_H640 ? renderer_mcmtext_80 : renderer_mcmtext_40;
648  break;
649  case 2: // Standard bitmap mode (ECM/BMM/MCM=0/1/0)
650  renderer_func = IS_H640 ? renderer_bitmap_640 : renderer_bitmap_320;
651  break;
652  case 3: // Multicolor bitmap mode (ECM/BMM/MCM=0/1/1)
653  renderer_func = IS_H640 ? renderer_mcmbitmap_640 : renderer_mcmbitmap_320;
654  break;
655  case 4: // ECM text mode (ECM/BMM/MCM=1/0/0)
656  renderer_func = IS_H640 ? renderer_ecmtext_80 : renderer_ecmtext_40;
657  break;
658  case 5: // Invalid text mode (ECM/BMM/MCM=1/0/1)
659  case 6: // Invalid bitmap mode 1 (ECM/BMM/MCM=1/1/0)
660  case 7: // Invalid bitmap mode 2 (ECM/BMM/MCM=1/1/1)
661  // Invalid modes generate only "blackness"
662  renderer_func = renderer_invalid_mode;
663  break;
664  }
665 }

726 // Caller should give only 0-$3F or 0-$7F addresses
727 void vic3_write_reg ( int addr, Uint8 data )
728 {
729  DEBUG("VIC3: write reg $%02X with data $%02X" NL, addr, data);
730  if (addr == 0x2F) {
731  if (!vic_new_mode && data == 0x96 && vic3_registers[0x2F] == 0xA5) {
732  vic_new_mode = VIC_NEW_MODE;
733  DEBUG("VIC3: switched into NEW I/O access mode :)" NL);
734  } else if (vic_new_mode) {
735  vic_new_mode = 0;
736  DEBUG("VIC3: switched into OLD I/O access mode :(" NL);
737  }
738  }
739  if (!vic_new_mode && addr > 0x2F) {
740  DEBUG("VIC3: ignoring writing register $%02X (with data $%02X) because of old I/O access mode selected" NL, addr, data);
741  return;
742  }
743  switch (addr) {
744  case 0x11:
745  vic3_registers[0x11] = data;
746  select_renderer_func();
747  compare_raster = (compare_raster & 0xFF) | ((data & 0x80) << 1);
748  DEBUG("VIC3: compare raster is now %d" NL, compare_raster);
749  break;
750  case 0x12:
751  compare_raster = (compare_raster & 0xFF00) | data;
752  DEBUG("VIC3: compare raster is now %d" NL, compare_raster);
753  break;
754  case 0x16:
755  vic3_registers[0x16] = data;
756  select_renderer_func();
757  break;
758  case 0x18:
759  vic3_registers[0x18] = data;
760  select_vic2_memory();
761  break;
762  case 0x19:
763  interrupt_status = interrupt_status & (~data) & 15;
764  vic3_interrupt_checker();
765  break;
766  case 0x1A:
767  data &= 15;
768  break;
769  case 0x30:
770  // Save some un-needed memory translating table rebuilds, if there is important bits (for us) changed.
771  // CRAM@DC00 is not handled by the translator directly, so bit0 does not apply here!
772  if (
773  (data & 0xB8) != (vic3_registers[0x30] & 0xB8)
774  ) {
775  DEBUG("MEM: applying new memory configuration because of VIC3 $30 is written" NL);
776  vic3_registers[0x30] = data; // early write because of apply_memory_config() needs it
777  apply_memory_config();
778  } else {
779  vic3_registers[0x30] = data; // we need this for select_chargen() below
780  DEBUG("MEM: no need for new memory configuration (because of VIC3 $30 is written): same ROM bit values are set" NL);
781  }
782  select_chargen();
783  palette = (data & 4) ? vic3_palette : vic3_rom_palette;
784  break;
785  case 0x31:
786  vic3_registers[0x31] = data;
787  attributes = (data & 32) ? 0xF0 : 0x00; // currently, I use $F0 for on, 00 off, to be able to use as an attrib mask directly for upper 4 bits
788  select_renderer_func();
789  if ((data & 64)) {
790  cpu_cycles_per_scanline = FAST_CPU_CYCLES_PER_SCANLINE;
791  cpu65_set_ce_timing(1);
792  strcpy(emulator_speed_title, "3.5MHz");
793  } else {
794  cpu_cycles_per_scanline = SLOW_CPU_CYCLES_PER_SCANLINE;
795  cpu65_set_ce_timing(0);
796  strcpy(emulator_speed_title, "1MHz");
797  }
798  if ((data & 15) && warn_ctrl_b_lo) {
799  INFO_WINDOW("VIC3 control-B register V400, H1280, MONO and INT features are not emulated yet!\nThere will be no further warnings on this issue.");
800  warn_ctrl_b_lo = 0;
801  }
802  break;
803  case 0x33:
804  case 0x34:
805  case 0x35:
806  case 0x36:
807  case 0x37:
808  case 0x38:
809  case 0x39:
810  case 0x3A:
811  bitplane_addr_320[addr - 0x33] = ((data & 14) << 12) + (addr & 1 ? 0 : 0x10000);
812  bitplane_addr_640[addr - 0x33] = ((data & 12) << 12) + (addr & 1 ? 0 : 0x10000);
813  break;
814  default:
815  if (addr >= 0x20 && addr < 0x2F)
816  data &= 15;
817  break;
818  }
819  vic3_registers[addr] = data;
820 }
