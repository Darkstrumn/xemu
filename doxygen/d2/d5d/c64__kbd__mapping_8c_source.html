153 Uint8 c64_get_joy_state ( void )
154 {
155  return
156  0xE0 |
157  (hid_read_joystick_up ( 0, 1 ) & (is_mouse_grab() ? hid_read_mouse_button_right(0, 1) : 1)) |
158  hid_read_joystick_down ( 0, 2 ) |
159  hid_read_joystick_left ( 0, 4 ) |
160  hid_read_joystick_right ( 0, 8 ) |
161  (hid_read_joystick_button( 0, 16 ) & (is_mouse_grab() ? hid_read_mouse_button_left (0, 16) : 16))
162  ;
163 }
