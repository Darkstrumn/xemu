59 static void write_descriptor ( int dsc_now, Uint8 data )
60 {
61  switch (dsc_now & 3) {
62  case 0:
63  segment_bases[sar] = (data << 8) | (segment_bases[sar] & 0xFF);
64  DEBUGPRINT("Z8010: segment #$%02X base high is set to $%02X, value is now = %04X" NL, sar, data, segment_bases[sar]);
65  break;
66  case 1:
67  segment_bases[sar] = (segment_bases[sar] & 0xFF00) | data;
68  DEBUGPRINT("Z8010: segment #$%02X base low is set to $%02X, value is now = %04X" NL, sar, data, segment_bases[sar]);
69  break;
70  case 2:
71  segment_limits[sar] = data;
72  DEBUGPRINT("Z8010: segment #$%02X limit is set to $%02X" NL, sar, data);
73  break;
74  case 3:
75  segment_attributes[sar] = data;
76  DEBUGPRINT("Z8010: segment #$%02X attribute is set to $%02X" NL, sar, data);
77  break;
78  }

97 // Z8010 has connection for the AD8-AD15 lines only with the CPU. That is,
98 // even writing MMU registers, only the high byte counts on the data and
99 // the MMU register selection as well! However, to simplify things, we assume

103 void z8010_write_register ( int port, Uint8 data )
104 {
105  switch (port) {
106  case 0x00:
107  mode = data;
108  break;
109  case 0x01:
110  sar = data & 63;
111  dsc = 0; // FIXME: do we need this?
112  break;
113  case 0x08: // base field of descriptor
114  write_descriptor(dsc & 1, data);
115  dsc++;
116  break;
117  case 0x09: // limit
118  write_descriptor(2, data);
119  break;
120  case 0x0A: // attribute field
121  write_descriptor(3, data);
122  break;
123  case 0x0B: // descriptor, ALL fields!!!!
124  write_descriptor(dsc, data);
125  dsc++;
126  case 0x0C: // base field, incrementing SAR
127  write_descriptor(dsc & 1, data);
128  sar = (sar + 1) & 63;
129  break;
130  case 0x0D: // limit field, incrementing SAR
131  write_descriptor(2, data);
132  sar = (sar + 1) & 63;
133  break;
134  case 0x0E: // attribute field, incrementing SAR
135  write_descriptor(3, data);
136  sar = (sar + 1) & 63;
137  break;
138  case 0x0F:
139  write_descriptor(dsc, data);
140  dsc = (dsc + 1) & 3;
141  if (!dsc)
142  sar = (sar + 1) & 63;
143  break;
144 
145 
146  case 0x20: // descriptor counter
147  dsc = data & 3;
148  DEBUGPRINT("Z8010: DSC is set to %d" NL, dsc);
149  break;

151  }
