45 static int snapcallback_memory_loader ( const struct xemu_snapshot_definition_st *def, struct xemu_snapshot_block_st *block )
46 {
47  const struct memblock_st *mem = (const struct memblock_st *)def->user_data;
48  if (block->block_version != M65_MEMORY_BLOCK_VERSION || block->sub_counter != 0 || block->sub_size > mem->size)
49  RETURN_XSNAPERR_USER("Bad memory block syntax @ %s", def->idstr);
50  memset(mem->data, 0xFF, mem->size);
51  return xemusnap_read_file(mem->data, block->sub_size); // read that damn memory dump
52 }

55 static int snapcallback_memory_saver ( const struct xemu_snapshot_definition_st *def )
56 {
57  const struct memblock_st *mem = (const struct memblock_st *)def->user_data;
58  int ret = xemusnap_write_block_header(def->idstr, M65_MEMORY_BLOCK_VERSION);
59  if (ret) return ret;
60  ret = mem->size - 1;
61  while (ret && mem->data[ret] == 0xFF)
62  ret--;
63  return xemusnap_write_sub_block(mem->data, ret + 1);
64 }
