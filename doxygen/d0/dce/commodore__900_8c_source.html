41 Uint16 z8k1_read_word_cb ( int seg, Uint16 ofs )
42 {
43  ofs &= 0xFFFE;
44  if (seg)
45  FATAL("Currently only segment-0 can be used, given = seg#$%02X", seg);
46  if (ofs >= 0x8000)

65 int main ( int argc, char **argv )
66 {
67  xemu_pre_init(APP_ORG, TARGET_NAME, "The new-territory-for-me Commodore 900 emulator from LGB");
68  xemucfg_define_str_option("bootrom", BOOT_ROM, "Set BOOT ROM to be loaded");
69  if (xemucfg_parse_all(argc, argv))
70  return 1;
71  if (xemu_load_file(xemucfg_get_str("bootrom"), memory, 0x8000, 0x8000, "The boot ROM of Commodore 900 (combined HI+LO)") != 0x8000)
72  return 1;
73  z8k1_init();
74  puts("\nCURRENT GOAL: only disassembly the BOOT ROM and compare it with well known disasm result done by others on boot ROMs\n");
75  z8k1_reset();
76  z8010_init();
77  z8010_reset();
78  for (;;) {
79  if (z8k1_step(0) < 0)
80  break;
81  }
82  return 0;
83 }
