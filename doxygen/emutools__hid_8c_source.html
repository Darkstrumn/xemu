58 int hid_key_event ( SDL_Scancode key, int pressed )
59 {
60  const struct KeyMappingUsed *map = key_map;
61  //OSD(-1, -1, "Key %s <%s>", pressed ? "press " : "release", SDL_GetScancodeName(key));
62  while (map->pos >= 0) {
63  if (map->scan == key) {
64  if (map->pos > 0xFF) { // special emulator key!
65  switch (map->pos) { // handle "built-in" events, if emulator target uses them at all ...
66  case XEMU_EVENT_EXIT:
67  if (ARE_YOU_SURE(str_are_you_sure_to_exit, i_am_sure_override | ARE_YOU_SURE_DEFAULT_YES))
68  exit(0);
69  break;
70  case XEMU_EVENT_FAKE_JOY_UP:
71  if (pressed) hid_state |= JOYSTATE_UP; else hid_state &= ~JOYSTATE_UP;
72  break;
73  case XEMU_EVENT_FAKE_JOY_DOWN:
74  if (pressed) hid_state |= JOYSTATE_DOWN; else hid_state &= ~JOYSTATE_DOWN;
75  break;
76  case XEMU_EVENT_FAKE_JOY_LEFT:
77  if (pressed) hid_state |= JOYSTATE_LEFT; else hid_state &= ~JOYSTATE_LEFT;
78  break;
79  case XEMU_EVENT_FAKE_JOY_RIGHT:
80  if (pressed) hid_state |= JOYSTATE_RIGHT; else hid_state &= ~JOYSTATE_RIGHT;
81  break;
82  case XEMU_EVENT_FAKE_JOY_FIRE:
83  if (pressed) hid_state |= JOYSTATE_BUTTON; else hid_state &= ~JOYSTATE_BUTTON;
84  break;
85  case XEMU_EVENT_TOGGLE_FULLSCREEN:
86  if (pressed)
87  xemu_set_full_screen(-1);
88  break;
89  default:
90  return emu_callback_key(map->pos, key, pressed, 0);
91  }
92  return emu_callback_key(map->pos, key, pressed, 1);
93  }
94  if (map->pos & 8) // shifted key emu?
95  KBD_SET_KEY(virtual_shift_pos, pressed); // maintain the shift key
96  KBD_SET_KEY(map->pos, pressed);
97  return emu_callback_key(map->pos, key, pressed, 1);
98  }
99  map++;
100  }
101  return emu_callback_key(-1, key, pressed, 0);
102 }

130 const char *hid_keymap_add_mapping ( const char *emu_key_name, const char *host_key_name )
131 {
132  SDL_Scancode scan;
133  if (strcmp(host_key_name, scan_name_unknown)) {
134  scan = SDL_GetScancodeFromName(host_key_name);
135  if (scan == SDL_SCANCODE_UNKNOWN)
136  return HID_ERR_STR_UNKNOWN_HOST_KEYNAME;
137  } else
138  scan = SDL_SCANCODE_UNKNOWN;
139  // Look up matrix position from the default mapping given to hid_init() which also carries the "emu key names"
140  int pos;
141  int source;
142  for (source = 0 ;; source++) {
143  if (key_map_default[source].pos < 0)
144  return HID_ERR_STR_UNKNOWN_EMU_KEYNAME;
145  if (key_map_default[source].name && (!strcmp(key_map_default[source].name, emu_key_name))) {
146  //DEBUGPRINT("FOUND KEY: %s for %s" NL, key_map_default[source].name, emu_key_name);
147  pos = key_map_default[source].pos;
148  break;
149  }
150  }
151  // At this point we have the SDL scancode, and the "matrix position" ...
152  // We change all mappings' scan code to the desired ones (maybe more PC keys is mapped to a single emulated key!)
153  for (int a = 0; key_map[a].pos >= 0 ; a++) {
154  if (key_map[a].pos == pos && key_map[a].set != source) {
155  if (key_map[a].scan != scan && key_map[a].set != -2) {
156  DEBUGPRINT("HID: altering keyboard mapping for \"%s\" from \"%s\" to \"%s\" at array index #%d" NL,
157  emu_key_name, SDL_GetScancodeName(key_map[a].scan), host_key_name, a
158  );
159  }
160  key_map[a].scan = scan;
161  key_map[a].set = source;
162  break;
163  }
164  }
165  return NULL;
166 }

169 const char *hid_keymap_add_mapping_from_config_line ( const char *p, int *num_of_items )
170 {
171  char emu_name[64], host_name[64];
172  while (*p <= 0x20 && *p)
173  p++;
174  if (*p == '\0' || *p == '#')
175  goto skip_rest;
176  const char *emu_p = p;
177  while (*p > 32)
178  p++;
179  if (*p == '\0' || *p == '#' || p - emu_p >= sizeof(emu_name))
180  goto skip_rest;
181  memcpy(emu_name, emu_p, p - emu_p);
182  emu_name[p - emu_p] = 0;
183  if (!strcmp(emu_name, CLEARALLMAPPING)) {
184  for (int a = 0; key_map[a].pos >= 0; a++) {
185  key_map[a].set = -2;
186  key_map[a].scan = 0;
187  }
188  goto skip_rest;
189  }
190  while (*p <= 32 && *p)
191  p++;
192  if (*p == 0 || *p == '#')
193  goto skip_rest;
194  const char *host_p = p;
195  while (*p && *p != 13 && *p != 10 && *p != '#')
196  p++;
197  //p--;
198  while (p[-1] <= 0x20 && p > host_p)
199  p--;
200  if (p - host_p >= sizeof(host_name))
201  goto skip_rest;
202  memcpy(host_name, host_p, p - host_p);
203  host_name[p - host_p] = 0;
204  const char *res = hid_keymap_add_mapping(emu_name, host_name);
205  if (res)
206  DEBUGPRINT("HID: error on keymap user config: emu_name=[%s] host_key=<%s>: %s\n", emu_name, host_name, res);
207  else
208  (*num_of_items)++;
209 skip_rest:
210  while (*p != 10 && *p != 13 && *p)
211  p++;
212  return p;
213 }

240 void hid_init ( const struct KeyMappingDefault *key_map_in, Uint8 virtual_shift_pos_in, int joy_enable )
241 {
242  int a;
243 #ifdef HID_KBD_MAP_CFG_SUPPORT
244  char kbdcfg[8192];
245  char *kp = kbdcfg + sprintf(kbdcfg,
246  "# default settings for keyboard mapping" NL
247  "# copy this file to filename '%s' in the same directory, and customize (this file is OVERWRITTEN every time, you must copy and customize that one!)" NL
248  "# you can also use the -keymap option of the emulator to specify a keymap file to load (if the specific Xemu emulator supports, use -h to get help)" NL
249  "# Syntax is: EMU-KEY-NAME PC-KEY-NAME" NL
250  "# one assignment per line (EMU-KEY-NAME is always uppercase and one word, while PC-KEY-NAME is case/space/etc sensitive, must be put as is!)" NL
251  "# EMU-KEY-NAME ends in '*' means that it's a virtual key, it is emulated by emulating pressed shift key at the same time" NL
252  "# special line " CLEARALLMAPPING " can be put to clear all existing mappings, it make sense only as the first statement" NL
253  "# without " CLEARALLMAPPING ", only maps are modified which are part of the keymap config file, the rest is left at their default state" NL
254  "# PC-KEY-NAME Unknown means that the certain feature for the emulated keyboard is not mapped to a PC key" NL
255  "# EMU-KEY-NAME strings starting with XEMU- are special Xemu related 'hot keys'" NL
256  NL
257  ,
258  KEYMAP_USER_FILENAME + 1
259  );
260 #endif
261  for (a = 0;;) {
262  if (a >= 0x100)
263  FATAL("Too long default keymapping table for hid_init()");
264  key_map[a].pos = key_map_in[a].pos;
265  key_map[a].scan = key_map_in[a].scan;
266  key_map[a].set = 0;
267  if (key_map[a].pos < 0) {
268 #ifdef HID_KBD_MAP_CFG_SUPPORT
269  int fd = xemu_open_file(KEYMAP_DEFAULT_FILENAME, O_WRONLY|O_TRUNC|O_CREAT, NULL, NULL);
270  if (fd >= 0) {
271  xemu_safe_write(fd, kbdcfg, kp - kbdcfg);
272  close(fd);
273  }
274 #endif
275  DEBUGPRINT("HID: %d key bindings has been added as the default built-in configuration" NL, a);
276  break;
277  }
278 #ifdef HID_KBD_MAP_CFG_SUPPORT
279  register const char *scan_name = (key_map_in[a].scan == SDL_SCANCODE_UNKNOWN ? scan_name_unknown : SDL_GetScancodeName(key_map_in[a].scan));
280  if (scan_name && *scan_name && key_map_in[a].name && key_map_in[a].name[0])
281  kp += sprintf(kp, "%s %s" NL, key_map_in[a].name, scan_name);
282  else
283  DEBUGPRINT("HID: skipping keyboard entry on writing default map: scan <%s> name <%s>" NL, scan_name ? scan_name : "NULL", key_map_in[a].name ? key_map_in[a].name : "NULL");
284 #endif
285  a++;
286  }
287  key_map_default = key_map_in;
288  virtual_shift_pos = virtual_shift_pos_in;
289  SDL_GameControllerEventState(SDL_DISABLE);
290  SDL_JoystickEventState(joy_enable);
291  hid_reset_events(0);
292  for (int a = 0; a < MAX_JOYSTICKS; a++)
293  joysticks[a] = NULL;
294 }

322 void hid_joystick_device_event ( int which , int is_attach )
323 {
324  if (which >= MAX_JOYSTICKS)
325  return;
326  if (is_attach) {
327  if (joysticks[which])
328  hid_joystick_device_event(which, 0);
329  joysticks[which] = SDL_JoystickOpen(which);
330  if (joysticks[which]) {
331  INFO_WINDOW("HID: joystick device #%d \"%s\" has been added." NL, which, SDL_JoystickName(joysticks[which]));
332  } else
333  DEBUG("HID: joystick device #%d problem, cannot be opened on 'add' event: %s." NL, which, SDL_GetError());
334  } else {
335  if (joysticks[which]) {
336  SDL_JoystickClose(joysticks[which]);
337  joysticks[which] = NULL;
338  DEBUG("HID: joystick device #%d has been removed." NL, which);
339  // This is needed to avoid "stuck" joystick state if removed in that state ...
340  hid_state &= ~(JOYSTATE_UP | JOYSTATE_DOWN | JOYSTATE_LEFT | JOYSTATE_RIGHT | JOYSTATE_BUTTON);
341  }
342  }

453 int hid_handle_one_sdl_event ( SDL_Event *event )
454 {
455  int handled = 1;
456  switch (event->type) {
457 #ifdef CONFIG_DROPFILE_CALLBACK
458  case SDL_DROPFILE:
459  if (event->drop.file && event->drop.file[0]) {
460  emu_dropfile_callback(event->drop.file);
461  SDL_free(event->drop.file);
462  }
463  break;
464 #endif
465  case SDL_QUIT:
466  if (ARE_YOU_SURE(str_are_you_sure_to_exit, i_am_sure_override | ARE_YOU_SURE_DEFAULT_YES)) {
467 #ifdef CONFIG_QUIT_CALLBACK
468  emu_quit_callback();
469 #endif
470  exit(0);
471  }
472  break;
473  case SDL_KEYUP:
474  case SDL_KEYDOWN:
475  if (
476 #ifndef CONFIG_KBD_ALSO_REPEATS
477  event->key.repeat == 0 &&
478 #endif
479  event->key.keysym.scancode != SDL_SCANCODE_UNKNOWN
480 #ifdef CONFIG_KBD_SELECT_FOCUS
481  && (event->key.windowID == sdl_winid || event->key.windowID == 0)
482 #endif
483 #ifdef CONFIG_KBD_AVOID_LALTTAB
484  /* ALT-TAB is usually used by the OS, and it can be ignored to "leak" this event into Xemu
485  * so if it's requested, we filter out. NOTE: it's only for LALT (left ALT), right ALT maybe
486  * used for other emulation purposes ... */
487  && !(event->key.keysym.scancode == SDL_SCANCODE_TAB && (event->key.keysym.mod & KMOD_LALT))
488 #endif
489  ) {
490 #ifdef CONFIG_KBD_ALSO_RAW_SDL_CALLBACK
491  emu_callback_key_raw_sdl(&event->key);
492 #endif
493  hid_key_event(event->key.keysym.scancode, event->key.state == SDL_PRESSED);
494  }
495  break;
496  case SDL_JOYDEVICEADDED:
497  case SDL_JOYDEVICEREMOVED:
498  hid_joystick_device_event(event->jdevice.which, event->type == SDL_JOYDEVICEADDED);
499  break;
500  case SDL_JOYBUTTONDOWN:
501  case SDL_JOYBUTTONUP:
502  hid_joystick_button_event(event->type == SDL_JOYBUTTONDOWN);
503  break;
504  case SDL_JOYHATMOTION:
505  hid_joystick_hat_event(event->jhat.value);
506  break;
507  case SDL_JOYAXISMOTION:
508  if (event->jaxis.axis < 2)
509  hid_joystick_motion_event(event->jaxis.axis, event->jaxis.value);
510  break;
511  case SDL_MOUSEMOTION:
512  if (is_mouse_grab())
513  hid_mouse_motion_event(event->motion.xrel, event->motion.yrel);
514  break;
515  case SDL_MOUSEBUTTONDOWN:
516  case SDL_MOUSEBUTTONUP:
517  if (is_mouse_grab())
518  hid_mouse_button_event(event->button.button, event->type == SDL_MOUSEBUTTONDOWN);
519  else
520  emu_callback_key(-2, 0, event->type == SDL_MOUSEBUTTONDOWN, event->button.button);
521  break;
522 #ifdef CONFIG_KBD_ALSO_TEXTEDITING_SDL_CALLBACK
523  case SDL_TEXTEDITING:
524  emu_callback_key_texteditng_sdl(&event->edit);
525  break;
526 #endif
527 #ifdef CONFIG_KBD_ALSO_TEXTINPUT_SDL_CALLBACK
528  case SDL_TEXTINPUT:
529  emu_callback_key_textinput_sdl(&event->text);
530  break;
531 #endif
532  default:
533  handled = 0;
534  break;
535  }
536  return handled;
537 }
