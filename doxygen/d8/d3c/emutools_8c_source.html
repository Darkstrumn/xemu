127 static inline int get_elapsed_time ( Uint64 t_old, Uint64 *t_new, time_t *store_unix_time )
128 {
129 #ifdef XEMU_OLD_TIMING
130 #define __TIMING_METHOD_DESC "gettimeofday"

132  gettimeofday(&tv, NULL);
133  if (store_unix_time)
134  *store_unix_time = tv.tv_sec;
135  *t_new = tv.tv_sec * 1000000UL + tv.tv_usec;
136  return *t_new - t_old;

139  if (store_unix_time)

265 static inline void do_sleep ( int td )
266 {
267 #ifdef XEMU_ARCH_HTML
268 #define __SLEEP_METHOD_DESC "emscripten_set_main_loop_timing"
269  // Note: even if td is zero (or negative ...) give at least a little time for the browser
270  // do not detect the our JS script as a run-away one, suggesting to kill ...
271  // Note: this is not an actual sleep, we can't do that in JS. Instead of just "throttle"
272  // the "frequency" our main loop is called. This also means, that do_sleep should be
273  // called as last in case of emscripten target, since this does not sleep at all for real,
274  // unlike the other sleep methods for non-js targets.
275  emscripten_set_main_loop_timing(EM_TIMING_SETTIMEOUT, td > 999 ? td / 1000 : 1);
276 #elif defined(XEMU_SLEEP_IS_SDL_DELAY)
277 #define __SLEEP_METHOD_DESC "SDL_Delay"
278  if (td > 0)
279  SDL_Delay(td / 1000);
280 #elif defined(XEMU_SLEEP_IS_USLEEP)
281 #define __SLEEP_METHOD_DESC "usleep"
282  if (td > 0)
283  usleep(td);
284 #elif defined(XEMU_SLEEP_IS_NANOSLEEP)
285 #define __SLEEP_METHOD_DESC "nanosleep"
286  struct timespec req, rem;
287  if (td <= 0)
288  return;
289  td *= 1000;
290  req.tv_sec = td / 1000000000UL;
291  req.tv_nsec = td % 1000000000UL;
292  for (;;) {
293  if (nanosleep(&req, &rem)) {
294  if (errno == EINTR) {
295  req.tv_sec = rem.tv_sec;
296  req.tv_nsec = rem.tv_nsec;
297  } else
298  FATAL("nanosleep() returned with unhandable error");
299  } else
300  return;
301  }
302 #else
303 #error "No SLEEP method is defined with XEMU_SLEEP_IS_* macros!"
304 #endif
305 }

309 /* Should be called regularly (eg on each screen global update), this function
310  tries to keep the emulation speed near to real-time of the emulated machine.
311  It's assumed that this function is called at least at every 0.1 sec or even
312  more frequently ... Ideally it should be used at 25Hz rate (for PAL full TV
313  frame) or 50Hz (for PAL half frame).
314  Input: td_em: time in microseconds would be need on the REAL (emulated)
315  machine to do the task, since the last call of this function! */
316 void xemu_timekeeping_delay ( int td_em )
317 {
318  int td, td_pc;
319  time_t old_unix_time = unix_time;
320  Uint64 et_new;
321  td_pc = get_elapsed_time(et_old, &et_new, NULL); // get realtime since last call in microseconds
322  if (td_pc < 0) return; // time goes backwards? maybe time was modified on the host computer. Skip this delay cycle
323  td = td_em - td_pc; // the time difference (+X = emu is faster (than emulated machine) - real time emulation, -X = emu is slower - real time emulation is not possible)
324  td_balancer += td;
325  if (td_balancer > 0)
326  do_sleep(td_balancer);
327  /* Purpose:
328  * get the real time spent sleeping (sleep is not an exact science on a multitask OS)
329  * also this will get the starter time for the next frame
330  */
331  // calculate real time slept
332  td = get_elapsed_time(et_new, &et_old, &unix_time);
333  seconds_timer_trigger = (unix_time != old_unix_time);
334  if (seconds_timer_trigger) {
335  snprintf(window_title_buffer_end, 32, " [%d%% %d%%] %s %s",
336  ((td_em_ALL < td_pc_ALL) && td_pc_ALL) ? td_em_ALL * 100 / td_pc_ALL : 100,
337  td_em_ALL ? (td_pc_ALL * 100 / td_em_ALL) : -1,
338  window_title_custom_addon ? window_title_custom_addon : "running",
339  window_title_info_addon ? window_title_info_addon : ""
340  );
341  SDL_SetWindowTitle(sdl_win, window_title_buffer);
342  td_pc_ALL = td_pc;
343  td_em_ALL = td_em;
344  } else {
345  td_pc_ALL += td_pc;
346  td_em_ALL += td_em;
347  }
348  if (td < 0) return; // invalid, sleep was about for _minus_ time? eh, give me that time machine, dude! :)
349  // Balancing real and wanted sleep time on long run
350  // Insane big values are forgotten, maybe emulator was stopped, or something like that
351  td_balancer -= td;
352  if (td_balancer > 1000000)
353  td_balancer = 0;
354  else if (td_balancer < -1000000) {
355  // reaching this means the anomaly above, OR simply the fact, that emulator is too slow to emulate in real time!
356  td_balancer = 0;
357  }
358 }

433 void xemu_pre_init ( const char *app_organization, const char *app_name, const char *slogan )
434 {
435 #ifdef XEMU_ARCH_UNIX
436  signal(SIGHUP, SIG_IGN); // ignore SIGHUP, eg closing the terminal Xemu was started from ...
437 #endif
438 #ifdef XEMU_ARCH_HTML

441  MKDIR(emscripten_sdl_base_dir);

443  sdl_pref_dir = (void*)emscripten_sdl_base_dir;
444  sdl_inst_dir = (void*)emscripten_sdl_base_dir;
445  // In case of emscripten we do all the SDL init here!!!
446  // Please note: with emscripten, you can't use SDL_INIT_TIMER and SDL_INIT_HAPTIC subsystems it seems, it will
447  // give error on SDL_Init (I can understand with timer, as it would require multithreading)
448  if (SDL_Init(SDL_INIT_EVERYTHING & ~(SDL_INIT_TIMER | SDL_INIT_HAPTIC)))
449  FATAL("Cannot initialize SDL: %s", SDL_GetError());
450  atexit(shutdown_emulator);
451 #else
452  char *p;
453  sysconsole_open();
454  if (chatty_xemu)
455  xemu_dump_version(stdout, slogan);
456  // Initialize SDL with no subsystems
457  // This is needed, because SDL_GetPrefPath and co. are not safe on every platforms without it.
458  // But we DO want to use *before* the real SDL_Init, as the configuration file may describe
459  // parameters on the init itself, but to read the config file we must know where it is.
460  // You see, chicken & egg problem. We try to work this around with the solution to do the
461  // minimal init for the path info functions, then we shuts SDL down, and the "real" SDL_Init
462  // with subsystems etc will happen later!
463  if (SDL_Init(0))
464  FATAL("Cannot pre-initialize SDL without any subsystem: %s", SDL_GetError());
465  atexit(shutdown_emulator);
466  p = SDL_GetBasePath();
467  if (p) {
468  sdl_base_dir = xemu_strdup(p);
469  SDL_free(p);
470  } else
471  FATAL("Cannot query SDL base directory: %s", SDL_GetError());
472  p = GetHackedPrefDir(sdl_base_dir, app_name);
473  if (!p)
474  p = SDL_GetPrefPath(app_organization, app_name);
475  if (p) {
476  sdl_pref_dir = xemu_strdup(p); // we are too careful: I can't be sure the used SQL_Quit messes up the allocated buffer, so we "clone" it

478  sprintf(sdl_inst_dir, "%s%s" DIRSEP_STR, p, INSTALL_DIRECTORY_ENTRY_NAME);
479  SDL_free(p);
480  } else
481  FATAL("Cannot query SDL preference directory: %s", SDL_GetError());
482 #endif
483  xemu_app_org = xemu_strdup(app_organization);
484  xemu_app_name = xemu_strdup(app_name);
485 }

743  after render is done. Then pixels can be written but especially in locked_texture
744  mode, you CAN'T read the previous frame pixels back! Also then you need to update
745  *ALL* pixels of the texture before calling xemu_update_screen() func at the end!

750 {
