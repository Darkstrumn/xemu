41 #define ICR_CHECK() \
42  do { \
43  if (cia->ICRmask & cia->ICRdata & 31) { \
44  cia->ICRdata |= 128; \
45  if (!cia->intLevel) { cia->intLevel = 1; cia->setint(1); DEBUG("%s IRQ to 1" NL, cia->name); } \
46  } else { \
47  cia->ICRdata &= 127; \
48  if ( cia->intLevel) { cia->intLevel = 0; cia->setint(0); DEBUG("%s IRQ to 0" NL, cia->name); } \
49  } \
50  } while(0)
51 #define ICR_SET(mask) \
52  do { \
53  cia->ICRdata |= (mask) & 31; \
54  DEBUG("%s ICR set to data $%02X, mask is $%02X" NL, cia->name, cia->ICRdata, cia->ICRmask); \
55  ICR_CHECK(); \
56  } while(0)
57 #define ICR_CLEAR(mask) \
58  do { \
59  cia->ICRdata &= 255 - (mask); \
60  ICR_CHECK(); \
61  } while(0)

69  cia->regWritten[a] = -1;

123 Uint8 cia_read ( struct Cia6526 *cia, int addr )
124 {
125  Uint8 temp;
126  switch (addr & 0xF) {
127  case 0: // reg#0: port A data
128  // Note: actually - it seems - CIA always reads the pins even if output DDR is set
129  // We leave the implementation details of the emulator targets to think about this ...
130  return cia->ina();
131  case 1: // reg#1: port B data
132  // Note: see the note above, with port A
133  return cia->inb();
134  case 2: // reg#2: DDR A
135  return cia->DDRA;
136  case 3: // reg#3: DDR B
137  return cia->DDRB;
138  case 4: // reg#4: timer A counter low
139  return cia->TCA & 0xFF;
140  case 5: // reg#5: timer A counter high
141  return cia->TCA >> 8;
142  case 6: // reg#6: timer B counter low
143  return cia->TCB & 0xFF;
144  case 7: // reg#7: timer B counter high
145  return cia->TCB >> 8;
146  case 8: // reg#8: TOD 10ths
147  return cia->tod[0];
148  case 9: // reg#9: TOD sec
149  return cia->tod[1];
150  case 10: // reg#A: TOD min
151  return cia->tod[2];
152  case 11: // reg#B: TOD hours + PM
153  return cia->tod[3];
154  case 12: // reg#C: SP
155  return cia->SDR;
156  case 13: // reg#D: ICR data
157  temp = cia->ICRdata;
158  cia->ICRdata = 0;
159  ICR_CHECK();
160  return temp;
161  case 14: // reg#E: CRA
162  return cia->CRA;
163  case 15: // reg#F: CRB
164  return cia->CRB;
165  }
166  return 0; // to make GCC happy :-/
167 }

171 void cia_write ( struct Cia6526 *cia, int addr, Uint8 data )
172 {
173  switch (addr & 0xF) {
174  case 0: // reg#0: port A
175  // Note: we leave the details for the emulator targets to handle
176  // the situation what is really output in relation of the DDR register too ...
177  cia->PRA = data;
178  cia->outa(data);
179  break;
180  case 1: // reg#1: port B
181  // Note: see the note above with Port A
182  cia->PRB = data;
183  cia->outb(data);
184  break;
185  case 2: // reg#2: DDR A
186  // Note: we "notify" the emulation target that DDR changed with re-output the value
187  // Note: see the notes with registers 0, 1 on output/DDR related thing
188  if (cia->DDRA != data) {
189  cia->DDRA = data;
190  cia->outa(cia->PRA);
191  }
192  break;
193  case 3: // reg#3: DDR B
194  // Note: see the note above with DDR of port A
195  if (cia->DDRB != data) {
196  cia->DDRB = data;
197  cia->outb(cia->PRB);
198  }
199  break;
200  case 4: // reg#4: timer A latch low
201  cia->TLAL = data;
202  break;
203  case 5: // reg#5: timer A latch high
204  cia->TLAH = data;
205  if (!(cia->CRA & 1) || (cia->CRA & 16))
206  cia->TCA = cia->TLAL | (cia->TLAH << 8);
207  break;
208  case 6: // reg#6: timer B latch low
209  cia->TLBL = data;
210  break;
211  case 7: // reg#7: timer B latch high
212  cia->TLBH = data;
213  if (!(cia->CRB & 1) || (cia->CRB & 16))
214  cia->TCB = cia->TLBL | (cia->TLBH << 8);
215  break;
216  case 8: // reg#8: TOD 10ths
217  if (cia->CRB & 128)
218  cia->todAlarm[0] = data & 15;
219  else
220  cia->tod[0] = data & 15;
221  break;
222  case 9: // reg#9: TOD sec
223  if (cia->CRB & 128)
224  cia->todAlarm[1] = data & 127;
225  else
226  cia->tod[1] = data & 127;
227  break;
228  case 10: // reg#A: TOD min
229  if (cia->CRB & 128)
230  cia->todAlarm[2] = data & 127;
231  else
232  cia->tod[2] = data & 127;
233  break;
234  case 11: // reg#B: TOD hours + PM
235  if (cia->CRB & 128)
236  cia->todAlarm[3] = data;
237  else
238  cia->tod[3] = data;
239  break;
240  case 12: // reg#C: SP
241  cia->SDR = data;
242  if (cia->CRA & 64)
243  cia->outsr(data);
244  break;
245  case 13: // reg#D: ICR mask
246  if (data & 128)
247  cia->ICRmask |= data & 31;
248  else
249  cia->ICRmask &= 255 - (data & 31);
250  DEBUG("%s set ICRmask to %02X with byte of %02X" NL, cia->name, cia->ICRmask, data);
251  ICR_CHECK();
252  break;
253  case 14: // reg#E: CRA
254  cia->CRA = data;
255  if (data & 16) { // force load?
256  cia->CRA &= 255 - 16; // strobe bit, force to be on zero
257  cia->TCA = cia->TLAL | (cia->TLAH << 8);
258  }
259  break;
260  case 15: // reg#F: CRB
261  cia->CRB = data;
262  if (data & 16) { // force load?
263  cia->CRB &= 255 - 16; // storbe bit, force to be on zero
264  cia->TCB = cia->TLBL | (cia->TLBH << 8);
265  }
266  break;
267  }
268  cia->regWritten[addr] = data;
269 }

272 void cia_ugly_tod_updater ( struct Cia6526 *cia, struct tm *t )
273 {
274  // Ugly CIA trick to maintain realtime TOD in CIAs :)
275  // FIXME: of course, that's simple crazy, not in sync with emu, no "stopping" clock on read, no setting etc ...
276  cia->tod[0] = 0;
277  cia->tod[1] = ((t->tm_sec / 10) << 4) | (t->tm_sec % 10);
278  cia->tod[2] = ((t->tm_min / 10) << 4) | (t->tm_min % 10);
279  cia->tod[3] = ((t->tm_hour / 10) << 4) | (t->tm_hour % 10);
280 }

283 void cia_tick ( struct Cia6526 *cia, int ticks )
284 {
285  /* Timer A */
286  if (cia->CRA & 1) {
287  cia->TCA -= ticks;
288  if (cia->TCA <= 0) {
289  DEBUG("%s timer-A expired!" NL, cia->name);
290  ICR_SET(1);
291  cia->TCA = cia->TLAL | (cia->TLAH << 8);
292  if (cia->CRA & 8)
293  cia->CRA &= 254; // one shot mode: reset bit 1 (timer stop)
294  }
295  }
296  /* Timer B */
297  if (cia->CRB & 1) {
298  cia->TCB -= ticks;
299  if (cia->TCB <= 0) {
300  DEBUG("%s timer-B expired!" NL, cia->name);
301  ICR_SET(2);
302  cia->TCB = cia->TLBL | (cia->TLBH << 8);
303  if (cia->CRB & 8)
304  cia->CRB &= 254; // one shot mode: reset bit 1 (timer stop)
305  }
306  }
307  /* TOD stuffs are ignored ;-/ */
308 }

334 int cia_snapshot_load_state ( const struct xemu_snapshot_definition_st *def, struct xemu_snapshot_block_st *block )
335 {
336  Uint8 buffer[SNAPSHOT_CIA_BLOCK_SIZE];
337  struct Cia6526 *cia = (struct Cia6526 *)def->user_data;
338  int a;
339  if (block->block_version != SNAPSHOT_CIA_BLOCK_VERSION || block->sub_counter || block->sub_size != sizeof buffer)
340  RETURN_XSNAPERR_USER("Bad CIA block syntax");
341  a = xemusnap_read_file(buffer, sizeof buffer);
342  if (a) return a;
343  /* loading state ... */
344  for (a = 0; a < 16; a++) {
345  cia_write(cia, a, (cia->regWritten[a] = buffer[a + 16] ? -1 : buffer[a]));
346  }
347  cia->TCA = (int)P_AS_BE32(buffer + 128);
348  cia->TCB = (int)P_AS_BE32(buffer + 132);
349  cia->intLevel = (int)P_AS_BE32(buffer + 136);
350  cia->CRA = buffer[140];
351  cia->CRB = buffer[141];
352  cia->ICRdata = buffer[142];
353  cia->ICRmask = buffer[143];
354  ICR_CHECK();
355  cia->setint(cia->intLevel); // just to be sure ...
356  cia->outa(cia->PRA);
357  cia->outb(cia->PRB);
358  return 0;
359 }
