35 static int xemugtkgui_iteration ( void )
36 {
37  if (XEMU_UNLIKELY(_gtkgui_active && is_xemugui_ok)) {
38  int n = 0;
39  while (gtk_events_pending()) {
40  gtk_main_iteration();
41  n++;
42  }
43  if (n > 0)
44  DEBUGGUI("GUI: GTK used %d iterations." NL, n);
45  if (n == 0 && _gtkgui_active == 2) {
46  _gtkgui_active = 0;
47  DEBUGGUI("GUI: stopping GTK3 iterator" NL);
48  }
49  return n;
50  } else
51  return 0;
52 }

57 static int xemugtkgui_init ( void )
58 {
59  is_xemugui_ok = 0;
60  _gtkgui_popup_is_open = 0;
61  _gtkgui_active = 0;
62  if (!gtk_init_check(NULL, NULL)) {
63  DEBUGGUI("GUI: GTK3 cannot be initialized, no GUI is available ..." NL);
64  ERROR_WINDOW("Cannot initialize GTK");
65  return 1;
66  }
67  is_xemugui_ok = 1;
68  xemugtkmenu.num_of_menus = 0;
69  _gtkgui_active = 2;
70  int n = xemugtkgui_iteration();
71  DEBUGGUI("GUI: GTK3 initialized, %d iterations." NL, n); // consume possible pending (if any?) GTK stuffs after initialization - maybe not needed at all?
72  atexit(xemugtkgui_shutdown);
73  return 0;
74 }

91 {
92  if (!is_xemugui_ok)

94  _gtkgui_active = 1;

96  GtkWidget *dialog = gtk_file_chooser_dialog_new(dialog_title,
97  NULL, // parent window! We have NO GTK parent window, and it seems GTK is lame, that dumps warning, but we can't avoid this ...
98  action,
99  "_Cancel",
100  GTK_RESPONSE_CANCEL,
101  "_Open",
102  GTK_RESPONSE_ACCEPT,
103  NULL
104  );
105  if (default_dir)
106  gtk_file_chooser_set_current_folder(GTK_FILE_CHOOSER(dialog), default_dir);
107  *selected = '\0';
108  gint res = gtk_dialog_run(GTK_DIALOG(dialog));
109  if (res == GTK_RESPONSE_ACCEPT) {
110  char *filename;
111  filename = gtk_file_chooser_get_filename(GTK_FILE_CHOOSER(dialog));
112  if (strlen(filename) < path_max_size) {
113  strcpy(selected, filename);
114  store_dir_from_file_selection(default_dir, filename, dialog_mode);
115  } else
116  res = GTK_RESPONSE_CANCEL;
117  g_free(filename);
118  }
119  gtk_widget_destroy(dialog);
120  xemu_drop_events();
121  _gtkgui_active = 2;
122  return res != GTK_RESPONSE_ACCEPT;
123 }

149 static GtkWidget *_gtkgui_recursive_menu_builder ( const struct menu_st desc[] )
150 {
151  if (xemugtkmenu.num_of_menus >= XEMUGUI_MAX_SUBMENUS) {
152  ERROR_WINDOW("Too much submenus");
153  goto PROBLEM;
154  }
155  GtkWidget *menu = gtk_menu_new();
156  xemugtkmenu.menus[xemugtkmenu.num_of_menus++] = menu;
157  for (int a = 0; desc[a].name; a++) {
158  if (!desc[a].handler || !desc[a].name) {
159  DEBUGPRINT("GUI: invalid meny entry found, skipping it" NL);
160  continue;
161  }
162  GtkWidget *item = NULL;
163  int type = desc[a].type;
164  switch (type & 0xFF) {
165  case XEMUGUI_MENUID_SUBMENU:
166  item = gtk_menu_item_new_with_label(desc[a].name);
167  if (!item)
168  goto PROBLEM;
169  gtk_menu_shell_append(GTK_MENU_SHELL(menu), item);
170  GtkWidget *submenu = _gtkgui_recursive_menu_builder(desc[a].handler); // who does not like recursion, seriously? :-)
171  if (!submenu)
172  goto PROBLEM;
173  gtk_menu_item_set_submenu(GTK_MENU_ITEM(item), submenu);
174  break;
175  case XEMUGUI_MENUID_CALLABLE:
176  if ((type & XEMUGUI_MENUFLAG_QUERYBACK)) {
177  DEBUGGUI("GUI: query-back for \"%s\"" NL, desc[a].name);
178  ((xemugui_callback_t)(desc[a].handler))(&desc[a], &type);
179  }
180  item = gtk_menu_item_new_with_label(desc[a].name);
181  if (!item)
182  goto PROBLEM;
183  gtk_menu_shell_append(GTK_MENU_SHELL(menu), item);
184  g_signal_connect_swapped(
185  item,
186  "activate",
187  G_CALLBACK(_gtkgui_callback),
188  (gpointer)&desc[a]
189  );
190  break;
191  }
192  if (item)
193  gtk_widget_show(item);
194  }
195  gtk_widget_show(menu);
196  return menu;
197 PROBLEM:
198  xemugtkmenu.problem = 1;
199  return NULL;
200 }

216 static void _gtkgui_disappear ( const char *signal_name )
217 {
218  // Basically we don't want to waste CPU time in GTK for the iterator (ie event loop) if you
219  // don't need it. So when pop-up menu deactivated, this callback is called, which sets _gtkgui_active to 2.
220  // this is a signal for the iterator to stop itself if there was no events processed once at its run
221  DEBUGGUI("GUI: requesting iterator stop on g-signal \"%s\"" NL, signal_name);
222  // Unfortunately, we can't destroy widget here, since it makes callback never executed :( Oh main, GTK is hard :-O
223  _gtkgui_active = 2;
224  _gtkgui_popup_is_open = 0;
225 }

257 static int xemugtkgui_popup ( const struct menu_st desc[] )
258 {
259  static const char disappear_signal[] = "deactivate";
260  if (_gtkgui_popup_is_open) {
261  DEBUGGUI("GUI: trying to enter popup mode, while we're already there" NL);
262  return 0;
263  }
264  if (!is_xemugui_ok /*|| gtk_menu_problem*/) {
265  DEBUGGUI("GUI: MENU: GUI was not successfully initialized yet, or GTK menu creation problem occured back to the first attempt" NL);
266  return 1;
267  }
268  _gtkgui_active = 1;
269  GtkWidget *menu = _gtkgui_create_menu(desc);
270  if (!menu) {
271  _gtkgui_active = 0;
272  ERROR_WINDOW("Could not build GTK pop-up menu :(");
273  return 1;
274  }
275  // this signal will be fired, to request iterator there, since the menu should be run "in the background" unlike the file selector window ...
276  g_signal_connect_swapped(menu, disappear_signal, G_CALLBACK(_gtkgui_disappear), (gpointer)disappear_signal);
277  // FIXME: yes, I should use gtk_menu_popup_at_pointer() as this function is deprecated already!
278  // however that function does not work since the event parameter being NULL causes not to display anything
279  // I guess it's because there is no "parent" for the pop-up menu, as this is not a GTK app, just using the pop-up ...
280 #ifdef USE_OLD_GTK_POPUP
281  gtk_menu_popup(
282  GTK_MENU(menu),
283  NULL, NULL, NULL,
284  NULL, 0, 0
285  );
286 #else
287  GdkEventButton fake_event = {
288  .type = GDK_BUTTON_PRESS,
289  //.window = gdk_x11_window_foreign_new_for_display(
290  //.window = gdk_x11_window_lookup_for_display(
291  // gdk_display_get_default(),
292  // NULL
293  //)
294  .window = super_ugly_gtk_hack()
295  };
296  gtk_menu_popup_at_pointer(GTK_MENU(menu), (GdkEvent*)&fake_event);
297 #endif
298  _gtkgui_popup_is_open = 1;
299  xemugtkgui_iteration();
300  return 0;
301 }
