111  DEBUG("FDC: %s sector track=%d sector=%d side=%d @ PC=$%04X" NL, opdesc, track, sector, side, cpu65.old_pc);

129 static void read_sector ( void )
130 {
131  int error;
132  if (have_disk) {
133  int offset = calc_offset("reading");
134  error = (offset < 0);
135  if (!error) {
136  Uint8 read_buffer[512];
137  error = fdc_cb_rd_sec(read_buffer, offset);
138  if (error)
139  DEBUG("FDC: sector read-callback returned with error!" NL);
140  else {
141  int n;
142  DEBUG("FDC: sector has been read." NL);
143  for (n = 0; n < 512; n++) {
144  cache[cache_p_fdc] = read_buffer[n];
145  cache_p_fdc = (cache_p_fdc + 1) & 511;
146  }
147  }
148  }
149  } else {
150  error = 1;
151  DEBUG("FDC: no disk in drive" NL);
152  if (warn_disk) {
153  INFO_WINDOW("No disk image was given or can be loaded!");
154  warn_disk = 0;
155  }
156  }
157  if (error) {
158  status_a |= 16; // record not found ...
159  status_b &= 15; // RDREQ/WTREQ/RUN/GATE off
160  } else {
161  status_a |= 64; // DRQ, for buffered reads indicates that FDC accessed the buffer last (DRQ should be cleared by CPU reads, set by FDC access)
162  status_b |= 128 | 32 ; // RDREQ, RUN to set! (important: ROM waits for RDREQ to be high after issued read operation, also we must clear it SOME time later ...)
163  status_a &= ~32; // clear EQ, missing this in general freezes DOS as it usually does not expect to have EQ set even before the first data read [??]
164  }
165 }

214 void fdc_write_reg ( int addr, Uint8 data )
215 {
216  DEBUG("FDC: writing register %d with data $%02X" NL, addr, data);
217 #ifdef DEBUG_FOR_PAUL
218  printf("PAUL: FDC register %d will be written now, data is $%02X buffer pointer is %d now" NL, addr, data, cache_p_cpu);
219 #endif
220  switch (addr) {
221  case 0:
222 #if 0
223  if (status_a & 128) {
224  DEBUG("FDC: WARN: trying to write control register ($%02X) while FDC is busy." NL, data);
225  return;
226  }
227 #endif
228  control = data;
229  status_a |= 128; // writing control register also causes to set the BUSY flag for some time ...
230  if (curcmd == -1)
231  curcmd = 0x100; // "virtual" command, by writing the control register
232  drive = data & 7; // drive selection
233  head_side = (data >> 3) & 1;
234  if (drive)
235  DEBUG("FDC: WARN: not drive-0 is selected: %d!" NL, drive);
236  else
237  DEBUG("FDC: great, drive-0 is selected" NL);
238  if (data & 16) {
239  DEBUG("FDC: WARN: SWAP bit emulation is experimental!" NL);
240  if (warn_swap_bit) {
241  DEBUGPRINT("FDC: warning: SWAP bit emulation is experimental! There will be no further warnings on this." NL);
242  warn_swap_bit = 0;
243  }
244  swap_mask = 0x100;
245  } else
246  swap_mask = 0;
247  break;
248  case 1:
249  // FIXME: I still don't what happens if a running operation (ie BUSY) is in progress and new command is tried to be given to the F011 FDC
250  DEBUG("FDC: command=$%02X (lower bits: $%X)" NL, data & 0xF8, data & 7);
251  if ((status_a & 128) && ((data & 0xF8))) { // if BUSY, and command is not the cancel command ..
252  DEBUG("FDC: WARN: trying to issue another command ($%02X) while the previous ($%02X) is running." NL, data, cmd);
253  return;
254  }
255  cmd = data;
256  curcmd = data;
257  status_a |= 128; // simulate busy status ...
258  status_b &= 255 - 2; // turn IRQ flag OFF
259  status_a &= 255 - (4 + 8 + 16); // turn RNF/CRC/LOST flags OFF
260  status_b &= 255 - (128 + 64); // also turn RDREQ and WRREQ OFF [IS THIS REALLY NEEDED?]
261  break;
262  case 4:
263  track = data;
264  break;
265  case 5:
266  sector = data;
267  break;
268  case 6:
269  side = data;
270  break;
271  case 7:
272  // FIXME: this algorithm do not "enforce" the internals of F011, just if the software comply the rules otherwise ......
273  status_a &= ~64; // clear DRQ
274  //status_b &= 127; // turn RDREQ off after the first access, this is somewhat incorrect :-P
275  if (status_a & 32) // if EQ was already set and another byte passed ---> LOST
276  status_a |= 4; // LOST!!!! but probably incorrect, since no new read is done by FDC since then just "wrapping" the read data, LOST remains till next command!
277  cache[cache_p_cpu ^ swap_mask] = data;
278  cache_p_cpu = (cache_p_cpu + 1) & 511;
279  // always check EQ-situation _after_ the operation! Since initially they are same and it won't be loved by C65 DOS at all, honoured with a freeze
280  if (cache_p_cpu == cache_p_fdc)
281  status_a |= 32; // turn EQ on
282  else
283  status_a &= ~32; // turn EQ off
284  break;
285  // TODO: write DATA register (7) [only for writing it is needed anyway]
286  case 8:
287  dskclock = data;
288  break;
289  case 9:
290  step = data;
291  break;
292  }
293  if (status_a & 128) {
294  emulate_busy = 10;
295  execute_command(); // do it NOW!!!! With this setting now: there is no even BUSY state, everything happens within one OPC... seems to work still. Real F011 won't do this surely
296  }
297 #ifdef DEBUG_FOR_PAUL
298  printf("PAUL: FDC register %d has been written, data was $%02X buffer pointer is %d now" NL, addr, data, cache_p_cpu);
299 #endif
300 }

307 static void execute_command ( void )
308 {
309 #ifdef DEBUG_FOR_PAUL
310  printf("PAUL: issuing FDC command $%02X pointer was %d" NL, cmd, cache_p_cpu);
311 #endif
312  status_a &= 127; // turn BUSY flag OFF
313  status_b |= 2; // turn IRQ flag ON
314  if (control & 128)
315  INFO_WINDOW("Sorry, FDC-IRQ is not supported yet, by FDC emulation!");
316  if (curcmd < 0)
317  return; // no cmd was given?!
318  if (curcmd > 0xFF)
319  return; // only control register was written, not the command register
320 #ifdef SOME_DEBUG
321  printf("Command: $%02X" NL, cmd);
322 #endif
323  switch (cmd & 0xF8) { // high 5 bits of the command ...
324  case 0x40: // read sector
325  //status_a |= 16; // record not found for testing ...
326  status_b |= 128; // RDREQ: if it's not here, you won't get a READY. prompt!
327  //status_b |= 32; // RUN?!
328  status_a |= 64; // set DRQ
329  status_a &= (255 - 32); // clear EQ
330  //status_a |= 32; // set EQ?!
331  //cache_p_cpu = cache_p_fdc; // yayy .... If it's not here we can't get READY. prompt!!
332  read_sector();
333  //cache_p_drive = (cache_p_drive + BLOCK_SIZE) & 511;
335 #ifdef SOME_DEBUG
336  printf("READ: cache_p_cpu=%d / cache_p_fdc=%d drive_selected=%d" NL, cache_p_cpu, cache_p_fdc, drive);
337 #endif
338  DEBUG("FDC: READ: head_track=%d need_track=%d head_side=%d need_side=%d need_sector=%d drive_selected=%d" NL,
339  head_track, track, head_side, side, sector, drive
340  );
341  break;
342  case 0x80: // write sector
343  if (!(status_a & 2)) { // if not write protected ....
344  status_a |= 64; // set DRQ
345  status_a &= (255 - 32); // clear EQ
346  write_sector();
347 #ifdef SOME_DEBUG
348  printf("WRITE: cache_p_cpu=%d / cache_p_fdc=%d drive_selected=%d" NL, cache_p_cpu, cache_p_fdc, drive);
349 #endif
350  DEBUG("FDC: WRITE: head_track=%d need_track=%d head_side=%d need_side=%d need_sector=%d drive_selected=%d" NL,
351  head_track, track, head_side, side, sector, drive
352  );
353  } else {
354 #ifdef SOME_DEBUG
355  printf("Sorry, write protected stuff ..." NL);
356 #endif
357  }
358  break;
359  case 0x10: // head step out or no step
360  if (!(cmd & 4)) { // if only not TIME operation, which does not step!
361  if (head_track)
362  head_track--;
363  if (!head_track)
364  status_a |= 1; // track 0 flag
365  DEBUG("FDC: head position = %d" NL, head_track);
366  }
367  break;
368  case 0x18: // head step in
369  if (head_track < 128)
370  head_track++;
371  DEBUG("FDC: head position = %d" NL, head_track);
372  status_a &= 0xFE; // track 0 flag off
373  break;
374  case 0x20: // motor spin up
375  control |= 32;
376  status_a |= 16; // according to the specification, RNF bit should be set at the end of the operation
377  break;
378  case 0x00: // cancel running command?? NOTE: also if low bit is 1: clear pointer!
379  // Note: there was a typo in my previous versions ... to have break HERE! So pointer reset never executed actually ...... :-@
380  // It seems C65 DOS uses this, without this, the pointer maintaince is bad, and soon it will freeze on EQ check somewhere!
381  // That's why I had to introduce some odd workarounds to reset pointers etc manually, which is NOT what F011 would do, I guess!
382  if (cmd & 1) {
383 #ifdef SOME_DEBUG
384  printf("BEFORE pointer reset: cache_p_cpu=%d cache_p_fdc=%d" NL, cache_p_cpu, cache_p_fdc);
385 #endif
386  cache_p_cpu = 0;
387  cache_p_fdc = 0;
388  DEBUG("FDC: WARN: resetting cache pointers" NL);
389  //status_a |= 32; // turn EQ on
390  status_a &= 255 - 64; // turn DRQ off
391  status_b &= 127; // turn RDREQ off
392 
393  }
394  break;
395  default:
396  DEBUG("FDC: WARN: unknown comand: $%02X" NL, cmd);
397  //status_a &= 127; // well, not a valid command, revoke busy status ...
398  break;
399  }
400  curcmd = -1;
401 #ifdef DEBUG_FOR_PAUL
402  printf("PAUL: issued FDC command $%02X pointer is %d" NL, cmd, cache_p_cpu);
403 #endif
404 }

410 Uint8 fdc_read_reg ( int addr )
411 {
412  Uint8 result;
413 #ifdef DEBUG_FOR_PAUL
414  printf("PAUL: FDC register %d will be read, buffer pointer is %d now" NL, addr, cache_p_cpu);
415 #endif
416  /* Emulate BUSY timing, with a very bad manner: ie, decrement a counter on each register read to give some time to wait.
417  FIXME: not sure if it's needed and what happen if C65 DOS gots "instant" operations done without BUSY ever set ... Not a real happening, but it can be with my primitive emulation :)
418  Won't work if DOS is IRQ driven ... */
419  if (status_a & 128) { // check the BUSY flag
420  // Note: this is may not used at all, check the end of write reg func!
421  if (emulate_busy > 0)
422  emulate_busy--;
423  if (emulate_busy <= 0) {
424 #ifdef SOME_DEBUG
425  printf("Delayed command execution!!!" NL);
426 #endif
427  execute_command(); // execute the command only now for real ... (it will also turn BUSY flag - bit 7 - OFF in status_a)
428  }
429  }
430  switch (addr) {
431  case 0:
432  result = control;
433  break;
434  case 1:
435  result = cmd;
436  break;
437  case 2: // STATUS register A
438  result = drive ? 0 : status_a; // FIXME: Not sure: if no drive 0 is selected, other status is shown, ie every drives should have their own statuses?
439  //result = status_a;
440  break;
441  case 3: // STATUS register B
442  result = drive ? 0 : status_b; // FIXME: Not sure: if no drive 0 is selected, other status is shown, ie every drives should have their own statuses?
443  //result = status_b;
444  status_b &= ~64; // turn WTREQ off, as it seems CPU noticed with reading this register, that is was the case for a while. Somewhat incorrect implementation ... :-/
445  break;
446  case 4:
447  result = track;
448  break;
449  case 5:
450  result = sector;
451  break;
452  case 6:
453  result = side;
454  break;
455  case 7:
456  // FIXME: this algorithm do not "enforce" the internals of F011, just if the software comply the rules otherwise ......
457  status_a &= ~64; // clear DRQ
458  status_b &= 127; // turn RDREQ off after the first access, this is somewhat incorrect :-P
459  if (status_a & 32) // if EQ was already set and another byte passed ---> LOST
460  status_a |= 4; // LOST!!!! but probably incorrect, since no new read is done by FDC since then just "wrapping" the read data, LOST remains till next command!
461  result = cache[cache_p_cpu ^ swap_mask];
462  cache_p_cpu = (cache_p_cpu + 1) & 511;
463  // always check EQ-situation _after_ the operation! Since initially they are same and it won't be loved by C65 DOS at all, honoured with a freeze
464  if (cache_p_cpu == cache_p_fdc)
465  status_a |= 32; // turn EQ on
466  else
467  status_a &= ~32; // turn EQ off
468  break;
469  case 8:
470  result = dskclock;
471  break;
472  case 9:
473  result = step;
474  break;
475  case 10:
476  result = 0; // No "protection code" ...
477  break;
478  default:
479  result = 0xFF;
480  break;
481  }
482  DEBUG("FDC: reading register %d result is $%02X" NL, addr, result);
483 #ifdef DEBUG_FOR_PAUL
484  printf("PAUL: FDC register %d has been read, result was $%02X buffer pointer is %d now" NL, addr, result, cache_p_cpu);
485 #endif
486  return result;
487 }
