243 int xemu_basic_to_text_malloc ( Uint8 **buffer, int output_super_limit, const Uint8 *prg, int real_addr, const Uint8 *prg_limit, int basic_dialect, int flags )
244 {
245  int size = xemu_basic_to_text(NULL, output_super_limit, prg, real_addr, prg_limit, basic_dialect, flags);
246  if (size < 0) {
247  *buffer = NULL;
248  return size;
249  }
250  *buffer = malloc(size + 1);
251  if (!*buffer) {
252  ERROR_WINDOW("%sCannot allocate memory", ERROR_HEAD);
253  return -1;
254  }
255  size = xemu_basic_to_text(*buffer, size, prg, real_addr, prg_limit, basic_dialect, flags);
256  if (size < 0) {
257  free(*buffer);
258  return size;
259  }
260  buffer[size] = 0;
261  return size;
262 }

265 int xemu_basic_to_text ( Uint8 *output, int output_size, const Uint8 *prg, int real_addr, const Uint8 *prg_limit, int basic_dialect, int flags )
266 {
267  int output_used = 0;
268  char outbuf[256]; // the longest thing here is a single one-time decoded entity. This is waaay too much, but let's play safe ...
269  const char *o = NULL;
270  int num = -1, old_num = -1;
271  int ptr = 0;
272  int context = CONTEXT_BEGIN;
273  int token = 0;
274  for (;;) {
275  if (XEMU_UNLIKELY(prg > prg_limit)) {
276  ERROR_WINDOW("%sProgram flows outside of the allowed memory area after line %d", ERROR_HEAD, old_num);
277  return -1;
278  }
279  Uint8 c = *prg++;
280  real_addr++;
281  if (context == CONTEXT_BEGIN) {
282  ptr = c;
283  context = CONTEXT_POINTER1; // CONTEXT_BEGIN is same as CONTEXT_POINTER0, just occures once, so at the first line, we simply skip CONTEXT_POINTER0 since they're the same in a way ;-P Hard to explain
284  if ((flags & BASIC_TO_TEXT_FLAG_TEX))
285  o = tex_head;
286  else
287  continue;
288  } else if (context == CONTEXT_POINTER0) {
289  printf("ptr_to_here=%d real_addr=%d, delta=%d\n", ptr, real_addr, real_addr-ptr);
290  if (ptr != real_addr - 1) {
291  ERROR_WINDOW("%sBad BASIC chaining near line %d", ERROR_HEAD, num);
292  return -1;
293  }
294  ptr = c;
295  context = CONTEXT_POINTER1;
296  continue;
297  } else if (context == CONTEXT_POINTER1) {
298  ptr += c << 8;
299  if (XEMU_UNLIKELY(!ptr)) {
300  if ((flags & BASIC_TO_TEXT_FLAG_TEX)) {
301  context = CONTEXT_END;
302  o = tex_end;
303  } else
304  break; // END OF BASIC PROGRAM ;)
305  } else {
306  context = CONTEXT_LINENO0;
307  continue;
308  }
309  } else if (context == CONTEXT_LINENO0) {
310  context = CONTEXT_LINENO1;
311  old_num = num;
312  num = c;
313  continue;
314  } else if (context == CONTEXT_LINENO1) {
315  num += c << 8;
316  context = CONTEXT_NORMAL;
317  sprintf(outbuf, "%5d ", num);
318  o = outbuf;
319  } else if (context == CONTEXT_NORMAL && (c >= 0x80 || token != 0)) {
320  if ((c == 0xCE || c == 0xFE) && token == 0) {
321  token = c << 8;
322  } else {
323  token += c;
324  for (int a = 0 ;; a++) {
325  if (basic_tokens[a].token == token) {
326  o = basic_tokens[a].str;
327  break; // token found, "o" holds what it must be output
328  } else if (basic_tokens[a].token == 0) {
329  ERROR_WINDOW("%sUnknown token $%04X in line %d", ERROR_HEAD, token, num);
330  return -1;
331  }
332  }
333  if (token == 0x8F) // REM
334  context = CONTEXT_REM;
335  token = 0;
336  }
337  } else if (c == 0) { // end of basic line
338  if (XEMU_UNLIKELY(token)) { // TODO: maybe it's useless to check? see the condition above ...
339  ERROR_WINDOW("%sUnfinished extended token sequence in line %d", ERROR_HEAD, num);
340  return -1;
341  }
342  context = CONTEXT_POINTER0;
343  o = end_of_line;
344  } else {
345  if (c == '"') {
346  if (context == CONTEXT_NORMAL)
347  context = CONTEXT_QUOTED;
348  else if (context == CONTEXT_QUOTED)
349  context = CONTEXT_NORMAL;
350  }
351  if (c >= 'A' && c <= 'Z')
352  c += 'a' - 'A';
353  else if (c >= ('A' | 0x80) && c <= ('Z' | 0x80))
354  c &= 0x7F;
355  outbuf[0] = c;
356  outbuf[1] = 0;
357  o = outbuf;
358  }
359  // flush output in a safe way
360  if (XEMU_UNLIKELY(!o)) { // just for code errors, should not happen!
361  ERROR_WINDOW("%sInternal error, uninitialized output buffer in line %d", ERROR_HEAD, num);
362  return -1;
363  }
364  while (*o) {
365  if (XEMU_UNLIKELY(output_used >= output_size)) {
366  ERROR_WINDOW("%sOutput buffer is too small for this program in line %d", ERROR_HEAD, num);
367  return -1;
368  }
369  output_used++;
370  if (output)
371  *output++ = *o++;
372  else
373  o++; // this mode (when output is NULL) is just for calculating the size of output
374  }
375  if (XEMU_UNLIKELY(context == CONTEXT_END))
376  break; // END OF BASIC PROGRAM :)
377  o = NULL; // just for code errors, should not happen!
378  }
379  return output_used; // END OF BASIC PROGRAM :) Give back the size of output buffer used
380 }
