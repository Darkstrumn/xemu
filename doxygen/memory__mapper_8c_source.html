290 static const struct m65_memory_map_st m65_memory_map[] = {
291  // 126K chip-RAM (last 2K is not availbale because it's colour RAM), with physical zero page excluded (this is because it needs the CPU port handled with different handler!)

330 static void phys_addr_decoder ( int phys, int slot, int hint_slot )
331 {
332  const struct m65_memory_map_st *p;
333  phys &= 0xFFFFF00; // we map only at 256 bytes boundaries!!!! It also helps to wrap around 28 bit M65 addresses TODO/FIXME: is this correct behaviour?
334  if (mem_page_phys[slot] == phys) // kind of "mapping cache" for the given cache slot
335  return; // skip, if the slot already contains info on the current physical address
336  mem_page_phys[slot] = phys;
337  // tricky part: if hint_slot is non-negative, it's used for "contiunity" information related to this slot,
338  // ie check, if the current map request can be fit into the region already mapped by hint_slot, then no
339  // need for the search loop. hint_slot can be any slot, but logically it's sane to be used when the given
340  // hint_slot is "likely" to have some contiunity with the slot given by "slot" otherwise it's just makes
341  // thing worse. If not used, hint_slot should be negative to skip this feature. hint_slot can be even same
342  // as "slot" if you need a "moving" mapping in a "caching" slot, ie DMA-aux access functions, etc.
343  if (hint_slot >= 0 && mem_page_refp[hint_slot]->end < 0xFFFFFFF) { // FIXME there was a serious bug here, takinking invalid mem slot for anything after hinting that
344  p = mem_page_refp[hint_slot];
345  if (phys >= p->start && phys <= p->end) {
346 #ifdef DEBUGMEM
347  DEBUGMEM("MEM: PHYS-MAP: slot#$%03X: slot hint TAKEN :)" NL, slot);
348 #endif
349  goto found;
350  }
351  }
352  // Scan the memory map, as not found "cached" result on the same slot, or by the hinting slot
353  for (p = m65_memory_map; phys < p->start || phys > p->end; p++)
354  ;
355 found:
356  mem_page_rd_o[slot] = mem_page_wr_o[slot] = phys - p->start;
357  mem_page_rd_f[slot] = p->rd_f;
358  mem_page_wr_f[slot] = p->wr_f;
359  //if (p->rd_f == invalid_mem_reader)
360  // FATAL("Invalid memory region is tried to be mapped to slot $%X for phys addr $%X" NL, slot, phys);
361  mem_page_refp[slot] = p;
362 #ifdef DEBUGMEM
363  DEBUGMEM("MEM: PHYS-MAP: slot#$%03X: phys = $%X mapped (area: $%X-$%X, rd_o=%X, wr_o=%X) [hint slot was: %03X]" NL,
364  slot,
365  phys,
366  p->start, p->end,
367  mem_page_rd_o[slot], mem_page_wr_o[slot],
368  hint_slot
369  );
370 #endif
371 }

640 // must be called when VIC-III register $D030 is written, with the written value exactly
641 void memory_set_vic3_rom_mapping ( Uint8 value )
642 {
643  // D030 regiser of VIC-III is:
644  // 7 6 5 4 3 2 1 0
645  // | ROM | CROM | ROM | ROM | ROM | PAL | EXT | CRAM |
646  // | @E000 | @9000 | @C000 | @A000 | @8000 | | SYNC | @DC00 |
647  if (in_hypervisor)
648  value = 0; // in hypervisor, VIC-III ROM banking should *not* work (newer M65 change)
649  else
650  value &= VIC3_ROM_MASK_8000 | VIC3_ROM_MASK_A000 | VIC3_ROM_MASK_C000 | VIC3_ROM_MASK_E000; // only keep bits we're interested in
651  if (value != memcfg_vic3_rom_mapping_last) { // only do, if there was a change
652  Uint8 change = memcfg_vic3_rom_mapping_last ^ value; // change mask, bits have 1 only if there was a change
653  DEBUG("MEM: VIC-III ROM mapping change $%02X -> %02X" NL, memcfg_vic3_rom_mapping_last, value);
654  memcfg_vic3_rom_mapping_last = value; // don't forget to store the current state for next check!
655  // now check bits changed in ROM mapping
656  if (change & VIC3_ROM_MASK_8000)
657  apply_memory_config_8000_to_9FFF();
658  if (change & VIC3_ROM_MASK_A000)
659  apply_memory_config_A000_to_BFFF();
660  if (change & VIC3_ROM_MASK_C000)
661  apply_memory_config_C000_to_CFFF();
662  if (change & VIC3_ROM_MASK_E000)
663  apply_memory_config_E000_to_FFFF();
664  }
665 }

742 // This implements the MAP opcode, ie "AUG" in case of 65CE02, which was re-defined to "MAP" in C65's CPU
743 // M65's extension to select "MB" (ie: megabyte slice, which wraps within!) is supported as well
744 void cpu65_do_aug_callback ( void )
745 {
746  /* 7 6 5 4 3 2 1 0 BIT
747  +-------+-------+-------+-------+-------+-------+-------+-------+
748  | LOWER | LOWER | LOWER | LOWER | LOWER | LOWER | LOWER | LOWER | A
749  | OFF15 | OFF14 | OFF13 | OFF12 | OFF11 | OFF10 | OFF9 | OFF8 |
750  +-------+-------+-------+-------+-------+-------+-------+-------+
751  | MAP | MAP | MAP | MAP | LOWER | LOWER | LOWER | LOWER | X
752  | BLK3 | BLK2 | BLK1 | BLK0 | OFF19 | OFF18 | OFF17 | OFF16 |
753  +-------+-------+-------+-------+-------+-------+-------+-------+
754  | UPPER | UPPER | UPPER | UPPER | UPPER | UPPER | UPPER | UPPER | Y
755  | OFF15 | OFF14 | OFF13 | OFF12 | OFF11 | OFF10 | OFF9 | OFF8 |
756  +-------+-------+-------+-------+-------+-------+-------+-------+
757  | MAP | MAP | MAP | MAP | UPPER | UPPER | UPPER | UPPER | Z
758  | BLK7 | BLK6 | BLK5 | BLK4 | OFF19 | OFF18 | OFF17 | OFF16 |
759  +-------+-------+-------+-------+-------+-------+-------+-------+
760  -- C65GS extension: Set the MegaByte register for low and high mobies
761  -- so that we can address all 256MB of RAM.
762  if reg_x = x"0f" then
763  reg_mb_low <= reg_a;
764  end if;
765  if reg_z = x"0f" then
766  reg_mb_high <= reg_y;
767  end if; */
768  cpu65.cpu_inhibit_interrupts = 1; // disable interrupts till the next "EOM" (ie: NOP) opcode
769  DEBUG("CPU: MAP opcode, input A=$%02X X=$%02X Y=$%02X Z=$%02X" NL, cpu65.a, cpu65.x, cpu65.y, cpu65.z);
770  map_offset_low = (cpu65.a << 8) | ((cpu65.x & 15) << 16); // offset of lower half (blocks 0-3)
771  map_offset_high = (cpu65.y << 8) | ((cpu65.z & 15) << 16); // offset of higher half (blocks 4-7)
772  map_mask = (cpu65.z & 0xF0) | ( cpu65.x >> 4); // "is mapped" mask for blocks (1 bit for each)
773  // M65 specific "MB" (megabyte) selector "mode":
774  if (cpu65.x == 0x0F)
775  map_megabyte_low = (int)cpu65.a << 20;
776  if (cpu65.z == 0x0F)
777  map_megabyte_high = (int)cpu65.y << 20;
778  DEBUG("MEM: applying new memory configuration because of MAP CPU opcode" NL);
779  DEBUG("LOW -OFFSET = $%03X, MB = $%02X" NL, map_offset_low , map_megabyte_low >> 20);
780  DEBUG("HIGH-OFFSET = $%03X, MB = $%02X" NL, map_offset_high, map_megabyte_high >> 20);
781  DEBUG("MASK = $%02X" NL, map_mask);
782  memory_set_do_map();
783 }
